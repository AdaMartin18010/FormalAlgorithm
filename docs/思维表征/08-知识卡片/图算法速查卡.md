# 图算法速查卡

## 图的存储方式

```
╔═════════════════════════════════════════════════════════════════════════════╗
║                      邻接矩阵 vs 邻接表                                      ║
╠═════════════════════════════════════════════════════════════════════════════╣
║                                                                             ║
║  【邻接矩阵】                                                                ║
║  int[][] graph = new int[n][n];              // n个顶点                     ║
║  graph[u][v] = weight;                       // u到v的边权                  ║
║                                                                             ║
║  优点: O(1)查边，适合稠密图                                                  ║
║  缺点: O(n²)空间，遍历邻接点O(n)                                             ║
║                                                                             ║
║  【邻接表】                                                                  ║
║  List<int[]>[] graph = new ArrayList[n];     // Java                        ║
║  for (int i = 0; i < n; i++) graph[i] = new ArrayList<>();                  ║
║  graph[u].add(new int[]{v, weight});         // 添加边 u->v                 ║
║                                                                             ║
║  优点: O(V+E)空间，遍历邻接点O(度数)                                         ║
║  缺点: 查边需要遍历                                                          ║
╚═════════════════════════════════════════════════════════════════════════════╝
```

## 图遍历算法

### DFS (深度优先搜索)

```
╔═════════════════════════════════════════════════════════════════════════════╗
║                      DFS 模板                                                ║
╠═════════════════════════════════════════════════════════════════════════════╣
║                                                                             ║
║  【递归版本】                                                                ║
║  boolean[] visited = new boolean[n];                                        ║
║                                                                             ║
║  void dfs(int u) {                                                          ║
║      visited[u] = true;                                                     ║
║      for (int v : graph[u]) {                                               ║
║          if (!visited[v]) {                                                 ║
║              dfs(v);                                                        ║
║          }                                                                  ║
║      }                                                                      ║
║  }                                                                          ║
║                                                                             ║
║  【非递归版本】                                                              ║
║  void dfs(int start) {                                                      ║
║      Stack<Integer> stack = new Stack<>();                                  ║
║      boolean[] visited = new boolean[n];                                    ║
║      stack.push(start);                                                     ║
║      while (!stack.isEmpty()) {                                             ║
║          int u = stack.pop();                                               ║
║          if (visited[u]) continue;                                          ║
║          visited[u] = true;                                                 ║
║          for (int v : graph[u]) {                                           ║
║              if (!visited[v]) stack.push(v);                                ║
║          }                                                                  ║
║      }                                                                      ║
║  }                                                                          ║
║                                                                             ║
║  【时间复杂度】O(V + E)                                                      ║
╚═════════════════════════════════════════════════════════════════════════════╝
```

### BFS (广度优先搜索)

```
╔═════════════════════════════════════════════════════════════════════════════╗
║                      BFS 模板 (无权图最短路径)                               ║
╠═════════════════════════════════════════════════════════════════════════════╣
║                                                                             ║
║  int[] bfs(int start) {                                                     ║
║      int[] dist = new int[n];                                               ║
║      Arrays.fill(dist, -1);                  // -1表示未访问                 ║
║      Queue<Integer> queue = new LinkedList<>();                             ║
║      dist[start] = 0;                                                       ║
║      queue.offer(start);                                                    ║
║      while (!queue.isEmpty()) {                                             ║
║          int u = queue.poll();                                              ║
║          for (int v : graph[u]) {                                           ║
║              if (dist[v] == -1) {                                           ║
║                  dist[v] = dist[u] + 1;                                     ║
║                  queue.offer(v);                                            ║
║              }                                                              ║
║          }                                                                  ║
║      }                                                                      ║
║      return dist;                                                           ║
║  }                                                                          ║
║                                                                             ║
║  【时间复杂度】O(V + E)                                                      ║
╚═════════════════════════════════════════════════════════════════════════════╝
```

## 最短路径算法

```
╔═════════════════════════════════════════════════════════════════════════════╗
║                   最短路径算法对比                                           ║
╠═════════════════════════════════════════════════════════════════════════════╣
║  ┌──────────────┬────────────────┬───────────────┬───────────────────────┐  ║
║  │    算法      │    适用条件    │    时间复杂度  │        特点           │  ║
║  ├──────────────┼────────────────┼───────────────┼───────────────────────┤  ║
║  │   Dijkstra   │  非负权边      │  O((V+E)logV) │ 单源最短路，贪心      │  ║
║  │ Bellman-Ford │  可负权边      │    O(VE)      │ 可判负环              │  ║
║  │    Floyd     │  全源最短路    │    O(V³)      │ 动态规划，代码短      │  ║
║  └──────────────┴────────────────┴───────────────┴───────────────────────┘  ║
╚═════════════════════════════════════════════════════════════════════════════╝

╔═════════════════════════════════════════════════════════════════════════════╗
║                   Dijkstra 算法 (堆优化)                                     ║
╠═════════════════════════════════════════════════════════════════════════════╣
║                                                                             ║
║  int[] dijkstra(int start, List<int[]>[] graph) {                           ║
║      int n = graph.length;                                                  ║
║      int[] dist = new int[n];                                               ║
║      Arrays.fill(dist, Integer.MAX_VALUE);                                  ║
║      dist[start] = 0;                                                       ║
║      PriorityQueue<int[]> pq = new PriorityQueue<>((a, b) -> a[1] - b[1]);  ║
║      pq.offer(new int[]{start, 0});                                         ║
║      while (!pq.isEmpty()) {                                                ║
║          int[] curr = pq.poll();                                            ║
║          int u = curr[0], d = curr[1];                                      ║
║          if (d > dist[u]) continue;                                         ║
║          for (int[] edge : graph[u]) {                                      ║
║              int v = edge[0], w = edge[1];                                  ║
║              if (dist[u] + w < dist[v]) {                                   ║
║                  dist[v] = dist[u] + w;                                     ║
║                  pq.offer(new int[]{v, dist[v]});                           ║
║              }                                                              ║
║          }                                                                  ║
║      }                                                                      ║
║      return dist;                                                           ║
║  }                                                                          ║
║                                                                             ║
║  【注意】不能处理负权边                                                      ║
╚═════════════════════════════════════════════════════════════════════════════╝

╔═════════════════════════════════════════════════════════════════════════════╗
║                   Floyd-Warshall 全源最短路                                  ║
╠═════════════════════════════════════════════════════════════════════════════╣
║                                                                             ║
║  int[][] floyd(int n, int[][] edges) {                                      ║
║      int[][] dist = new int[n][n];                                          ║
║      for (int i = 0; i < n; i++) {                                          ║
║          Arrays.fill(dist[i], INF);                                         ║
║          dist[i][i] = 0;                                                    ║
║      }                                                                      ║
║      for (int[] e : edges) {                                                ║
║          dist[e[0]][e[1]] = e[2];                                           ║
║      }                                                                      ║
║      for (int k = 0; k < n; k++) {                                          ║
║          for (int i = 0; i < n; i++) {                                      ║
║              for (int j = 0; j < n; j++) {                                  ║
║                  if (dist[i][k] + dist[k][j] < dist[i][j]) {                ║
║                      dist[i][j] = dist[i][k] + dist[k][j];                  ║
║                  }                                                          ║
║              }                                                              ║
║          }                                                                  ║
║      }                                                                      ║
║      return dist;                                                           ║
║  }                                                                          ║
╚═════════════════════════════════════════════════════════════════════════════╝
```

## 最小生成树 (MST)

```
╔═════════════════════════════════════════════════════════════════════════════╗
║                   Kruskal 算法                                               ║
╠═════════════════════════════════════════════════════════════════════════════╣
║                                                                             ║
║  int kruskal(int n, int[][] edges) {                                        ║
║      // edges: [u, v, weight]                                               ║
║      Arrays.sort(edges, (a, b) -> a[2] - b[2]);                             ║
║      UnionFind uf = new UnionFind(n);                                       ║
║      int mstWeight = 0, edgesUsed = 0;                                      ║
║      for (int[] e : edges) {                                                ║
║          int u = e[0], v = e[1], w = e[2];                                  ║
║          if (uf.find(u) != uf.find(v)) {                                    ║
║              uf.union(u, v);                                                ║
║              mstWeight += w;                                                ║
║              if (++edgesUsed == n - 1) break;                               ║
║          }                                                                  ║
║      }                                                                      ║
║      return edgesUsed == n - 1 ? mstWeight : -1;                            ║
║  }                                                                          ║
║                                                                             ║
║  时间: O(E log E)   适用: 稀疏图                                             ║
╚═════════════════════════════════════════════════════════════════════════════╝

╔═════════════════════════════════════════════════════════════════════════════╗
║                   Prim 算法                                                  ║
╠═════════════════════════════════════════════════════════════════════════════╣
║                                                                             ║
║  int prim(int n, List<int[]>[] graph) {                                     ║
║      boolean[] visited = new boolean[n];                                    ║
║      PriorityQueue<int[]> pq = new PriorityQueue<>((a, b) -> a[1] - b[1]);  ║
║      pq.offer(new int[]{0, 0});                                             ║
║      int mstWeight = 0, edgesUsed = 0;                                      ║
║      while (!pq.isEmpty() && edgesUsed < n) {                               ║
║          int[] curr = pq.poll();                                            ║
║          int u = curr[0], w = curr[1];                                      ║
║          if (visited[u]) continue;                                          ║
║          visited[u] = true;                                                 ║
║          mstWeight += w;                                                    ║
║          edgesUsed++;                                                       ║
║          for (int[] edge : graph[u]) {                                      ║
║              if (!visited[edge[0]]) pq.offer(edge);                         ║
║          }                                                                  ║
║      }                                                                      ║
║      return edgesUsed == n ? mstWeight : -1;                                ║
║  }                                                                          ║
║                                                                             ║
║  时间: O((V+E)logV)   适用: 稠密图                                           ║
╚═════════════════════════════════════════════════════════════════════════════╝
```

## 拓扑排序

```
╔═════════════════════════════════════════════════════════════════════════════╗
║                   拓扑排序 (Kahn算法)                                        ║
╠═════════════════════════════════════════════════════════════════════════════╣
║                                                                             ║
║  List<Integer> topologicalSort(int n, List<Integer>[] graph) {              ║
║      int[] inDegree = new int[n];                                           ║
║      for (int u = 0; u < n; u++) {                                          ║
║          for (int v : graph[u]) inDegree[v]++;                              ║
║      }                                                                      ║
║      Queue<Integer> queue = new LinkedList<>();                             ║
║      for (int i = 0; i < n; i++) {                                          ║
║          if (inDegree[i] == 0) queue.offer(i);                              ║
║      }                                                                      ║
║      List<Integer> result = new ArrayList<>();                              ║
║      while (!queue.isEmpty()) {                                             ║
║          int u = queue.poll();                                              ║
║          result.add(u);                                                     ║
║          for (int v : graph[u]) {                                           ║
║              if (--inDegree[v] == 0) queue.offer(v);                        ║
║          }                                                                  ║
║      }                                                                      ║
║      return result.size() == n ? result : null;   // null表示有环           ║
║  }                                                                          ║
║                                                                             ║
║  【时间复杂度】O(V + E)                                                      ║
╚═════════════════════════════════════════════════════════════════════════════╝
```

## 并查集 (Union-Find)

```
╔═════════════════════════════════════════════════════════════════════════════╗
║                   并查集模板                                                 ║
╠═════════════════════════════════════════════════════════════════════════════╣
║                                                                             ║
║  class UnionFind {                                                          ║
║      int[] parent, rank;                                                    ║
║                                                                             ║
║      UnionFind(int n) {                                                     ║
║          parent = new int[n];                                               ║
║          rank = new int[n];                                                 ║
║          for (int i = 0; i < n; i++) parent[i] = i;                         ║
║      }                                                                      ║
║                                                                             ║
║      int find(int x) {                                                      ║
║          if (parent[x] != x) {                                              ║
║              parent[x] = find(parent[x]);    // 路径压缩                    ║
║          }                                                                  ║
║          return parent[x];                                                  ║
║      }                                                                      ║
║                                                                             ║
║      boolean union(int x, int y) {                                          ║
║          int px = find(x), py = find(y);                                    ║
║          if (px == py) return false;                                        ║
║          // 按秩合并                                                        ║
║          if (rank[px] < rank[py]) parent[px] = py;                          ║
║          else if (rank[px] > rank[py]) parent[py] = px;                     ║
║          else {                                                             ║
║              parent[py] = px;                                               ║
║              rank[px]++;                                                    ║
║          }                                                                  ║
║          return true;                                                       ║
║      }                                                                      ║
║  }                                                                          ║
║                                                                             ║
║  【时间复杂度】接近 O(alpha(n))，即反阿克曼函数，实际可视为 O(1)              ║
╚═════════════════════════════════════════════════════════════════════════════╝
```

## 强连通分量 (SCC) - Tarjan算法

```
╔═════════════════════════════════════════════════════════════════════════════╗
║                   Tarjan 求强连通分量                                        ║
╠═════════════════════════════════════════════════════════════════════════════╣
║                                                                             ║
║  int[] dfn = new int[n], low = new int[n];                                  ║
║  boolean[] inStack = new boolean[n];                                        ║
║  Stack<Integer> stack = new Stack<>();                                      ║
║  int timestamp = 0, sccCount = 0;                                           ║
║  int[] sccId = new int[n];                                                  ║
║                                                                             ║
║  void tarjan(int u, List<Integer>[] graph) {                                ║
║      dfn[u] = low[u] = ++timestamp;                                         ║
║      stack.push(u);                                                         ║
║      inStack[u] = true;                                                     ║
║                                                                             ║
║      for (int v : graph[u]) {                                               ║
║          if (dfn[v] == 0) {                                                 ║
║              tarjan(v, graph);                                              ║
║              low[u] = Math.min(low[u], low[v]);                             ║
║          } else if (inStack[v]) {                                           ║
║              low[u] = Math.min(low[u], dfn[v]);                             ║
║          }                                                                  ║
║      }                                                                      ║
║                                                                             ║
║      if (dfn[u] == low[u]) {               // 找到一个SCC                   ║
║          sccCount++;                                                        ║
║          int v;                                                             ║
║          do {                                                               ║
║              v = stack.pop();                                               ║
║              inStack[v] = false;                                            ║
║              sccId[v] = sccCount;                                           ║
║          } while (v != u);                                                  ║
║      }                                                                      ║
║  }                                                                          ║
╚═════════════════════════════════════════════════════════════════════════════╝
```
