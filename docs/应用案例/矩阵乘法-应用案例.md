```rust
/// 标准矩阵乘法
pub fn standard_multiply(a: &[Vec<f64>], b: &[Vec<f64>]) -> Vec<Vec<f64>> {
    let n = a.len();
    let m = b[0].len();
    let p = b.len();

    let mut result = vec![vec![0.0; m]; n];

    for i in 0..n {
        for j in 0..m {
            for k in 0..p {
                result[i][j] += a[i][k] * b[k][j];
            }
        }
    }

    result
}

/// 分治矩阵求逆（使用LU分解）
pub fn matrix_inverse(a: &[Vec<f64>]) -> Option<Vec<Vec<f64>>> {
    let n = a.len();

    // 创建增广矩阵 [A | I]
    let mut aug: Vec<Vec<f64>> = (0..n)
        .map(|i| {
            let mut row = a[i].clone();
            row.extend((0..n).map(|j| if i == j { 1.0 } else { 0.0 }));
            row
        })
        .collect();

    // 高斯消元
    for i in 0..n {
        // 找主元
        let mut max_row = i;
        for k in (i + 1)..n {
            if aug[k][i].abs() > aug[max_row][i].abs() {
                max_row = k;
            }
        }

        if aug[max_row][i].abs() < 1e-10 {
            return None;  // 奇异矩阵
        }

        aug.swap(i, max_row);

        // 归一化
        let pivot = aug[i][i];
        for j in i..(2 * n) {
            aug[i][j] /= pivot;
        }

        // 消元
        for k in 0..n {
            if k != i {
                let factor = aug[k][i];
                for j in i..(2 * n) {
                    aug[k][j] -= factor * aug[i][j];
                }
            }
        }
    }

    // 提取逆矩阵
    Some((0..n).map(|i| aug[i][n..].to_vec()).collect())
}
```

### 复杂度分析

| 算法 | 时间复杂度 | 适用场景 |
|-----|-----------|---------|
| 标准 | $O(n^3)$ | 小矩阵、缓存友好 |
| Strassen | $O(n^{2.81})$ | 大矩阵 ($n > 100$) |
| Coppersmith-Winograd | $O(n^{2.37})$ | 理论最优，实际不用 |
