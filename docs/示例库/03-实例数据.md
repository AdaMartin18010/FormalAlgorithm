## 1. 实例分级体系

### 1.1 分级标准

| 级别 | 规模范围 | 主要用途 | 验证方式 | 存储格式 |
|------|----------|----------|----------|----------|
| **Small** | n ≤ 10 | 手工追踪、教学演示、算法理解 | 人工验证 | `.json`, `.txt` |
| **Medium** | 10 < n ≤ 1000 | 单元测试、功能验证、基准测试 | 程序验证 | `.json`, `.csv` |
| **Large** | 1000 < n ≤ 10⁶ | 性能测试、压力测试、扩展性测试 | 抽样验证 | `.bin`, `.gz` |
| **Extreme** | n > 10⁶ | 极限测试、大数据算法验证 | 统计验证 | `.bin`, `.parquet` |

### 1.2 数据类型分类

```markdown
数据特性标签:
- 随机 (random): 均匀随机分布
- 有序 (sorted): 升序排列
- 逆序 (reverse): 降序排列
- 部分有序 (partial): 基本有序，少量错位
- 重复 (duplicate): 包含大量重复元素
- 稀疏 (sparse): 大部分为零/空
- 密集 (dense): 完全填充
- 极端值 (extreme): 包含最大/最小边界值
```

### 1.3 文件命名规范

```
[类别]_[类型]_[规模]_[序号].[格式]

示例:
- sorting_random_small_001.json
- graph_dense_medium_010.csv
- dp_knapsack_large_005.json
```

## 8. 数据生成脚本

### 8.1 Python生成脚本

```python
import json
import random
import os
from typing import List, Dict, Any

class InstanceGenerator:
    """测试实例生成器"""

    @staticmethod
    def generate_sorting_instance(n: int, instance_type: str = "random",
                                   seed: int = None) -> Dict[str, Any]:
        """生成排序实例"""
        if seed is not None:
            random.seed(seed)

        if instance_type == "random":
            arr = [random.randint(0, 1000) for _ in range(n)]
        elif instance_type == "sorted":
            arr = list(range(n))
        elif instance_type == "reverse":
            arr = list(range(n-1, -1, -1))
        elif instance_type == "partial":
            arr = list(range(n))
            # 随机交换5%的元素
            for _ in range(max(1, n // 20)):
                i, j = random.sample(range(n), 2)
                arr[i], arr[j] = arr[j], arr[i]
        elif instance_type == "duplicate":
            arr = [random.randint(1, max(2, n // 10)) for _ in range(n)]
        else:
            raise ValueError(f"Unknown type: {instance_type}")

        return {
            "metadata": {
                "type": "sorting",
                "size": "small" if n <= 10 else ("medium" if n <= 1000 else "large"),
                "n": n,
                "instance_type": instance_type,
                "seed": seed
            },
            "input": {"array": arr},
            "expected": {"sorted_array": sorted(arr)}
        }

    @staticmethod
    def generate_graph_instance(n: int, m: int = None,
                                 graph_type: str = "random",
                                 weighted: bool = False,
                                 seed: int = None) -> Dict[str, Any]:
        """生成图实例"""
        if seed is not None:
            random.seed(seed)

        if graph_type == "random":
            max_edges = n * (n - 1) // 2
            m = m or min(n * 3, max_edges)
            edges = set()
            while len(edges) < m:
                u, v = random.sample(range(n), 2)
                if u > v:
                    u, v = v, u
                edges.add((u, v))
            edges = list(edges)
        elif graph_type == "path":
            edges = [(i, i+1) for i in range(n-1)]
        elif graph_type == "complete":
            edges = [(i, j) for i in range(n) for j in range(i+1, n)]
        elif graph_type == "star":
            edges = [(0, i) for i in range(1, n)]
        else:
            raise ValueError(f"Unknown type: {graph_type}")

        if weighted:
            edges = [(u, v, random.randint(1, 100)) for u, v in edges]

        return {
            "metadata": {
                "type": "graph",
                "size": "small" if n <= 10 else ("medium" if n <= 1000 else "large"),
                "n_vertices": n,
                "n_edges": len(edges),
                "graph_type": graph_type,
                "weighted": weighted,
                "seed": seed
            },
            "input": {
                "vertices": list(range(n)),
                "edges": edges
            }
        }

    @staticmethod
    def save_instance(instance: Dict, directory: str, filename: str):
        """保存实例到文件"""
        os.makedirs(directory, exist_ok=True)
        filepath = os.path.join(directory, filename)
        with open(filepath, 'w', encoding='utf-8') as f:
            json.dump(instance, f, indent=2, ensure_ascii=False)
        return filepath


def generate_all_instances():
    """生成所有标准实例"""
    generator = InstanceGenerator()

    # Small sorting instances
    sorting_configs = [
        (5, "random", 42, "sorting_random_small_001.json"),
        (8, "sorted", 43, "sorting_sorted_small_001.json"),
        (8, "reverse", 44, "sorting_reverse_small_001.json"),
        (10, "partial", 45, "sorting_partial_small_001.json"),
        (10, "duplicate", 46, "sorting_duplicate_small_001.json"),
    ]

    for n, type_, seed, filename in sorting_configs:
        instance = generator.generate_sorting_instance(n, type_, seed)
        generator.save_instance(instance, "instances/sorting/small", filename)

    # Medium sorting instances
    instance = generator.generate_sorting_instance(100, "random", 42)
    generator.save_instance(instance, "instances/sorting/medium",
                           "sorting_random_medium_001.json")

    # Small graph instances
    graph_configs = [
        (5, None, "path", False, 42, "graph_path_small_001.json"),
        (5, None, "complete", False, 43, "graph_complete_small_001.json"),
        (6, None, "random", True, 44, "graph_weighted_small_001.json"),
    ]

    for n, m, type_, weighted, seed, filename in graph_configs:
        instance = generator.generate_graph_instance(n, m, type_, weighted, seed)
        generator.save_instance(instance, "instances/graphs/small", filename)

    print("所有实例生成完成！")


if __name__ == "__main__":
    generate_all_instances()
```

### 8.2 实例验证脚本

```python
import json
import os
from pathlib import Path

def validate_instance(filepath: str) -> bool:
    """验证实例文件格式"""
    try:
        with open(filepath, 'r', encoding='utf-8') as f:
            data = json.load(f)

        # 检查必需字段
        required_fields = ["metadata", "input"]
        for field in required_fields:
            if field not in data:
                print(f"错误: {filepath} 缺少字段 {field}")
                return False

        # 检查元数据
        meta = data["metadata"]
        meta_required = ["type", "size"]
        for field in meta_required:
            if field not in meta:
                print(f"错误: {filepath} 元数据缺少 {field}")
                return False

        return True
    except Exception as e:
        print(f"错误: 验证 {filepath} 时出错: {e}")
        return False

def validate_all_instances(base_dir: str = "instances"):
    """验证所有实例文件"""
    base_path = Path(base_dir)
    all_valid = True

    for filepath in base_path.rglob("*.json"):
        if not validate_instance(str(filepath)):
            all_valid = False

    if all_valid:
        print("所有实例验证通过！")
    else:
        print("存在验证失败的实例！")

    return all_valid

if __name__ == "__main__":
    validate_all_instances()
```
