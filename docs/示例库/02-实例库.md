## 实例生成脚本

### Python生成脚本模板

```python
import json
import random
import os

def generate_sorting_instance(n, min_val, max_val, seed=None, instance_type="random"):
    """生成排序实例"""
    if seed:
        random.seed(seed)

    if instance_type == "random":
        arr = [random.randint(min_val, max_val) for _ in range(n)]
    elif instance_type == "sorted":
        arr = list(range(min_val, min_val + n))
    elif instance_type == "reverse":
        arr = list(range(min_val + n - 1, min_val - 1, -1))
    elif instance_type == "nearly_sorted":
        arr = list(range(min_val, min_val + n))
        # 随机交换5%的元素
        for _ in range(n // 20):
            i, j = random.sample(range(n), 2)
            arr[i], arr[j] = arr[j], arr[i]

    return {
        "metadata": {
            "type": "sorting",
            "size": "small" if n <= 10 else ("medium" if n <= 1000 else "large"),
            "n": n,
            "instance_type": instance_type
        },
        "input": {"array": arr},
        "expected": {"sorted_array": sorted(arr)}
    }

def save_instance(instance, directory, filename):
    """保存实例到文件"""
    os.makedirs(directory, exist_ok=True)
    filepath = os.path.join(directory, filename)
    with open(filepath, 'w') as f:
        json.dump(instance, f, indent=2)
    return filepath

# 生成示例实例
if __name__ == "__main__":
    # Small实例
    for i, (n, type_) in enumerate([
        (5, "random"), (5, "sorted"), (5, "reverse"),
        (8, "nearly_sorted"), (10, "random")
    ], 1):
        instance = generate_sorting_instance(n, 1, 100, seed=42+i, instance_type=type_)
        save_instance(instance, "instances/sorting/small", f"sort_small_{i:03d}.json")

    print("实例生成完成!")
```

## 实例使用指南

### 在Python中加载实例

```python
import json

def load_instance(filepath):
    """加载实例文件"""
    with open(filepath, 'r') as f:
        return json.load(f)

def run_sorting_test(algorithm, instance_path):
    """运行排序算法测试"""
    instance = load_instance(instance_path)
    input_arr = instance['input']['array']
    expected = instance['expected']['sorted_array']

    result = algorithm(input_arr.copy())

    return {
        'passed': result == expected,
        'input_size': len(input_arr),
        'comparisons': getattr(algorithm, 'comparisons', None)
    }
```

### 在Rust中加载实例

```rust
use serde::{Deserialize, Serialize};
use std::fs;

#[derive(Debug, Deserialize)]
struct SortingInstance {
    metadata: Metadata,
    input: Input,
    expected: Expected,
}

#[derive(Debug, Deserialize)]
struct Metadata {
    #[serde(rename = "type")]
    type_: String,
    size: String,
    n: usize,
}

#[derive(Debug, Deserialize)]
struct Input {
    array: Vec<i32>,
}

#[derive(Debug, Deserialize)]
struct Expected {
    #[serde(rename = "sorted_array")]
    sorted_array: Vec<i32>,
}

fn load_instance(path: &str) -> Result<SortingInstance, Box<dyn std::error::Error>> {
    let content = fs::read_to_string(path)?;
    let instance: SortingInstance = serde_json::from_str(&content)?;
    Ok(instance)
}
```
