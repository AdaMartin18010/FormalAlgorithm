```python
def check_missing_citations(document):
    """
    检查文档中未引用的定义和定理
    """
    issues = []

    # 定义模式: **定义 X.X.X** (名称) [...]
    definition_pattern = r'\*\*定义\s+[\d.]+\s*\*\*\s*\([^)]+\)\s*(\[[^\]]*\])?'

    # 定理模式: **定理 X.X.X** (名称) [...]
    theorem_pattern = r'\*\*定理\s+[\d.]+\s*\*\*\s*\([^)]+\)\s*(\[[^\]]*\])?'

    content = read_file(document)

    # 检查定义
    for match in re.finditer(definition_pattern, content):
        citation = match.group(1)
        if not citation or citation == '[]':
            issues.append({
                'type': 'missing_citation',
                'element': 'definition',
                'text': match.group(0),
                'line': get_line_number(content, match.start()),
                'severity': 'error'
            })

    # 检查定理
    for match in re.finditer(theorem_pattern, content):
        citation = match.group(1)
        if not citation or citation == '[]':
            issues.append({
                'type': 'missing_citation',
                'element': 'theorem',
                'text': match.group(0),
                'line': get_line_number(content, match.start()),
                'severity': 'error'
            })

    return issues
```

```python
def validate_citation_format(citation):
    """
    验证引用格式是否符合ACM标准
    """
    # ACM格式检查规则
    rules = [
        {
            'name': '作者格式',
            'pattern': r'[A-Z][a-z]+,\s*[A-Z]\.\s*(?:[A-Z]\.)?',
            'description': '作者名应为 "姓氏, 名字首字母." 格式'
        },
        {
            'name': '年份格式',
            'pattern': r'\(\d{4}\)',
            'description': '年份应括在圆括号中，如 (2022)'
        },
        {
            'name': '页码格式',
            'pattern': r'\d+–\d+',
            'description': '页码范围应使用en-dash (–)'
        },
        {
            'name': 'DOI格式',
            'pattern': r'DOI:\s*10\.\d{4,}/\S+',
            'description': 'DOI格式: DOI: 10.xxxx/xxxxx'
        }
    ]

    errors = []
    for rule in rules:
        if not re.search(rule['pattern'], citation):
            errors.append({
                'rule': rule['name'],
                'description': rule['description']
            })

    return errors
```

```python
def check_broken_links(document, base_path):
    """
    检查文档中的失效链接
    """
    issues = []
    content = read_file(document)

    # Markdown链接模式: [text](url)
    link_pattern = r'\[([^\]]+)\]\(([^)]+)\)'

    for match in re.finditer(link_pattern, content):
        text = match.group(1)
        url = match.group(2)

        # 检查内部链接
        if not url.startswith(('http://', 'https://', 'mailto:')):
            # 相对路径链接
            full_path = resolve_path(base_path, url)
            if not os.path.exists(full_path):
                issues.append({
                    'type': 'broken_link',
                    'text': text,
                    'url': url,
                    'line': get_line_number(content, match.start()),
                    'severity': 'error'
                })
        else:
            # 外部链接 - 可选检查
            if should_check_external():
                if not check_url_accessible(url):
                    issues.append({
                        'type': 'unreachable_external_link',
                        'text': text,
                        'url': url,
                        'line': get_line_number(content, match.start()),
                        'severity': 'warning'
                    })

    return issues
```

```python
def check_document_structure(document):
    """
    检查文档结构是否符合模板
    """
    issues = []
    content = read_file(document)

    # 必需章节
    required_sections = [
        {
            'name': '标题',
            'pattern': r'^#\s+.+',
            'required': True
        },
        {
            'name': '摘要',
            'pattern': r'###\s+摘要|###\s+Executive Summary',
            'required': True
        },
        {
            'name': '术语表',
            'pattern': r'###\s+关键术语|###\s+Glossary',
            'required': True
        },
        {
            'name': '目录',
            'pattern': r'###\s+目录|###\s+Table of Contents',
            'required': True
        },
        {
            'name': '参考文献',
            'pattern': r'###\s+参考文献|###\s+References',
            'required': True
        }
    ]

    for section in required_sections:
        if section['required'] and not re.search(section['pattern'], content, re.MULTILINE):
            issues.append({
                'type': 'missing_section',
                'section': section['name'],
                'message': f"缺少必需章节: {section['name']}",
                'severity': 'error'
            })

    # 检查标题层级
    issues.extend(check_heading_hierarchy(content))

    return issues

def check_heading_hierarchy(content):
    """检查标题层级是否正确"""
    issues = []
    headings = re.findall(r'^(#{1,6})\s+', content, re.MULTILINE)

    prev_level = 0
    for heading in headings:
        level = len(heading)
        if level > prev_level + 1:
            issues.append({
                'type': 'heading_hierarchy',
                'message': f'标题层级跳跃: H{prev_level} -> H{level}',
                'severity': 'warning'
            })
        prev_level = level

    return issues
```

```python
def check_mixed_punctuation(content):
    """检查中英文标点混用"""
    issues = []

    # 英文后接中文标点
    patterns = [
        (r'[a-zA-Z]，', '英文后使用了中文逗号'),
        (r'[a-zA-Z]。', '英文后使用了中文句号'),
        (r'[a-zA-Z]；', '英文后使用了中文分号'),
        (r'[\u4e00-\u9fff],', '中文后使用了英文逗号'),
        (r'[\u4e00-\u9fff]\.', '中文后使用了英文句号（非小数点）'),
    ]

    for pattern, message in patterns:
        for match in re.finditer(pattern, content):
            issues.append({
                'type': 'punctuation',
                'message': message,
                'line': get_line_number(content, match.start()),
                'severity': 'warning'
            })

    return issues
```

```python
def check_duplicate_content(documents):
    """
    检查跨文档的重复内容
    """
    from difflib import SequenceMatcher

    issues = []

    # 提取所有段落
    all_paragraphs = {}
    for doc in documents:
        content = read_file(doc)
        paragraphs = extract_paragraphs(content)
        all_paragraphs[doc] = paragraphs

    # 比较段落相似度
    for doc1, paras1 in all_paragraphs.items():
        for doc2, paras2 in all_paragraphs.items():
            if doc1 \geq doc2:  # 避免重复比较
                continue

            for i, para1 in enumerate(paras1):
                for j, para2 in enumerate(paras2):
                    similarity = SequenceMatcher(None, para1, para2).ratio()
                    if similarity > 0.8:  # 80%相似度阈值
                        issues.append({
                            'type': 'duplicate_content',
                            'document1': doc1,
                            'document2': doc2,
                            'paragraph1_index': i,
                            'paragraph2_index': j,
                            'similarity': similarity,
                            'severity': 'warning' if similarity < 0.95 else 'error'
                        })

    return issues
```

```python
def check_outdated_information(document, reference_date='2025-04-08'):
    """
    检查文档中可能过时的信息
    """
    issues = []
    content = read_file(document)

    # 过时信息模式
    outdated_patterns = [
        {
            'pattern': r'将于\s*(\d{4})\s*年',
            'check': lambda year: int(year) < int(reference_date[:4]),
            'message': '包含已过去的未来时间'
        },
        {
            'pattern': r'最新的.*?(\d{4})\s*年',
            'check': lambda year: int(reference_date[:4]) - int(year) > 3,
            'message': '信息可能已过时（超过3年）'
        },
        {
            'pattern': r'目前|当前|现在',
            'context': lambda text: check_context_for_time(text),
            'message': '相对时间表述，建议改为绝对时间'
        }
    ]

    for pattern in outdated_patterns:
        for match in re.finditer(pattern['pattern'], content):
            if pattern.get('check') and pattern['check'](match.group(1)):
                issues.append({
                    'type': 'outdated_info',
                    'message': pattern['message'],
                    'line': get_line_number(content, match.start()),
                    'severity': 'warning'
                })

    return issues
```

## 4. 集成检查工具

### 4.1 主检查脚本

```python
#!/usr/bin/env python3
"""
文档质量自动检查主脚本
Usage: python quality_checker.py <document_path> [options]
"""

import argparse
import json
import sys
from pathlib import Path

class QualityChecker:
    def __init__(self, config_path='check_config.yaml'):
        self.config = load_config(config_path)
        self.checkers = {
            'citation': CitationChecker(),
            'format': FormatChecker(),
            'content': ContentChecker(),
            'link': LinkChecker(),
            'symbol': SymbolChecker()
        }

    def check_document(self, document_path, checks=None):
        """运行所有检查"""
        if checks is None:
            checks = self.checkers.keys()

        results = {
            'document': str(document_path),
            'check_date': datetime.now().isoformat(),
            'total_issues': 0,
            'issues_by_severity': {'error': 0, 'warning': 0, 'info': 0},
            'issues': []
        }

        content = read_file(document_path)

        for check_name in checks:
            if check_name in self.checkers:
                checker = self.checkers[check_name]
                issues = checker.check(document_path, content)
                results['issues'].extend(issues)

        # 统计
        results['total_issues'] = len(results['issues'])
        for issue in results['issues']:
            severity = issue.get('severity', 'warning')
            results['issues_by_severity'][severity] += 1

        return results

    def check_batch(self, directory, pattern='*.md'):
        """批量检查目录中的文档"""
        all_results = []

        for doc_path in Path(directory).rglob(pattern):
            result = self.check_document(doc_path)
            all_results.append(result)

        return all_results

def main():
    parser = argparse.ArgumentParser(description='文档质量检查工具')
    parser.add_argument('path', help='文档路径或目录')
    parser.add_argument('--batch', '-b', action='store_true', help='批量模式')
    parser.add_argument('--output', '-o', help='输出文件')
    parser.add_argument('--format', '-f', choices=['json', 'markdown', 'html'],
                       default='json', help='输出格式')
    parser.add_argument('--checks', '-c', nargs='+',
                       choices=['citation', 'format', 'content', 'link', 'symbol', 'all'],
                       default=['all'], help='要运行的检查')

    args = parser.parse_args()

    checker = QualityChecker()

    if args.batch:
        results = checker.check_batch(args.path)
    else:
        results = checker.check_document(args.path)

    # 输出结果
    output = format_output(results, args.format)

    if args.output:
        with open(args.output, 'w', encoding='utf-8') as f:
            f.write(output)
    else:
        print(output)

    # 返回退出码
    if isinstance(results, list):
        error_count = sum(r['issues_by_severity']['error'] for r in results)
    else:
        error_count = results['issues_by_severity']['error']

    sys.exit(1 if error_count > 0 else 0)

if __name__ == '__main__':
    main()
```

## 6.2 预提交钩子

### 6.2 预提交钩子

```bash
#!/bin/sh
# .git/hooks/pre-commit

echo "Running document quality checks\cdots"

# 获取待提交的Markdown文件
files=$(git diff --cached --name-only --diff-filter=ACM | grep '\.md$')

if [ -n "$files" ]; then
    for file in $files; do
        result=$(python scripts/quality_checker.py "$file" --format json)
        errors=$(echo "$result" | jq '.issues_by_severity.error')

        if [ "$errors" -gt 0 ]; then
            echo "Error: $file has $errors quality issues."
            echo "Run 'python scripts/quality_checker.py $file' for details."
            exit 1
        fi
    done
fi

echo "Quality checks passed."
exit 0
```
