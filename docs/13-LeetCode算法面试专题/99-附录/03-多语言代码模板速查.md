## 1. 二分查找模板 / Binary Search Template

**核心思想**：每次将搜索区间减半，循环不变式保证答案在区间内。

### Python

```python
# 闭区间模板 [l, r] — 查找精确值
# Closed interval template [l, r] — find exact value
def binary_search(nums, target):
    l, r = 0, len(nums) - 1
    while l <= r:                # 区间非空时继续 / Continue while interval non-empty
        m = l + (r - l) // 2     # 防溢出中点 / Mid avoiding overflow
        if nums[m] == target:
            return m               # 找到目标 / Found target
        elif nums[m] < target:
            l = m + 1              # 目标在右半 / Target in right half
        else:
            r = m - 1              # 目标在左半 / Target in left half
    return -1                      # 未找到 / Not found
```

### Rust

```rust

        } else {
            r = m - 1;               // 目标在左半 / Target in left half
        }
    }
    -1                              // 未找到 / Not found
}
```

## 2. 双指针模板 / Two Pointers Template

**核心思想**：两个指针从两端或同向移动，利用有序性或特定条件缩小区间。

### Python

```python
# 相向双指针 — 两数之和（有序数组）
# Opposite two pointers — Two Sum in sorted array
def two_sum_sorted(nums, target):
    l, r = 0, len(nums) - 1
    while l < r:
        s = nums[l] + nums[r]
        if s == target:
            return [l, r]           # 找到配对 / Found pair
        elif s < target:
            l += 1                  # 和太小，左指针右移 / Sum too small
        else:
            r -= 1                  # 和太大，右指针左移 / Sum too large
    return []
```

### Rust

```rust
// 相向双指针
// Opposite two pointers
pub fn two_sum_sorted(nums: &[i32], target: i32) -> Vec<i32> {
    let (mut l, mut r) = (0usize, nums.len() - 1);
    while l < r {
        let s = nums[l] + nums[r];
        if s == target { return vec![l as i32, r as i32]; }
        else if s < target { l += 1; }  // 和太小 / Sum too small
        else { r -= 1; }                // 和太大 / Sum too large
    }
    vec![]
}
```

### Go

```go

            l++                      // 和太小 / Sum too small
        } else {
            r--                      // 和太大 / Sum too large
        }
    }
    return []int{}
}
```

## 3. 滑动窗口模板 / Sliding Window Template

**核心思想**：维护一个可变大小的窗口，右端扩展，左端收缩，满足特定条件。

### Python

```python
# 可变窗口 — 找满足条件的最短子串
// Variable window — shortest substring satisfying condition
def min_window(s, t):
    from collections import Counter
    need, window = Counter(t), Counter()
    left, valid = 0, 0
    start, length = 0, float('inf')
    for right, ch in enumerate(s):      # 右指针扩展 / Right pointer expands
        window[ch] += 1
        if window[ch] == need[ch]:
            valid += 1
        while valid == len(need):       # 窗口有效，收缩左边界 / Window valid, shrink left
            if right - left + 1 < length:
                start, length = left, right - left + 1
            d = s[left]
            left += 1
            if window[d] == need[d]:
                valid -= 1
            window[d] -= 1
    return "" if length == float('inf') else s[start:start+length]
```

### Rust

```rust
// 滑动窗口框架（伪代码示意）
// Sliding window framework (pseudo-code)
pub fn sliding_window(s: &str) -> String {
    let bytes = s.as_bytes();
    let mut left = 0usize;
    let mut result = String::new();
    for right in 0..bytes.len() {       // 右指针扩展 / Right expands
        // 将 s[right] 加入窗口 / Add s[right] to window
        // update state
        while /* 窗口满足收缩条件 / window satisfies shrink condition */ {
            // 更新结果 / Update result
            // 将 s[left] 移出窗口 / Remove s[left] from window
            left += 1;                   // 左指针收缩 / Left shrinks
        }
    }
    result
}
```

### Go

```go

            left++                       // 左指针收缩 / Left shrinks
        }
    }
    return result
}
```

## 4. DFS/BFS 模板 / DFS & BFS Template

**核心思想**：DFS 用栈/递归深入，BFS 用队列层级扩展。

### Python

```python
# DFS 递归模板
# DFS recursive template
def dfs(node, visited):
    if not node or node in visited:
        return
    visited.add(node)
    process(node)                       # 处理当前节点 / Process current node
    for neighbor in node.neighbors:
        dfs(neighbor, visited)          # 递归访问邻居 / Recurse on neighbors

# BFS 模板
# BFS template
from collections import deque
def bfs(start):
    queue = deque([start])
    visited = {start}
    while queue:
        node = queue.popleft()
        process(node)                   # 处理当前节点 / Process current node
        for neighbor in node.neighbors:
            if neighbor not in visited:
                visited.add(neighbor)
                queue.append(neighbor)  # 入队 / Enqueue
```

### Rust

```rust
// DFS 递归模板
// DFS recursive template
pub fn dfs(node: Option<Rc<RefCell<TreeNode>>>, visited: &mut HashSet<i32>) {
    if let Some(n) = node {
        let val = n.borrow().val;
        if visited.contains(&val) { return; }
        visited.insert(val);
        // process(n)
        dfs(n.borrow().left.clone(), visited);
        dfs(n.borrow().right.clone(), visited);
    }
}

// BFS 模板
// BFS template
use std::collections::VecDeque;
pub fn bfs(start: usize, graph: &Vec<Vec<usize>>) {
    let mut queue = VecDeque::from([start]);
    let mut visited = vec![false; graph.len()];
    visited[start] = true;
    while let Some(node) = queue.pop_front() {
        // process(node)
        for &neighbor in &graph[node] {
            if !visited[neighbor] {
                visited[neighbor] = true;
                queue.push_back(neighbor);  // 入队 / Enqueue
            }
        }
    }
}
```

### Go

```go
// DFS 递归模板
// DFS recursive template
func dfs(node *TreeNode, visited map[int]bool) {
    if node == nil || visited[node.Val] {
        return
    }
    visited[node.Val] = true
    // process(node)
    dfs(node.Left, visited)   // 递归左子树 / Recurse left
    dfs(node.Right, visited)  // 递归右子树 / Recurse right
}

// BFS 模板
// BFS template
func bfs(start int, graph [][]int) {
    queue := []int{start}
    visited := make([]bool, len(graph))
    visited[start] = true
    for len(queue) > 0 {
        node := queue[0]
        queue = queue[1:]
        // process(node)
        for _, neighbor := range graph[node] {
            if !visited[neighbor] {
                visited[neighbor] = true
                queue = append(queue, neighbor)  // 入队 / Enqueue
            }
        }
    }
}
```

## 5. 回溯模板 / Backtracking Template

**核心思想**：递归构建解，不满足条件时回退（剪枝）。

### Python

```python
# 回溯模板 — 全排列/组合/子集
# Backtracking template — Permutations/Combinations/Subsets
def backtrack(path, choices):
    if满足结束条件:                         # End condition met
        result.append(path[:])
        return
    for choice in choices:
        if not is_valid(choice):        # 剪枝 / Pruning
            continue
        path.append(choice)             # 做选择 / Make choice
        backtrack(path, next_choices)   # 递归 / Recurse
        path.pop()                      # 撤销选择 / Undo choice
```

### Rust

```rust
// 回溯模板
// Backtracking template
pub fn backtrack(path: &mut Vec<i32>, choices: &[i32], result: &mut Vec<Vec<i32>>) {
    if /* 满足结束条件 / end condition */ {
        result.push(path.clone());
        return;
    }
    for &choice in choices {
        if !is_valid(choice) { continue; }  // 剪枝 / Prune
        path.push(choice);                   // 做选择 / Choose
        backtrack(path, next_choices, result); // 递归 / Recurse
        path.pop();                          // 撤销 / Unchoose
    }
}
```

### Go

```go
// 回溯模板
// Backtracking template
func backtrack(path *[]int, choices []int, result *[][]int) {
    if /* 满足结束条件 */ {
        tmp := make([]int, len(*path))
        copy(tmp, *path)
        *result = append(*result, tmp)
        return
    }
    for _, choice := range choices {
        if !isValid(choice) { continue }  // 剪枝 / Prune
        *path = append(*path, choice)     // 做选择 / Choose
        backtrack(path, nextChoices, result)
        *path = (*path)[:len(*path)-1]    // 撤销 / Unchoose
    }
}
```

## 6. 动态规划模板 / Dynamic Programming Template

**核心思想**：定义状态，找出状态转移方程，初始化边界，迭代求解。

### Python

```python
# DP 模板 — 一维状态
# DP template — 1D state
def dp_template(nums):
    n = len(nums)
    dp = [0] * n                        # dp[i] 定义 / Definition of dp[i]
    dp[0] = base_case                   # 初始化 / Initialize
    for i in range(1, n):
        dp[i] = max/min(dp[i-1], ...)   # 状态转移 / State transition
    return dp[n-1]

# DP 模板 — 二维状态
# DP template — 2D state
def dp_2d(m, n):
    dp = [[0] * n for _ in range(m)]
    dp[0][0] = base_case
    for i in range(m):
        for j in range(n):
            dp[i][j] = dp[i-1][j] + dp[i][j-1]  # 状态转移示例
    return dp[m-1][n-1]
```

### Rust

```rust
// DP 模板 — 一维状态
// DP template — 1D state
pub fn dp_1d(nums: &[i32]) -> i32 {
    let n = nums.len();
    let mut dp = vec![0; n];             // dp[i] 定义 / Definition
    dp[0] = nums[0];                     // 初始化 / Initialize
    for i in 1..n {
        dp[i] = dp[i-1].max(nums[i]);   // 状态转移 / Transition
    }
    dp[n-1]
}

// DP 模板 — 二维状态（滚动数组优化）
// DP template — 2D with rolling array
pub fn dp_2d_optimized(m: usize, n: usize) -> i32 {
    let mut dp = vec![0; n];             // 滚动数组 / Rolling array
    dp[0] = 1;                           // 初始化 / Initialize
    for _ in 0..m {
        for j in 1..n {
            dp[j] += dp[j-1];            // 状态转移 / Transition
        }
    }
    dp[n-1]
}
```

### Go

```go
// DP 模板 — 一维状态
// DP template — 1D state
func dp1D(nums []int) int {
    n := len(nums)
    dp := make([]int, n)               // dp[i] 定义 / Definition
    dp[0] = nums[0]                     // 初始化 / Initialize
    for i := 1; i < n; i++ {
        if dp[i-1] > nums[i] {         // 状态转移 / Transition
            dp[i] = dp[i-1]
        } else {
            dp[i] = nums[i]
        }
    }
    return dp[n-1]
}

// DP 模板 — 二维状态（滚动数组）
// DP template — 2D with rolling array
func dp2DOptimized(m, n int) int {
    dp := make([]int, n)               // 滚动数组 / Rolling array
    dp[0] = 1                          // 初始化 / Initialize
    for i := 0; i < m; i++ {
        for j := 1; j < n; j++ {
            dp[j] += dp[j-1]           // 状态转移 / Transition
        }
    }
    return dp[n-1]
}
```

## 7. 并查集模板 / Union-Find Template

**核心思想**：路径压缩 + 按秩合并，维护等价类，近乎 $O(1)$ 操作。

### Python

```python
# 并查集模板
# Union-Find template
class UnionFind:
    def __init__(self, n):
        self.parent = list(range(n))
        self.rank = [0] * n

    def find(self, x):                  # 路径压缩 / Path compression
        if self.parent[x] != x:
            self.parent[x] = self.find(self.parent[x])
        return self.parent[x]

    def union(self, x, y):              # 按秩合并 / Union by rank
        rx, ry = self.find(x), self.find(y)
        if rx == ry: return False
        if self.rank[rx] < self.rank[ry]:
            rx, ry = ry, rx
        self.parent[ry] = rx
        if self.rank[rx] == self.rank[ry]:
            self.rank[rx] += 1
        return True
```

### Rust

```rust
// 并查集模板
// Union-Find template
pub struct UnionFind {
    parent: Vec<usize>,
    rank: Vec<i32>,
}

impl UnionFind {
    pub fn new(n: usize) -> Self {
        UnionFind { parent: (0..n).collect(), rank: vec![0; n] }
    }
    pub fn find(&mut self, x: usize) -> usize {  // 路径压缩
        if self.parent[x] != x {
            self.parent[x] = self.find(self.parent[x]);
        }
        self.parent[x]
    }
    pub fn union(&mut self, x: usize, y: usize) { // 按秩合并
        let (mut rx, mut ry) = (self.find(x), self.find(y));
        if rx == ry { return; }
        if self.rank[rx] < self.rank[ry] { std::mem::swap(&mut rx, &mut ry); }
        self.parent[ry] = rx;
        if self.rank[rx] == self.rank[ry] { self.rank[rx] += 1; }
    }
}
```

### Go

```go
// 并查集模板

    return uf.parent[x]

    rx, ry := uf.Find(x), uf.Find(y)
    if rx == ry { return }
    if uf.rank[rx] < uf.rank[ry] { rx, ry = ry, rx }
    uf.parent[ry] = rx
    if uf.rank[rx] == uf.rank[ry] { uf.rank[rx]++ }
}
```

## 8. Trie 模板 / Trie Template

**核心思想**：前缀共享的树形结构，$O(m)$ 插入/查找，$m$ 为字符串长度。

### Python

```python
# Trie 模板
# Trie template
class TrieNode:
    def __init__(self):
        self.children = {}
        self.is_end = False             # 终止标记 / End marker

class Trie:
    def __init__(self):
        self.root = TrieNode()

    def insert(self, word: str):
        node = self.root
        for ch in word:
            node = node.children.setdefault(ch, TrieNode())
        node.is_end = True

    def search(self, word: str) -> bool:
        node = self._find(word)
        return node is not None and node.is_end

    def startsWith(self, prefix: str) -> bool:
        return self._find(prefix) is not None

    def _find(self, word: str):
        node = self.root
        for ch in word:
            node = node.children.get(ch)
            if node is None: return None
        return node
```

### Rust

```rust
// Trie 模板
// Trie template
use std::collections::HashMap;

pub struct TrieNode {
    pub children: HashMap<char, TrieNode>,
    pub is_end: bool,
}

impl TrieNode {
    pub fn new() -> Self { TrieNode { children: HashMap::new(), is_end: false } }
}

pub struct Trie { root: TrieNode }

impl Trie {
    pub fn new() -> Self { Trie { root: TrieNode::new() } }
    pub fn insert(&mut self, word: String) {
        let mut node = &mut self.root;
        for ch in word.chars() {
            node = node.children.entry(ch).or_insert_with(TrieNode::new);
        }
        node.is_end = true;
    }
}
```

### Go

```go
// Trie 模板
// Trie template
type TrieNode struct {
    children map[rune]*TrieNode
    isEnd    bool
}

type Trie struct { root *TrieNode }

func ConstructorTrie() Trie {
    return Trie{root: &TrieNode{children: make(map[rune]*TrieNode)}}
}

func (this *Trie) Insert(word string) {
    node := this.root
    for _, ch := range word {
        if _, ok := node.children[ch]; !ok {
            node.children[ch] = &TrieNode{children: make(map[rune]*TrieNode)}
        }
        node = node.children[ch]
    }
    node.isEnd = true
```

## 10. 拓扑排序模板 / Topological Sort Template

**核心思想**：Kahn 算法（BFS）或 DFS 后序逆序，处理有向无环图（DAG）的线性排序。

### Python

```python
from collections import deque

# Kahn 算法 — BFS 拓扑排序
# Kahn's algorithm — BFS topological sort
def topological_sort(n, graph):
    indegree = [0] * n
    for u in range(n):
        for v in graph[u]:
            indegree[v] += 1            # 计算入度 / Calculate in-degree
    queue = deque([i for i in range(n) if indegree[i] == 0])
    result = []
    while queue:
        u = queue.popleft()
        result.append(u)                # 输出入度为 0 的节点 / Output node
        for v in graph[u]:
            indegree[v] -= 1            # 移除边，减少入度 / Decrement in-degree
            if indegree[v] == 0:
                queue.append(v)         # 新入度为 0 的节点入队 / Enqueue
    return result if len(result) == n else []  # 有环则返回空 / Cycle detected
```

### Rust

```rust
use std::collections::VecDeque;

// Kahn 算法
// Kahn's algorithm
pub fn topological_sort(n: usize, graph: &Vec<Vec<usize>>) -> Vec<usize> {
    let mut indegree = vec![0; n];
    for u in 0..n {
        for &v in &graph[u] { indegree[v] += 1; }  // 计算入度 / Calc in-degree
    }
    let mut queue: VecDeque<usize> = (0..n).filter(|&i| indegree[i] == 0).collect();
    let mut result = Vec::new();
    while let Some(u) = queue.pop_front() {
        result.push(u);                              // 输出 / Output
        for &v in &graph[u] {
            indegree[v] -= 1;                        // 减少入度 / Decrement
            if indegree[v] == 0 { queue.push_back(v); }
        }
    }
    if result.len() == n { result } else { vec![] }  // 检查环 / Check cycle
}
```

### Go

```go
// Kahn 算法
// Kahn's algorithm
func topologicalSort(n int, graph [][]int) []int {
    indegree := make([]int, n)
    for _, neighbors := range graph {
        for _, v := range neighbors {
            indegree[v]++                // 计算入度 / Calc in-degree
        }
    }
    queue := make([]int, 0)
    for i := 0; i < n; i++ {
        if indegree[i] == 0 { queue = append(queue, i) }
    }
    result := make([]int, 0)
    for len(queue) > 0 {
        u := queue[0]; queue = queue[1:]
        result = append(result, u)       // 输出 / Output
        for _, v := range graph[u] {
            indegree[v]--
            if indegree[v] == 0 { queue = append(queue, v) }
        }
    }
    if len(result) == n { return result }
    return []int{}                       // 有环 / Cycle detected
}
```
