# 堆排序实际应用案例

## 应用场景1：任务调度系统优先队列

### 问题描述
- **背景**：操作系统任务调度器需要管理数千个进程，按优先级动态调度
- **问题**：需要高效支持插入、删除最大/最小值、修改优先级操作
- **约束**：
  - 时间约束：调度决策 < 1μs
  - 空间约束：内嵌内核，内存受限
  - 并发要求：支持多核并行访问
- **数据规模**：同时运行1000-10000个任务

### 算法选择理由
- **为什么选择堆**：
  - O(log n)的插入和删除操作
  - O(1)的查询最大/最小值
  - 原地实现，空间O(1)（除数组外）
  - 完全二叉树，数组存储，cache友好

- **与其他数据结构的对比**：
  | 数据结构 | 插入 | 删除最大 | 查询最大 | 空间 | 实现复杂度 |
  |----------|------|----------|----------|------|-----------|
  | 有序数组 | O(n) | O(1) | O(1) | O(n) | 低 |
  | 平衡BST | O(log n) | O(log n) | O(1) | O(n) | 高 |
  | 二叉堆 | **O(log n)** | **O(log n)** | **O(1)** | **O(n)** | **中** |

### 解决方案
```pseudo
Algorithm TaskScheduler:
    heap = MaxHeap()  // 最大堆，按优先级排序
    
    // 调度循环
    while systemRunning:
        // 1. 检查新任务
        newTasks = getNewTasks()
        for task in newTasks:
            heap.insert(task.priority, task)
        
        // 2. 执行最高优先级任务
        if not heap.empty():
            highestPriorityTask = heap.extractMax()
            execute(highestPriorityTask)
        
        // 3. 处理优先级变化
        priorityChanges = getPriorityUpdates()
        for (task, newPriority) in priorityChanges:
            heap.updatePriority(task.id, newPriority)
        
        // 4. 处理任务完成
        completedTasks = getCompletedTasks()
        for task in completedTasks:
            heap.remove(task.id)

// 支持任意位置更新的堆
Algorithm UpdatePriority(heap, taskId, newPriority):
    index = heap.findIndex(taskId)  // 通过哈希表O(1)查找
    oldPriority = heap[index].priority
    heap[index].priority = newPriority
    
    if newPriority > oldPriority:
        heap.siftUp(index)
    else:
        heap.siftDown(index)
```

### 实际代码示例（Rust）
```rust
use std::cmp::Ordering;
use std::collections::HashMap;

/// 任务结构
#[derive(Clone, Debug)]
pub struct Task {
    pub id: u64,
    pub priority: i32,
    pub name: String,
    pub cpu_usage: f64,
}

impl PartialEq for Task {
    fn eq(&self, other: &Self) -> bool {
        self.priority == other.priority
    }
}

impl Eq for Task {}

impl PartialOrd for Task {
    fn partial_cmp(&self, other: &Self) -> Option<Ordering> {
        Some(self.cmp(other))
    }
}

impl Ord for Task {
    fn cmp(&self, other: &Self) -> Ordering {
        self.priority.cmp(&other.priority)
    }
}

/// 支持O(log n)任意位置更新的优先队列
pub struct TaskScheduler {
    heap: Vec<Task>,
    id_to_index: HashMap<u64, usize>,
}

impl TaskScheduler {
    pub fn new() -> Self {
        Self {
            heap: Vec::new(),
            id_to_index: HashMap::new(),
        }
    }
    
    pub fn push(&mut self, task: Task) {
        let index = self.heap.len();
        self.heap.push(task.clone());
        self.id_to_index.insert(task.id, index);
        self.sift_up(index);
    }
    
    pub fn pop(&mut self) -> Option<Task> {
        if self.heap.is_empty() {
            return None;
        }
        
        let task = self.heap.swap_remove(0);
        self.id_to_index.remove(&task.id);
        
        if !self.heap.is_empty() {
            self.id_to_index.insert(self.heap[0].id, 0);
            self.sift_down(0);
        }
        
        Some(task)
    }
    
    pub fn update_priority(&mut self, task_id: u64, new_priority: i32) -> bool {
        if let Some(&index) = self.id_to_index.get(&task_id) {
            let old_priority = self.heap[index].priority;
            self.heap[index].priority = new_priority;
            
            if new_priority > old_priority {
                self.sift_up(index);
            } else {
                self.sift_down(index);
            }
            true
        } else {
            false
        }
    }
    
    pub fn peek(&self) -> Option<&Task> {
        self.heap.first()
    }
    
    pub fn len(&self) -> usize {
        self.heap.len()
    }
    
    fn sift_up(&mut self, mut index: usize) {
        while index > 0 {
            let parent = (index - 1) / 2;
            if self.heap[index] <= self.heap[parent] {
                break;
            }
            self.swap(index, parent);
            index = parent;
        }
    }
    
    fn sift_down(&mut self, mut index: usize) {
        let len = self.heap.len();
        loop {
            let left = 2 * index + 1;
            let right = 2 * index + 2;
            let mut largest = index;
            
            if left < len && self.heap[left] > self.heap[largest] {
                largest = left;
            }
            if right < len && self.heap[right] > self.heap[largest] {
                largest = right;
            }
            
            if largest == index {
                break;
            }
            
            self.swap(index, largest);
            index = largest;
        }
    }
    
    fn swap(&mut self, i: usize, j: usize) {
        self.id_to_index.insert(self.heap[i].id, j);
        self.id_to_index.insert(self.heap[j].id, i);
        self.heap.swap(i, j);
    }
}
```

## 应用场景2：流式数据Top-K实时监控

### 问题描述
- **背景**：电商平台需要实时监控Top-K热销商品，数据流每秒10万条
- **问题**：数据量巨大，无法存储全量，需要近似或精确Top-K
- **约束**：
  - 时间约束：每5秒输出一次Top-K
  - 空间约束：内存限制1GB
  - 准确性：允许微小误差（<1%）
- **数据规模**：每秒10万条订单，持续24小时

### 算法选择理由
- **为什么选择堆**：
  - 维护固定大小的堆，空间O(k)
  - 支持流式处理，无需存储所有数据
  - 可结合滑动窗口处理时序数据

### 解决方案
```python
import heapq
import time
from collections import defaultdict, deque
from typing import List, Tuple, Dict
import threading

class StreamingTopK:
    """流式数据Top-K监控器"""
    
    def __init__(self, k: int, window_seconds: int = 300):
        self.k = k
        self.window_seconds = window_seconds
        self.counts: Dict[str, int] = defaultdict(int)
        self.timestamps: Dict[str, deque] = defaultdict(lambda: deque())
        self.min_heap: List[Tuple[int, str]] = []  # (count, item)
        self.heap_items: set = set()
        self.lock = threading.Lock()
    
    def process_event(self, item: str, timestamp: float = None):
        """处理流式事件"""
        if timestamp is None:
            timestamp = time.time()
        
        with self.lock:
            # 清理过期数据
            self._cleanup_expired(timestamp)
            
            # 更新计数
            self.counts[item] += 1
            self.timestamps[item].append(timestamp)
            
            # 维护Top-K堆
            self._update_heap(item)
    
    def _cleanup_expired(self, current_time: float):
        """清理过期数据（滑动窗口）"""
        cutoff = current_time - self.window_seconds
        expired_items = []
        
        for item, timestamps in self.timestamps.items():
            while timestamps and timestamps[0] < cutoff:
                timestamps.popleft()
                self.counts[item] -= 1
            
            if not timestamps:
                expired_items.append(item)
        
        for item in expired_items:
            del self.counts[item]
            del self.timestamps[item]
            if item in self.heap_items:
                self._remove_from_heap(item)
    
    def _update_heap(self, item: str):
        """更新堆中的元素"""
        count = self.counts[item]
        
        if item in self.heap_items:
            # 更新现有元素（惰性删除+重新插入）
            self._remove_from_heap(item)
            heapq.heappush(self.min_heap, (count, item))
            self.heap_items.add(item)
        elif len(self.min_heap) < self.k:
            # 堆未满，直接插入
            heapq.heappush(self.min_heap, (count, item))
            self.heap_items.add(item)
        elif count > self.min_heap[0][0]:
            # 替换最小元素
            old_count, old_item = heapq.heapreplace(self.min_heap, (count, item))
            self.heap_items.remove(old_item)
            self.heap_items.add(item)
    
    def _remove_from_heap(self, item: str):
        """从堆中移除元素（标记删除）"""
        self.heap_items.discard(item)
    
    def get_top_k(self) -> List[Tuple[str, int]]:
        """获取当前Top-K"""
        with self.lock:
            # 过滤无效条目
            valid_items = [(item, count) for count, item in self.min_heap 
                          if item in self.heap_items and self.counts[item] == count]
            valid_items.sort(key=lambda x: x[1], reverse=True)
            return valid_items[:self.k]


# 基准测试
import random

def benchmark_streaming():
    topk = StreamingTopK(k=100, window_seconds=60)
    
    # 模拟10万条/秒的数据流
    items = [f"product_{i}" for i in range(10000)]
    
    start = time.time()
    events = 100000
    
    for _ in range(events):
        item = random.choice(items)
        topk.process_event(item)
    
    elapsed = time.time() - start
    
    print(f"处理 {events} 条事件耗时: {elapsed:.3f}s")
    print(f"处理速度: {events/elapsed:,.0f} 事件/秒")
    
    top_10 = topk.get_top_k()[:10]
    print(f"Top 10: {top_10}")

if __name__ == '__main__':
    benchmark_streaming()
```
