## 应用场景1：音频信号处理与降噪

### 问题描述

- **背景**：语音识别和音频编辑需要实时处理音频信号
- **问题**：
  - 频域分析（音调识别、滤波）
  - 实时降噪
  - 音频压缩
- **约束**：
  - 延迟约束：实时处理 < 10ms
  - 采样率：44.1kHz（CD音质）或16kHz（语音）
- **数据规模**：每秒44100个采样点

### 算法选择理由

- **为什么选择FFT**：
  - DFT从O(n²)降到O(n log n)
  - 1000点音频DFT从100万次运算降到约1万次
  - 广泛应用于信号处理标准库

- **算法对比**：

  | 算法 | 时间复杂度 | 应用 |
  |------|-----------|------|
  | DFT | O(n²) | 理论分析 |
  | FFT Cooley-Tukey | **O(n log n)** | **通用** |
  | Bluestein | O(n log n) | 任意长度 |
  | Number-Theoretic | O(n log n) | 整数变换 |

### 解决方案

```pseudo
Algorithm FFT(a, n, w):
    // a: 输入数组
    // n: 长度（必须是2的幂）
    // w: n次单位根

    if n == 1:
        return a

    // 分治：分为偶数和奇数下标
    a_even = [a[0], a[2], ..., a[n-2]]
    a_odd = [a[1], a[3], ..., a[n-1]]

    y_even = FFT(a_even, n/2, w^2)
    y_odd = FFT(a_odd, n/2, w^2)

    // 合并
    y = Array[n]
    for k from 0 to n/2 - 1:
        y[k] = y_even[k] + w^k * y_odd[k]
        y[k + n/2] = y_even[k] - w^k * y_odd[k]

    return y

// 迭代版本（实际使用）
Algorithm IterativeFFT(a):
    n = len(a)
    bit_reverse_copy(a, A)  // 位逆序重排

    for s from 1 to log(n):
        m = 2^s
        w_m = exp(2πi / m)

        for k from 0 to n-1 step m:
            w = 1
            for j from 0 to m/2 - 1:
                t = w * A[k + j + m/2]
                u = A[k + j]
                A[k + j] = u + t
                A[k + j + m/2] = u - t
                w = w * w_m

    return A
```

### 实际代码示例（Python + NumPy）

```python
import numpy as np
import time
from typing import Tuple, List
import matplotlib.pyplot as plt

class AudioProcessor:
    """音频处理器 - FFT应用"""

    def __init__(self, sample_rate: int = 44100):
        self.sample_rate = sample_rate
        self.frame_size = 2048  # FFT点数
        self.hop_size = 512     # 帧移

    def stft(self, signal: np.ndarray) -> np.ndarray:
        """
        短时傅里叶变换
        将时域信号转换为时频域表示
        """
        # 分帧
        num_frames = 1 + (len(signal) - self.frame_size) // self.hop_size

        # 预分配频谱矩阵
        spectrogram = np.zeros((self.frame_size // 2 + 1, num_frames), dtype=complex)

        # Hanning窗
        window = np.hanning(self.frame_size)

        for i in range(num_frames):
            start = i * self.hop_size
            frame = signal[start:start + self.frame_size] * window

            # FFT（使用NumPy优化实现）
            spectrum = np.fft.rfft(frame)
            spectrogram[:, i] = spectrum

        return spectrogram

    def istft(self, spectrogram: np.ndarray, length: int) -> np.ndarray:
        """逆短时傅里叶变换"""
        num_frames = spectrogram.shape[1]
        signal = np.zeros(length)
        window_sum = np.zeros(length)

        window = np.hanning(self.frame_size)

        for i in range(num_frames):
            start = i * self.hop_size
            # IFFT
            frame = np.fft.irfft(spectrogram[:, i], n=self.frame_size) * window

            signal[start:start + self.frame_size] += frame
            window_sum[start:start + self.frame_size] += window ** 2

        # 归一化
        signal = signal / np.maximum(window_sum, 1e-10)
        return signal

    def spectral_subtraction(self, signal: np.ndarray, noise_profile: np.ndarray = None) -> np.ndarray:
        """
        谱减法降噪
        """
        # STFT
        spectrogram = self.stft(signal)
        magnitude = np.abs(spectrogram)
        phase = np.angle(spectrogram)

        # 估计噪声（如果没有提供）
        if noise_profile is None:
            # 使用前100ms作为噪声估计
            noise_frames = int(0.1 * self.sample_rate / self.hop_size)
            noise_profile = np.mean(magnitude[:, :noise_frames], axis=1, keepdims=True)

        # 谱减法
        alpha = 2.0  # 过减因子
        beta = 0.01  # 谱下限

        cleaned_magnitude = np.maximum(magnitude - alpha * noise_profile, beta * magnitude)

        # 重建信号
        cleaned_spectrogram = cleaned_magnitude * np.exp(1j * phase)
        cleaned_signal = self.istft(cleaned_spectrogram, len(signal))

        return cleaned_signal

    def pitch_detection(self, signal: np.ndarray) -> float:
        """
        基频检测（使用FFT）
        """
        # 加窗
        window = np.hanning(len(signal))
        windowed = signal * window

        # FFT
        spectrum = np.fft.rfft(windowed)
        magnitude = np.abs(spectrum)

        # 找到峰值频率
        freqs = np.fft.rfftfreq(len(signal), 1/self.sample_rate)

        # 限制在人声范围 (80Hz - 1000Hz)
        valid_range = (freqs >= 80) & (freqs <= 1000)
        if not np.any(valid_range):
            return 0.0

        peak_idx = np.argmax(magnitude[valid_range])
        pitch = freqs[valid_range][peak_idx]

        return pitch


def benchmark_fft():
    """测试FFT性能"""
    sizes = [256, 512, 1024, 2048, 4096, 8192, 16384]

    print("FFT性能测试:")
    print(f"{'Size':>10} {'NumPy FFT(ms)':>15} {'理论加速比':>15}")
    print("-" * 45)

    for size in sizes:
        # 生成测试信号
        signal = np.random.randn(size)

        # 测试NumPy FFT
        start = time.time()
        for _ in range(1000):
            _ = np.fft.fft(signal)
        elapsed = (time.time() - start) / 1000 * 1000

        # 理论加速比 (DFT/FFT)
        theoretical_speedup = (size ** 2) / (size * np.log2(size))

        print(f"{size:>10} {elapsed:>15.3f} {theoretical_speedup:>15.1f}x")


def demo_audio_processing():
    """演示音频处理"""
    processor = AudioProcessor(sample_rate=16000)

    # 生成测试信号（带噪声的正弦波）
    duration = 1.0
    t = np.linspace(0, duration, int(processor.sample_rate * duration))

    # 440Hz + 880Hz 纯音
    clean_signal = np.sin(2 * np.pi * 440 * t) + 0.5 * np.sin(2 * np.pi * 880 * t)

    # 添加噪声
    noise = 0.3 * np.random.randn(len(t))
    noisy_signal = clean_signal + noise

    print("音频处理演示:")
    print(f"  采样率: {processor.sample_rate} Hz")
    print(f"  信号长度: {len(t)} 样本")

    # 音调检测
    start = time.time()
    pitch = processor.pitch_detection(noisy_signal[:2048])
    pitch_time = (time.time() - start) * 1000
    print(f"  检测基频: {pitch:.1f} Hz (耗时: {pitch_time:.2f}ms)")

    # 降噪
    start = time.time()
    cleaned = processor.spectral_subtraction(noisy_signal)
    denoise_time = (time.time() - start) * 1000

    # 计算信噪比改善
    noise_power_before = np.mean(noise**2)
    residual_noise = cleaned - clean_signal
    noise_power_after = np.mean(residual_noise**2)
    snr_improvement = 10 * np.log10(noise_power_before / noise_power_after)

    print(f"  降噪处理: {denoise_time:.2f}ms")
    print(f"  SNR改善: {snr_improvement:.1f} dB")

if __name__ == '__main__':
    benchmark_fft()
    print()
    demo_audio_processing()
```

## 应用场景2：图像处理与卷积加速

### 问题描述

- **背景**：图像处理中的卷积操作（模糊、边缘检测）计算量大
- **问题**：
  - 大卷积核计算慢
  - 需要实时处理高分辨率图像

### 解决方案

```python
import numpy as np
from scipy import signal
import time

class ImageProcessor:
    """图像处理器 - 使用FFT加速卷积"""

    def __init__(self):
        pass

    def fast_convolution(self, image: np.ndarray, kernel: np.ndarray) -> np.ndarray:
        """
        使用FFT加速卷积
        复杂度: O(n log n) vs 空间卷积 O(n*k^2)
        """
        # 计算FFT大小（避免循环卷积）
        shape = (image.shape[0] + kernel.shape[0] - 1,
                 image.shape[1] + kernel.shape[1] - 1)

        fshape = [next_pow2(x) for x in shape]

        # FFT
        image_fft = np.fft.rfft2(image, fshape)
        kernel_fft = np.fft.rfft2(kernel, fshape)

        # 频域相乘
        result_fft = image_fft * kernel_fft

        # IFFT
        result = np.fft.irfft2(result_fft, fshape)

        # 裁剪到有效区域
        valid_shape = (image.shape[0] - kernel.shape[0] + 1,
                       image.shape[1] - kernel.shape[1] + 1)
        start = ((fshape[0] - valid_shape[0]) // 2,
                 (fshape[1] - valid_shape[1]) // 2)

        return result[start[0]:start[0]+valid_shape[0],
                     start[1]:start[1]+valid_shape[1]]

    def gaussian_blur_fft(self, image: np.ndarray, sigma: float) -> np.ndarray:
        """高斯模糊 - FFT实现"""
        # 生成高斯核
        size = int(6 * sigma) | 1  # 奇数
        x = np.arange(size) - size // 2
        kernel = np.exp(-x**2 / (2 * sigma**2))
        kernel = kernel / kernel.sum()
        kernel = np.outer(kernel, kernel)

        return self.fast_convolution(image, kernel)

def next_pow2(n):
    return 1 << (n - 1).bit_length()

# 测试
def benchmark_convolution():
    sizes = [(512, 512), (1024, 1024), (2048, 2048)]
    kernel_size = 31

    print("卷积性能对比:")
    print(f"{'Image Size':>15} {'Spatial(ms)':>12} {'FFT(ms)':>12} {'Speedup':>10}")
    print("-" * 55)

    for h, w in sizes:
        image = np.random.rand(h, w).astype(np.float32)
        kernel = np.random.rand(kernel_size, kernel_size).astype(np.float32)
        kernel = kernel / kernel.sum()

        # 空间卷积
        start = time.time()
        _ = signal.convolve2d(image, kernel, mode='valid')
        spatial_time = (time.time() - start) * 1000

        # FFT卷积
        processor = ImageProcessor()
        start = time.time()
        _ = processor.fast_convolution(image, kernel)
        fft_time = (time.time() - start) * 1000

        speedup = spatial_time / fft_time
        print(f"{h}x{w:>8} {spatial_time:>12.1f} {fft_time:>12.1f} {speedup:>10.1f}x")

if __name__ == '__main__':
    benchmark_convolution()
```
