## 应用场景1：网络基础设施设计（光缆布线）

### 问题描述

- **背景**：ISP需要在城市间铺设光纤网络，连接所有数据中心
- **问题**：最小化总铺设成本，同时保证网络连通性
- **约束**：
  - 成本约束：每公里铺设成本10-50万美元
  - 可靠性：主要链路需有备份
  - 地形限制：某些区域无法铺设
- **数据规模**：50个主要城市，1225条可能链路

### 算法选择理由

- **为什么选择MST**：
  - 保证连通前提下成本最小
  - 不存在环路（避免广播风暴）
  - 算法成熟，实现简单
  - Prim适合稠密图，Kruskal适合稀疏图

- **算法对比**：

  | 算法 | 时间复杂度 | 空间复杂度 | 适用图类型 | 实现难度 |
  |------|-----------|-----------|------------|----------|
  | Prim | O(E log V) | O(V) | 稠密图 | 中 |
  | Kruskal | O(E log E) | O(V) | 稀疏图 | 低 |
  | Boruvka | O(E log V) | O(V) | 并行化 | 高 |

### 解决方案

```pseudo
Algorithm PrimMST(graph):
    // 初始化
    for each vertex v in graph:
        key[v] = infinity
        parent[v] = null
        inMST[v] = false

    key[source] = 0
    pq = PriorityQueue<(key, vertex)>()
    pq.push((0, source))

    while not pq.empty():
        (_, u) = pq.pop()

        if inMST[u]:
            continue
        inMST[u] = true

        for each (v, weight) in graph.adjacent(u):
            if not inMST[v] and weight < key[v]:
                key[v] = weight
                parent[v] = u
                pq.push((key[v], v))

    return parent  // 表示MST

Algorithm KruskalMST(graph):
    edges = sort(graph.edges)  // 按权重排序
    mst = []
    uf = UnionFind(graph.vertices)

    for (u, v, weight) in edges:
        if uf.find(u) != uf.find(v):
            mst.append((u, v, weight))
            uf.union(u, v)

            if mst.size() == graph.vertices.size() - 1:
                break

    return mst
```

### 实际代码示例（Python）

```python
import heapq
from typing import List, Tuple, Dict, Set
from dataclasses import dataclass
import math

@dataclass
class Edge:
    u: int
    v: int
    weight: float

    def __lt__(self, other):
        return self.weight < other.weight

class UnionFind:
    """并查集"""
    def __init__(self, n: int):
        self.parent = list(range(n))
        self.rank = [0] * n

    def find(self, x: int) -> int:
        if self.parent[x] != x:
            self.parent[x] = self.find(self.parent[x])
        return self.parent[x]

    def union(self, x: int, y: int):
        px, py = self.find(x), self.find(y)
        if px == py:
            return
        if self.rank[px] < self.rank[py]:
            px, py = py, px
        self.parent[py] = px
        if self.rank[px] == self.rank[py]:
            self.rank[px] += 1

class NetworkDesigner:
    """网络设计器 - MST应用"""

    def __init__(self, num_cities: int):
        self.n = num_cities
        self.edges: List[Edge] = []
        self.city_names: Dict[int, str] = {}
        self.coordinates: Dict[int, Tuple[float, float]] = {}

    def add_city(self, id: int, name: str, lat: float, lon: float):
        self.city_names[id] = name
        self.coordinates[id] = (lat, lon)

    def add_possible_link(self, u: int, v: int, cost_per_km: float = 1.0):
        """添加可能的链路（基于地理距离计算成本）"""
        if u not in self.coordinates or v not in self.coordinates:
            return

        lat1, lon1 = self.coordinates[u]
        lat2, lon2 = self.coordinates[v]

        # Haversine距离（公里）
        distance = self._haversine(lat1, lon1, lat2, lon2)
        cost = distance * cost_per_km

        self.edges.append(Edge(u, v, cost))

    def _haversine(self, lat1: float, lon1: float, lat2: float, lon2: float) -> float:
        """计算两点间距离（公里）"""
        R = 6371  # 地球半径
        lat1, lon1, lat2, lon2 = map(math.radians, [lat1, lon1, lat2, lon2])
        dlat = lat2 - lat1
        dlon = lon2 - lon1
        a = math.sin(dlat/2)**2 + math.cos(lat1) * math.cos(lat2) * math.sin(dlon/2)**2
        return 2 * R * math.asin(math.sqrt(a))

    def prim_mst(self) -> Tuple[List[Edge], float]:
        """Prim算法求MST"""
        # 构建邻接表
        adj = [[] for _ in range(self.n)]
        for e in self.edges:
            adj[e.u].append((e.v, e.weight))
            adj[e.v].append((e.u, e.weight))

        # Prim算法
        in_mst = [False] * self.n
        key = [float('inf')] * self.n
        parent = [-1] * self.n

        key[0] = 0
        pq = [(0, 0)]  # (key, vertex)

        while pq:
            _, u = heapq.heappop(pq)

            if in_mst[u]:
                continue
            in_mst[u] = True

            for v, w in adj[u]:
                if not in_mst[v] and w < key[v]:
                    key[v] = w
                    parent[v] = u
                    heapq.heappush(pq, (key[v], v))

        # 构建结果
        mst = []
        total_cost = 0
        for v in range(1, self.n):
            if parent[v] != -1:
                mst.append(Edge(parent[v], v, key[v]))
                total_cost += key[v]

        return mst, total_cost

    def kruskal_mst(self) -> Tuple[List[Edge], float]:
        """Kruskal算法求MST"""
        sorted_edges = sorted(self.edges)
        uf = UnionFind(self.n)
        mst = []
        total_cost = 0

        for edge in sorted_edges:
            if uf.find(edge.u) != uf.find(edge.v):
                uf.union(edge.u, edge.v)
                mst.append(edge)
                total_cost += edge.weight

                if len(mst) == self.n - 1:
                    break

        return mst, total_cost

    def print_network_plan(self, mst: List[Edge], total_cost: float):
        """打印网络规划方案"""
        print("=" * 60)
        print("光纤网络铺设方案 (MST)")
        print("=" * 60)
        print(f"总成本: ${total_cost:,.2f} 万")
        print(f"链路数: {len(mst)}")
        print(f"平均链路成本: ${total_cost/len(mst):,.2f} 万")
        print("\n详细链路:")
        print("-" * 60)
        print(f"{'序号':>4} {'城市A':>15} {'城市B':>15} {'距离(km)':>12} {'成本(万)':>12}")
        print("-" * 60)

        for i, edge in enumerate(mst, 1):
            city_a = self.city_names.get(edge.u, f"City{edge.u}")
            city_b = self.city_names.get(edge.v, f"City{edge.v}")
            print(f"{i:>4} {city_a:>15} {city_b:>15} {edge.weight:>12.1f} {edge.weight:>12.1f}")

        print("-" * 60)


# 中国主要城市网络设计示例
def design_china_network():
    designer = NetworkDesigner(10)

    # 添加主要城市
    cities = [
        (0, "北京", 39.9042, 116.4074),
        (1, "上海", 31.2304, 121.4737),
        (2, "广州", 23.1291, 113.2644),
        (3, "深圳", 22.5431, 114.0579),
        (4, "成都", 30.5728, 104.0668),
        (5, "武汉", 30.5928, 114.3055),
        (6, "西安", 34.3416, 108.9398),
        (7, "南京", 32.0603, 118.7969),
        (8, "杭州", 30.2741, 120.1551),
        (9, "重庆", 29.5630, 106.5516),
    ]

    for id, name, lat, lon in cities:
        designer.add_city(id, name, lat, lon)

    # 添加所有可能的链路（完全图）
    cost_per_km = 0.05  # 50万美元/1000km
    for i in range(10):
        for j in range(i+1, 10):
            designer.add_possible_link(i, j, cost_per_km)

    # 计算MST
    mst_prim, cost_prim = designer.prim_mst()
    mst_kruskal, cost_kruskal = designer.kruskal_mst()

    print("Prim算法结果:")
    designer.print_network_plan(mst_prim, cost_prim)

    print("\nKruskal算法结果:")
    designer.print_network_plan(mst_kruskal, cost_kruskal)

if __name__ == '__main__':
    design_china_network()
```

## 应用场景2：电路板布线（PCB设计）

### 问题描述

- **背景**：PCB设计需要连接多个元件引脚，最小化走线长度
- **问题**：避免信号干扰，减少层数，最小化总线长
- **约束**：
  - 最小线宽/线距
  - 过孔成本
  - 信号完整性

### 解决方案

```python
from typing import List, Tuple, Set
import heapq

class Point:
    def __init__(self, x: int, y: int, layer: int = 0):
        self.x = x
        self.y = y
        self.layer = layer

    def __hash__(self):
        return hash((self.x, self.y, self.layer))

    def __eq__(self, other):
        return (self.x, self.y, self.layer) == (other.x, other.y, other.layer)

    def distance(self, other):
        return abs(self.x - other.x) + abs(self.y - other.y) + \
               10 * abs(self.layer - other.layer)  # 层切换成本高

class PCBRouter:
    """PCB布线器 - 使用MST优化走线"""

    def __init__(self, width: int, height: int, layers: int = 2):
        self.width = width
        self.height = height
        self.layers = layers
        self.obstacles: Set[Tuple[int, int, int]] = set()
        self.pins: List[Point] = []

    def add_obstacle(self, x: int, y: int, layer: int):
        self.obstacles.add((x, y, layer))

    def add_pin(self, x: int, y: int, layer: int = 0):
        self.pins.append(Point(x, y, layer))

    def route_net(self) -> Tuple[List[Tuple[Point, Point]], int]:
        """
        对一组引脚进行布线
        返回: (走线列表, 总长度)
        """
        if len(self.pins) < 2:
            return [], 0

        # 构建完全图
        n = len(self.pins)
        edges = []

        for i in range(n):
            for j in range(i + 1, n):
                dist = self.pins[i].distance(self.pins[j])
                edges.append((dist, i, j))

        # Kruskal MST
        edges.sort()
        uf = UnionFind(n)
        routes = []
        total_length = 0

        for dist, i, j in edges:
            if uf.find(i) != uf.find(j):
                uf.union(i, j)
                routes.append((self.pins[i], self.pins[j]))
                total_length += int(dist)

                if len(routes) == n - 1:
                    break

        return routes, total_length

    def print_layout(self, routes: List[Tuple[Point, Point]]):
        """打印布线图"""
        print(f"PCB布线方案 ({self.width}x{self.height}, {self.layers}层)")
        print(f"引脚数: {len(self.pins)}")
        print(f"走线数: {len(routes)}")
        print("\n走线详情:")
        for i, (p1, p2) in enumerate(routes, 1):
            print(f"  {i}. ({p1.x},{p1.y},L{p1.layer}) -> ({p2.x},{p2.y},L{p2.layer})")

# 示例
def demo_pcb_router():
    router = PCBRouter(100, 100, layers=2)

    # 添加芯片引脚
    pins = [
        (10, 10, 0), (10, 90, 0), (90, 10, 0), (90, 90, 0),
        (50, 10, 1), (50, 90, 1), (10, 50, 1), (90, 50, 1),
    ]

    for x, y, layer in pins:
        router.add_pin(x, y, layer)

    routes, total = router.route_net()
    router.print_layout(routes)
    print(f"\n总线长: {total}")

if __name__ == '__main__':
    demo_pcb_router()
```
