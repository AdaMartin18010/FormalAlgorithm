## 应用场景1：高性能缓存系统

### 问题描述

- **背景**：大型Web应用需要缓存热点数据，减少数据库压力
- **问题**：需要O(1)读写、支持高并发、LRU淘汰策略
- **约束**：
  - 时间约束：P99延迟 < 1ms
  - 空间约束：内存限制64GB/节点
  - 并发要求：支持10万QPS
- **数据规模**：单节点存储1000万键值对，平均value 1KB

### 算法选择理由

- **为什么选择哈希表**：
  - O(1)平均时间复杂度的读写
  - 实现简单，内存布局可控
  - 配合链表可实现LRU
  - 支持高效扩容（渐进式rehash）

- **与其他数据结构的对比**：

  | 数据结构 | 查询 | 插入 | 删除 | 有序遍历 | 内存 |
  |----------|------|------|------|----------|------|
  | 平衡树 | O(log n) | O(log n) | O(log n) | 是 | 高 |
  | 跳表 | O(log n) | O(log n) | O(log n) | 是 | 中 |
  | 哈希表 | **O(1)** | **O(1)** | **O(1)** | 否 | **低** |

### 解决方案

```pseudo
Algorithm LRUCache:
    hashTable = HashMap<key, Node>()
    lruList = DoublyLinkedList()
    capacity = N

    Function Get(key):
        if hashTable.contains(key):
            node = hashTable[key]
            lruList.moveToFront(node)  // 更新LRU
            return node.value
        return null

    Function Put(key, value):
        if hashTable.contains(key):
            node = hashTable[key]
            node.value = value
            lruList.moveToFront(node)
        else:
            if hashTable.size() >= capacity:
                // 淘汰最久未使用
                evicted = lruList.removeLast()
                hashTable.remove(evicted.key)

            node = new Node(key, value)
            lruList.addToFront(node)
            hashTable[key] = node

    Function Remove(key):
        if hashTable.contains(key):
            node = hashTable[key]
            lruList.remove(node)
            hashTable.remove(key)

// 渐进式rehash避免停顿
Algorithm ProgressiveRehash:
    oldTable = hashTable
    newTable = new HashMap(newCapacity)
    rehashIndex = 0

    Function Get(key):
        // 先查新表，再查旧表
        if newTable.contains(key):
            return newTable[key]
        if oldTable.contains(key):
            return oldTable[key]
        return null

    Function Put(key, value):
        // 只写入新表
        newTable[key] = value
        if oldTable.contains(key):
            oldTable.remove(key)

        // 渐进迁移
        RehashStep()

    Function RehashStep():
        // 每次操作迁移少量数据
        for i from 0 to 10:
            if rehashIndex < oldTable.size():
                entry = oldTable[rehashIndex]
                if entry is not empty:
                    newTable[entry.key] = entry.value
                    oldTable.remove(entry.key)
                rehashIndex += 1
            else:
                oldTable = null  // 迁移完成
                break
```

### 实际代码示例（Rust）

```rust
use std::collections::HashMap;
use std::ptr::NonNull;
use std::alloc::{alloc, dealloc, Layout};

/// LRU缓存节点
struct Node<K, V> {
    key: K,
    value: V,
    prev: Option<NonNull<Node<K, V>>>,
    next: Option<NonNull<Node<K, V>>>,
}

/// 高性能LRU缓存
pub struct LRUCache<K: Eq + std::hash::Hash + Clone, V> {
    map: HashMap<K, NonNull<Node<K, V>>>,
    capacity: usize,
    // 双链表头尾
    head: Option<NonNull<Node<K, V>>>,
    tail: Option<NonNull<Node<K, V>>>,
}

impl<K: Eq + std::hash::Hash + Clone, V> LRUCache<K, V> {
    pub fn new(capacity: usize) -> Self {
        Self {
            map: HashMap::with_capacity(capacity),
            capacity,
            head: None,
            tail: None,
        }
    }

    pub fn get(&mut self, key: &K) -> Option<&V> {
        if let Some(&node_ptr) = self.map.get(key) {
            unsafe {
                self.move_to_front(node_ptr);
                Some(&(*node_ptr.as_ptr()).value)
            }
        } else {
            None
        }
    }

    pub fn put(&mut self, key: K, value: V) {
        if let Some(&node_ptr) = self.map.get(&key) {
            // 更新现有值
            unsafe {
                (*node_ptr.as_ptr()).value = value;
                self.move_to_front(node_ptr);
            }
        } else {
            // 插入新值
            if self.map.len() >= self.capacity {
                self.evict_lru();
            }

            let node = Box::new(Node {
                key: key.clone(),
                value,
                prev: None,
                next: self.head,
            });

            let node_ptr = unsafe {
                NonNull::new_unchecked(Box::into_raw(node))
            };

            if let Some(head) = self.head {
                unsafe { (*head.as_ptr()).prev = Some(node_ptr); }
            }

            self.head = Some(node_ptr);
            if self.tail.is_none() {
                self.tail = Some(node_ptr);
            }

            self.map.insert(key, node_ptr);
        }
    }

    unsafe fn move_to_front(&mut self, node: NonNull<Node<K, V>>) {
        if Some(node) == self.head {
            return;
        }

        // 从当前位置移除
        let prev = (*node.as_ptr()).prev;
        let next = (*node.as_ptr()).next;

        if let Some(p) = prev {
            (*p.as_ptr()).next = next;
        }
        if let Some(n) = next {
            (*n.as_ptr()).prev = prev;
        } else {
            self.tail = prev;
        }

        // 插入头部
        (*node.as_ptr()).prev = None;
        (*node.as_ptr()).next = self.head;

        if let Some(h) = self.head {
            (*h.as_ptr()).prev = Some(node);
        }
        self.head = Some(node);
    }

    fn evict_lru(&mut self) {
        if let Some(tail) = self.tail {
            unsafe {
                let key = (*tail.as_ptr()).key.clone();
                self.map.remove(&key);

                let new_tail = (*tail.as_ptr()).prev;
                if let Some(nt) = new_tail {
                    (*nt.as_ptr()).next = None;
                } else {
                    self.head = None;
                }
                self.tail = new_tail;

                drop(Box::from_raw(tail.as_ptr()));
            }
        }
    }
}

impl<K: Eq + std::hash::Hash + Clone, V> Drop for LRUCache<K, V> {
    fn drop(&mut self) {
        let mut current = self.head;
        while let Some(node) = current {
            unsafe {
                current = (*node.as_ptr()).next;
                drop(Box::from_raw(node.as_ptr()));
            }
        }
    }
}

/// 布谷鸟哈希表（解决冲突）
pub struct CuckooHashMap<K: Eq + std::hash::Hash + Clone, V> {
    table1: Vec<Option<(K, V)>>,
    table2: Vec<Option<(K, V)>>,
    size: usize,
    capacity: usize,
}

impl<K: Eq + std::hash::Hash + Clone, V> CuckooHashMap<K, V> {
    pub fn new(capacity: usize) -> Self {
        let cap = capacity.next_power_of_two();
        Self {
            table1: vec![None; cap],
            table2: vec![None; cap],
            size: 0,
            capacity: cap,
        }
    }

    fn hash1(&self, key: &K) -> usize {
        use std::collections::hash_map::DefaultHasher;
        use std::hash::{Hash, Hasher};
        let mut hasher = DefaultHasher::new();
        key.hash(&mut hasher);
        (hasher.finish() as usize) & (self.capacity - 1)
    }

    fn hash2(&self, key: &K) -> usize {
        use std::collections::hash_map::DefaultHasher;
        use std::hash::{Hash, Hasher};
        let mut hasher = DefaultHasher::new();
        key.hash(&mut hasher);
        ((hasher.finish() >> 32) as usize) & (self.capacity - 1)
    }

    pub fn get(&self, key: &K) -> Option<&V> {
        let idx1 = self.hash1(key);
        if let Some((k, v)) = &self.table1[idx1] {
            if k == key {
                return Some(v);
            }
        }

        let idx2 = self.hash2(key);
        if let Some((k, v)) = &self.table2[idx2] {
            if k == key {
                return Some(v);
            }
        }

        None
    }

    pub fn insert(&mut self, key: K, value: V) -> bool {
        let mut curr_key = key;
        let mut curr_val = value;
        let mut table_id = 1;

        for _ in 0..self.capacity {
            let idx = if table_id == 1 {
                self.hash1(&curr_key)
            } else {
                self.hash2(&curr_key)
            };

            let table = if table_id == 1 {
                &mut self.table1
            } else {
                &mut self.table2
            };

            if table[idx].is_none() {
                table[idx] = Some((curr_key, curr_val));
                self.size += 1;
                return true;
            }

            // 踢出现有元素
            std::mem::swap(&mut table[idx], &mut Some((curr_key, curr_val)));
            let (k, v) = table[idx].take().unwrap();
            curr_key = k;
            curr_val = v;
            table_id = 3 - table_id;  // 切换表
        }

        // 需要扩容
        false
    }
}

    }
}
```

## 应用场景2：大数据去重系统

### 问题描述

- **背景**：广告点击流处理系统需要实时去重，防止重复计费
- **问题**：每日10亿条点击记录，需要去重判断，99.9%是重复
- **约束**：
  - 时间约束：去重判断 < 1ms
  - 空间约束：内存限制128GB
  - 准确性：不能有误判（重复被当新数据）
- **数据规模**：每日10亿条，每条约50字节

### 算法选择理由

- **为什么选择布隆过滤器+哈希表**：
  - 布隆过滤器：O(1)判断"可能存在"或"肯定不存在"
  - 内存节省：1%误判率仅需约10 bits/元素
  - 哈希表：精确判断布隆过滤器命中后的去重

### 解决方案

```python
import mmh3  # MurmurHash3
import math
import time
from typing import List

class BloomFilter:
    """布隆过滤器实现"""

    def __init__(self, expected_items: int, false_positive_rate: float = 0.01):
        """
        expected_items: 预期元素数量
        false_positive_rate: 可接受的误判率
        """
        self.size = self._optimal_size(expected_items, false_positive_rate)
        self.hash_count = self._optimal_hash_count(self.size, expected_items)
        self.bit_array = bytearray((self.size + 7) // 8)
        self.item_count = 0

    def _optimal_size(self, n: int, p: float) -> int:
        """计算最优位数组大小: m = -n*ln(p) / (ln(2)^2)"""
        return int(-n * math.log(p) / (math.log(2) ** 2))

    def _optimal_hash_count(self, m: int, n: int) -> int:
        """计算最优哈希函数数量: k = m/n * ln(2)"""
        return max(1, int(m / n * math.log(2)))

    def _get_positions(self, item: str) -> List[int]:
        """获取位数组位置"""
        positions = []
        for i in range(self.hash_count):
            hash_val = mmh3.hash(item, i) % self.size
            positions.append(abs(hash_val))
        return positions

    def add(self, item: str):
        """添加元素"""
        for pos in self._get_positions(item):
            byte_index = pos // 8
            bit_index = pos % 8
            self.bit_array[byte_index] |= (1 << bit_index)
        self.item_count += 1

    def __contains__(self, item: str) -> bool:
        """检查元素可能存在（True）或肯定不存在（False）"""
        for pos in self._get_positions(item):
            byte_index = pos // 8
            bit_index = pos % 8
            if not (self.bit_array[byte_index] & (1 << bit_index)):
                return False
        return True

    def current_false_positive_rate(self) -> float:
        """计算当前误判率: (1 - e^(-kn/m))^k"""
        return (1 - math.exp(-self.hash_count * self.item_count / self.size)) ** self.hash_count


class DedupSystem:
    """两级去重系统：布隆过滤器 + 哈希表"""

    def __init__(self, memory_limit_gb: int = 128, daily_items: int = 1_000_000_000):
        self.memory_limit = memory_limit_gb * 1024 * 1024 * 1024

        # 布隆过滤器：快速过滤大部分重复
        # 128GB可支持约1000亿元素，误判率1%
        self.bloom = BloomFilter(daily_items * 3, false_positive_rate=0.001)

        # 哈希表：精确去重（仅存储布隆过滤器误判的可能新数据）
        self.exact_set = set()
        self.max_exact_size = 100_000_000  # 限制精确集大小

        self.stats = {
            'total': 0,
            'bloom_filtered': 0,
            'exact_deduped': 0,
            'new_items': 0
        }

    def is_new(self, item_id: str) -> bool:
        """
        判断是否为新的唯一记录
        返回True表示是新数据，需要处理
        返回False表示是重复数据，可丢弃
        """
        self.stats['total'] += 1

        # 第一级：布隆过滤器
        if item_id not in self.bloom:
            # 肯定不存在，是新数据
            self.bloom.add(item_id)
            self.stats['new_items'] += 1
            return True

        # 可能存在，需要精确判断
        self.stats['bloom_filtered'] += 1

        # 第二级：精确哈希表
        if item_id in self.exact_set:
            # 确认是重复
            self.stats['exact_deduped'] += 1
            return False

        # 布隆过滤器误判，实际是新数据
        if len(self.exact_set) < self.max_exact_size:
            self.exact_set.add(item_id)
        self.bloom.add(item_id)
        self.stats['new_items'] += 1
        return True

    def get_stats(self) -> dict:
        return {
            **self.stats,
            'dedup_rate': self.stats['bloom_filtered'] / max(1, self.stats['total']),
            'memory_mb': len(self.bloom.bit_array) / (1024 * 1024) +
                        len(self.exact_set) * 50 / (1024 * 1024)
        }
```
