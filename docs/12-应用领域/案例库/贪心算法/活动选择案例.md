## 应用场景1：会议室调度系统

### 问题描述

- **背景**：企业会议室资源有限，需要自动安排会议时间，最大化会议室利用率
- **问题**：
  - 多个会议请求，每个有开始和结束时间
  - 同一时间一个会议室只能容纳一个会议
  - 尽可能安排更多会议
- **约束**：
  - 会议室数量：10-50间
  - 每日会议请求：100-500个
  - 会议时长：30分钟-4小时
- **数据规模**：大型企业日均500+会议请求

### 算法选择理由

- **为什么选择贪心算法**：
  - 活动选择问题贪心策略可保证最优解
  - 按结束时间排序，每次选择最早结束的
  - 时间复杂度O(n log n)，效率高
  - 实现简单，易于维护

- **算法对比**：

  | 算法 | 时间复杂度 | 最优性 | 实现难度 | 适用场景 |
  |------|-----------|--------|----------|----------|
  | 暴力枚举 | O(2^n) | 是 | 低 | n < 20 |
  | 动态规划 | O(n^2) | 是 | 中 | 带权重 |
  | 贪心 | **O(n log n)** | **是** | **低** | **标准问题** |

### 解决方案

```pseudo
Algorithm GreedyActivitySelector(activities):
    // activities: [(start, end, id), ...]

    // 按结束时间排序
    sorted = sort(activities, by=end_time)

    selected = [sorted[0]]
    last_end = sorted[0].end

    for i from 1 to len(sorted)-1:
        if sorted[i].start >= last_end:
            selected.append(sorted[i])
            last_end = sorted[i].end

    return selected

// 多会议室版本
Algorithm MultiRoomScheduler(activities, room_count):
    sorted = sort(activities, by=start_time)

    // 最小堆跟踪各会议室最后结束时间
    rooms = MinHeap()  // (last_end_time, room_id)
    for i from 0 to room_count-1:
        rooms.push((0, i))

    schedule = {}  // room_id -> [activities]
    unscheduled = []

    for activity in sorted:
        earliest_room = rooms.peek()

        if earliest_room.end <= activity.start:
            // 有可用的会议室
            rooms.pop()
            rooms.push((activity.end, earliest_room.id))
            schedule[earliest_room.id].append(activity)
        else if rooms.size() < room_count:
            // 使用新会议室
            new_room_id = rooms.size()
            rooms.push((activity.end, new_room_id))
            schedule[new_room_id] = [activity]
        else:
            // 无法安排
            unscheduled.append(activity)

    return schedule, unscheduled
```

### 实际代码示例（Python）

```python
from typing import List, Tuple, NamedTuple
import heapq
from datetime import datetime, timedelta
from dataclasses import dataclass, field

@dataclass
class Meeting:
    id: str
    title: str
    start: datetime
    end: datetime
    attendees: int
    priority: int = 1  # 1-5，用于加权选择
    organizer: str = ""

    def duration(self) -> int:
        """返回分钟数"""
        return int((self.end - self.start).total_seconds() / 60)

    def conflicts_with(self, other: 'Meeting') -> bool:
        """检查是否时间冲突"""
        return not (self.end <= other.start or self.start >= other.end)

@dataclass
class Room:
    id: str
    name: str
    capacity: int
    features: List[str] = field(default_factory=list)

class MeetingScheduler:
    """会议室调度器"""

    def __init__(self, rooms: List[Room]):
        self.rooms = rooms

    def schedule_single_room(self, requests: List[Meeting]) -> List[Meeting]:
        """
        单会议室贪心调度
        返回可安排的最大会议集合
        """
        if not requests:
            return []

        # 按结束时间排序（贪心选择）
        sorted_requests = sorted(requests, key=lambda m: (m.end, -m.priority))

        selected = [sorted_requests[0]]
        last_end = sorted_requests[0].end

        for meeting in sorted_requests[1:]:
            if meeting.start >= last_end:
                selected.append(meeting)
                last_end = meeting.end

        return selected

    def schedule_multi_room(self, requests: List[Meeting]) -> dict:
        """
        多会议室调度
        返回: {room_id: [meetings]}
        """
        if not requests:
            return {}

        # 按开始时间排序
        sorted_requests = sorted(requests, key=lambda m: m.start)

        # 最小堆: (结束时间, 房间索引)
        room_heap = []
        room_schedules = {i: [] for i in range(len(self.rooms))}
        unscheduled = []

        for meeting in sorted_requests:
            # 找最早空闲的会议室
            if room_heap and room_heap[0][0] <= meeting.start:
                end_time, room_idx = heapq.heappop(room_heap)

                # 检查容量
                if self.rooms[room_idx].capacity >= meeting.attendees:
                    room_schedules[room_idx].append(meeting)
                    heapq.heappush(room_heap, (meeting.end, room_idx))
                else:
                    # 放回并尝试其他房间
                    heapq.heappush(room_heap, (end_time, room_idx))
                    self._try_find_room(meeting, room_heap, room_schedules, unscheduled)
            else:
                # 所有房间都忙，尝试找空闲房间
                self._try_find_room(meeting, room_heap, room_schedules, unscheduled)

        # 转换为房间名称
        return {self.rooms[i].name: schedule for i, schedule in room_schedules.items() if schedule}

    def _try_find_room(self, meeting: Meeting, room_heap: list,
                       room_schedules: dict, unscheduled: list):
        """尝试为会议找一个可用房间"""
        for room_idx in range(len(self.rooms)):
            if self.rooms[room_idx].capacity < meeting.attendees:
                continue

            # 检查是否与该房间已安排会议冲突
            conflicts = any(m.conflicts_with(meeting) for m in room_schedules[room_idx])

            if not conflicts:
                room_schedules[room_idx].append(meeting)
                heapq.heappush(room_heap, (meeting.end, room_idx))
                return

        unscheduled.append(meeting)

    def schedule_weighted(self, requests: List[Meeting]) -> List[Meeting]:
        """
        带权重的会议选择
        当优先级不同时，使用动态规划
        """
        if not requests:
            return []

        # 按结束时间排序
        sorted_requests = sorted(requests, key=lambda m: m.end)
        n = len(sorted_requests)

        # dp[i] = 前i个会议的最大权重
        dp = [0] * (n + 1)

        for i in range(1, n + 1):
            # 找到最后一个不冲突的会议
            last_compatible = 0
            for j in range(i - 1, 0, -1):
                if sorted_requests[j - 1].end <= sorted_requests[i - 1].start:
                    last_compatible = j
                    break

            # 选择：包含当前会议 或 不包含
            dp[i] = max(dp[i - 1],
                       dp[last_compatible] + sorted_requests[i - 1].priority)

        # 回溯找出选择的会议
        selected = []
        i = n
        while i > 0:
            if dp[i] != dp[i - 1]:
                selected.append(sorted_requests[i - 1])
                # 跳到上一个兼容位置
                for j in range(i - 1, 0, -1):
                    if sorted_requests[j - 1].end <= sorted_requests[i - 1].start:
                        i = j
                        break
                else:
                    i = 0
            else:
                i -= 1

        return list(reversed(selected))

    def print_schedule(self, schedule: dict):
        """打印调度结果"""
        total_meetings = sum(len(meetings) for meetings in schedule.values())

        print(f"\n会议室调度结果:")
        print(f"总会议室数: {len(self.rooms)}")
        print(f"成功安排: {total_meetings} 个会议")
        print()

        for room_name, meetings in schedule.items():
            if not meetings:
                continue

            print(f"{room_name} ({len(meetings)}个会议):")
            for m in meetings:
                print(f"  {m.start.strftime('%H:%M')}-{m.end.strftime('%H:%M')} "
                      f"| {m.title} | {m.attendees}人")
            print()


# 基准测试
def benchmark_scheduler():
    """测试调度器性能"""
    # 创建会议室
    rooms = [
        Room("R1", "会议室A", 20),
        Room("R2", "会议室B", 50),
        Room("R3", "会议室C", 10),
    ]

    # 生成随机会议请求
    import random
    requests = []
    base_time = datetime.now().replace(hour=9, minute=0, second=0, microsecond=0)

    for i in range(100):
        start_offset = random.randint(0, 480)  # 9:00-17:00
        duration = random.choice([30, 60, 90, 120])

        start = base_time + timedelta(minutes=start_offset)
        end = start + timedelta(minutes=duration)

        requests.append(Meeting(
            id=f"M{i:03d}",
            title=f"会议{i}",
            start=start,
            end=end,
            attendees=random.randint(5, 40),
            priority=random.randint(1, 5)
        ))

    scheduler = MeetingScheduler(rooms)

    # 单会议室贪心
    import time
    start = time.time()
    single_schedule = scheduler.schedule_single_room(requests)
    single_time = time.time() - start

    print(f"单会议室调度:")
    print(f"  安排会议数: {len(single_schedule)}")
    print(f"  耗时: {single_time*1000:.2f}ms")

    # 多会议室调度
    start = time.time()
    multi_schedule = scheduler.schedule_multi_room(requests)
    multi_time = time.time() - start

    print(f"\n多会议室调度:")
    scheduler.print_schedule(multi_schedule)
    print(f"  耗时: {multi_time*1000:.2f}ms")

    # 带权调度
    start = time.time()
    weighted_schedule = scheduler.schedule_weighted(requests)
    weighted_time = time.time() - start

    total_priority = sum(m.priority for m in weighted_schedule)
    print(f"\n带权调度:")
    print(f"  安排会议数: {len(weighted_schedule)}")
    print(f"  总优先级: {total_priority}")
    print(f"  耗时: {weighted_time*1000:.2f}ms")

if __name__ == '__main__':
    benchmark_scheduler()
```

## 应用场景2：任务调度与CPU调度

### 问题描述

- **背景**：操作系统需要调度CPU上的任务，最小化平均等待时间
- **问题**：
  - 多个任务，每个有执行时间和截止时间
  - 最小化平均周转时间
  - 避免任务饿死

### 解决方案

```python
from dataclasses import dataclass
from typing import List
import heapq

@dataclass
class Task:
    id: str
    burst_time: int  # 执行时间
    arrival_time: int
    deadline: int = float('inf')
    priority: int = 0

def sjf_schedule(tasks: List[Task]) -> List[str]:
    """
    最短作业优先(SJF)调度
    贪心策略：总是选择执行时间最短的任务
    """
    tasks = sorted(tasks, key=lambda t: (t.arrival_time, t.burst_time))

    schedule = []
    current_time = 0
    ready_queue = []  # (burst_time, task_id)
    task_idx = 0

    while task_idx < len(tasks) or ready_queue:
        # 将到达的任务加入就绪队列
        while task_idx < len(tasks) and tasks[task_idx].arrival_time <= current_time:
            heapq.heappush(ready_queue, (tasks[task_idx].burst_time, tasks[task_idx].id))
            task_idx += 1

        if ready_queue:
            burst, task_id = heapq.heappop(ready_queue)
            schedule.append(task_id)
            current_time += burst
        else:
            # 空闲，跳到下一个任务到达
            if task_idx < len(tasks):
                current_time = tasks[task_idx].arrival_time

    return schedule

# 测试
if __name__ == '__main__':
    tasks = [
        Task("T1", 6, 0),
        Task("T2", 8, 1),
        Task("T3", 7, 2),
        Task("T4", 3, 3),
    ]

    schedule = sjf_schedule(tasks)
    print("SJF调度顺序:", schedule)
```
