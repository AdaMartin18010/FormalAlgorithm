## 算法层面优化

### 1. 选择合适的算法

**核心原则**：降低时间复杂度比优化常数更重要。

```python
# O(n²) 算法

def find_duplicates_n2(arr):
    """O(n²) 的双重循环查找重复"""
    duplicates = []
    for i in range(len(arr)):
        for j in range(i + 1, len(arr)):
            if arr[i] == arr[j]:
                duplicates.append(arr[i])
    return duplicates

# O(n) 算法

def find_duplicates_n(arr):
    """O(n) 的哈希表查找重复"""
    seen = set()
    duplicates = set()
    for x in arr:
        if x in seen:
            duplicates.add(x)
        else:
            seen.add(x)
    return list(duplicates)
```

**复杂度对比**：

| 复杂度 | n=10 | n=100 | n=1000 | n=10⁶ |
|--------|------|-------|--------|-------|
| O(log n) | 3 | 7 | 10 | 20 |
| O(n) | 10 | 100 | 1000 | 10⁶ |
| O(n log n) | 30 | 700 | 10⁴ | 2×10⁷ |
| O(n²) | 100 | 10⁴ | 10⁶ | 10¹² |

def fibonacci_memo(n, memo=None):
    if memo is None:
        memo = {}
    if n in memo:
        return memo[n]
    if n <= 1:
        return n
    memo[n] = fibonacci_memo(n-1, memo) + fibonacci_memo(n-2, memo)
    return memo[n]

from functools import lru_cache

@lru_cache(maxsize=None)
def fibonacci_lru(n):

## 3. 预处理与缓存

### 3. 预处理与缓存

```python
class PreprocessedArray:
    def __init__(self, arr):
        self.arr = arr
        self.n = len(arr)
        # 预处理前缀和
        self.prefix = [0] * (self.n + 1)
        for i in range(self.n):
            self.prefix[i + 1] = self.prefix[i] + arr[i]

        # 预处理稀疏表（RMQ）
        self._build_sparse_table()

    def range_sum(self, l, r):
        """O(1) 区间查询"""
        return self.prefix[r + 1] - self.prefix[l]

    def _build_sparse_table(self):
        """构建稀疏表用于 RMQ"""
        import math
        k = int(math.log2(self.n)) + 1
        self.st = [[0] * k for _ in range(self.n)]

        for i in range(self.n):
            self.st[i][0] = self.arr[i]

        j = 1
        while (1 << j) <= self.n:
            for i in range(self.n - (1 << j) + 1):
                self.st[i][j] = min(self.st[i][j-1],
                                    self.st[i + (1 << (j-1))][j-1])
            j += 1

    def range_min(self, l, r):
        """O(1) 区间最小值查询"""
        import math
        j = int(math.log2(r - l + 1))
        return min(self.st[l][j], self.st[r - (1 << j) + 1][j])
```

### 4. 分治法优化

```python
# 普通递归

def power(x, n):
    """O(n) 计算 x^n"""
    if n == 0:
        return 1
    return x * power(x, n - 1)

# 快速幂 O(log n)

def fast_power(x, n):
    """O(log n) 计算 x^n"""
    if n == 0:
        return 1
    half = fast_power(x, n // 2)
    if n % 2 == 0:
        return half * half
    else:
        return half * half * x

# 迭代版本

def fast_power_iter(x, n, mod=None):
    result = 1
    base = x
    while n > 0:
        if n & 1:
            result = result * base
            if mod:
                result %= mod
        base = base * base
        if mod:
            base %= mod
        n >>= 1
    return result
```

## 3. 自定义数据结构

### 3. 自定义数据结构

```python
class DisjointSet:
    """并查集：接近 O(1) 的合并和查询"""
    def __init__(self, n):
        self.parent = list(range(n))
        self.rank = [0] * n

    def find(self, x):
        if self.parent[x] != x:
            self.parent[x] = self.find(self.parent[x])  # 路径压缩
        return self.parent[x]

    def union(self, x, y):
        px, py = self.find(x), self.find(y)
        if px == py:
            return False
        # 按秩合并
        if self.rank[px] < self.rank[py]:
            px, py = py, px
        self.parent[py] = px
        if self.rank[px] == self.rank[py]:
            self.rank[px] += 1
        return True
```

# 优化 1：减少循环内部计算

n = len(arr)

result = [x * 2 for x in arr]

# 低效
for i in range(len(my_dict)):

value = my_dict.get(key)  # 只查找一次
for i in range(len(my_dict)):
    use(value)

## 4. 使用生成器

### 4. 使用生成器

```python
# 优化 1：惰性求值

# 低效：创建完整列表
def get_squares(n):
    return [x**2 for x in range(n)]

# 高效：生成器
def get_squares_gen(n):
    for x in range(n):
        yield x**2

# 使用生成器表达式
squares = (x**2 for x in range(n))

# 优化 2：处理大文件

def process_large_file(filename):
    # 低效：读取全部内容
    with open(filename) as f:
        lines = f.readlines()
    for line in lines:
        process(line)

    # 高效：逐行读取
    with open(filename) as f:
        for line in f:
            process(line)
```

## 3. 缓存优化

### 3. 缓存优化

```python
from functools import lru_cache
import functools

# 优化 1：函数结果缓存

@lru_cache(maxsize=128)
def fibonacci(n):
    if n < 2:
        return n
    return fibonacci(n-1) + fibonacci(n-2)

# 优化 2：自定义缓存策略

def memoize(maxsize=128):
    def decorator(func):
        cache = {}
        @functools.wraps(func)
        def wrapper(*args):
            if args in cache:
                return cache[args]
            result = func(*args)
            if len(cache) >= maxsize:
                # LRU 策略：移除最旧的
                oldest = next(iter(cache))
                del cache[oldest]
            cache[args] = result
            return result
        return wrapper
    return decorator
```

def fetch_url(url):

@jit(nopython=True)
def matrix_multiply(A, B):
    m, n = A.shape
    n2, p = B.shape
    C = np.zeros((m, p))
    for i in range(m):

## 2. 性能分析

### 2. 性能分析

```python
import cProfile
import pstats
import io

def profile_function(func, *args, **kwargs):
    """性能分析函数"""
    profiler = cProfile.Profile()
    profiler.enable()
    result = func(*args, **kwargs)
    profiler.disable()

    # 输出统计
    s = io.StringIO()
    stats = pstats.Stats(profiler, stream=s)
    stats.sort_stats('cumulative')
    stats.print_stats(20)
    print(s.getvalue())

    return result
```
