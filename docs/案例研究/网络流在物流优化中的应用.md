## 概述

网络流问题是运筹学和计算机科学的经典课题，在物流优化中有着广泛应用。从供应链网络设计到最后一公里配送，从城市交通规划到航空网络优化，网络流算法帮助企业在复杂约束下做出最优决策。

## 1. 供应链优化

### 1.1 问题背景与动机

**场景**：

- 亚马逊全球物流网络：200+履行中心，服务数百万客户
- 丰田精益供应链：JIT生产要求精确的时间窗口
- 冷链物流：药品/食品的温度敏感配送

**业务价值**：

- 物流成本占企业总成本的10-30%
- 网络优化可节省成本15-25%
- 库存持有成本与缺货成本的平衡

**挑战**：

- 多层级网络（供应商→工厂→仓库→客户）
- 复杂约束（容量、时间窗、多商品）
- 不确定性（需求波动、运输延迟）
- 多目标优化（成本、时效、可持续性）

### 1.3 实现细节

#### 最小费用流算法实现

```python
import heapq
from typing import Dict, List, Tuple
from dataclasses import dataclass

@dataclass
class Edge:
    to: int
    rev: int
    capacity: float
    cost: float
    original_cap: float

class MinCostFlow:
    """
    最小费用最大流算法（连续最短路算法）
    基于势函数的Johnson算法优化，支持负权边
    """

    def __init__(self, n: int):
        self.n = n
        self.graph = [[] for _ in range(n)]
        self.INF = float('inf')

    def add_edge(self, fr: int, to: int, cap: float, cost: float):
        forward = Edge(to, len(self.graph[to]), cap, cost, cap)
        backward = Edge(fr, len(self.graph[fr]), 0, -cost, 0)
        self.graph[fr].append(forward)
        self.graph[to].append(backward)

    def flow(self, s: int, t: int, maxf: float = float('inf')) -> Tuple[float, float]:
        n = self.n
        res = 0.0
        flow = 0.0
        potential = [0.0] * n

        while flow < maxf:
            dist = [self.INF] * n
            dist[s] = 0
            prevv = [-1] * n
            preve = [-1] * n

            pq = [(0, s)]
            while pq:
                d, v = heapq.heappop(pq)
                if dist[v] < d:
                    continue
                for i, e in enumerate(self.graph[v]):
                    if e.capacity > 0:
                        ndist = dist[v] + e.cost + potential[v] - potential[e.to]
                        if dist[e.to] > ndist:
                            dist[e.to] = ndist
                            prevv[e.to] = v
                            preve[e.to] = i
                            heapq.heappush(pq, (ndist, e.to))

            if dist[t] == self.INF:
                break

            for v in range(n):
                if dist[v] < self.INF:
                    potential[v] += dist[v]

            d = maxf - flow
            v = t
            while v != s:
                d = min(d, self.graph[prevv[v]][preve[v]].capacity)
                v = prevv[v]

            flow += d
            res += d * potential[t]

            v = t
            while v != s:
                e = self.graph[prevv[v]][preve[v]]
                e.capacity -= d
                self.graph[v][e.rev].capacity += d
                v = prevv[v]

        return flow, res
```

class MultiCommodityFlow:
    """
    多商品流问题求解 - 列生成方法
    主问题：选择路径组合
    子问题：最短路径（为每种商品）
    """

    def __init__(self, network, commodities: List[Dict]):
        self.network = network
        self.commodities = commodities
        self.paths = {k: [] for k in range(len(commodities))}

    def solve_with_column_generation(self) -> Dict:
        iteration = 0
        while True:
            iteration += 1
            solution = self._solve_restricted_master_problem()
            dual_capacities = solution['dual_capacities']

            new_paths_added = 0
            for k, comm in enumerate(self.commodities):
                # 修改边权
                for u, v, data in self.network.edges(data=True):
                    data['reduced_cost'] = data['cost'] - dual_capacities.get((u, v), 0)

                # 找最短路径
                try:
                    path = nx.shortest_path(
                        self.network,
                        comm['origin'],
                        comm['dest'],
                        weight='reduced_cost'
                    )
                    path_cost = sum(
                        self.network[path[i]][path[i+1]]['reduced_cost']
                        for i in range(len(path)-1)
                    )

                    if path_cost < -1e-6:
                        self.paths[k].append(path)
                        new_paths_added += 1
                except:
                    pass

            if new_paths_added == 0:
                break

        return solution

### 2.2 实现细节

#### Contraction Hierarchies实现

```python
import heapq
from typing import List, Dict, Tuple, Set

class ContractionHierarchies:
    """
    Contraction Hierarchies (CH) 算法
    核心思想：
    1. 离线：按重要性排序节点，依次"收缩"（添加shortcut边）
    2. 在线：双向Dijkstra只向上搜索（到更高层级节点）
    """

    def __init__(self, graph):
        self.G = graph
        self.n = graph.number_of_nodes()
        self.rank = {}
        self.up_graph = {}
        self.down_graph = {}

    def preprocess(self):
        """预处理：节点收缩"""
        edge_difference = {}
        contracted = set()

        for v in self.G.nodes():
            edge_difference[v] = self._compute_edge_difference(v, set())

        importance_queue = [(edge_difference[v], v) for v in self.G.nodes()]
        heapq.heapify(importance_queue)

        current_rank = 0

        while importance_queue:
            _, v = heapq.heappop(importance_queue)
            if v in contracted:
                continue

            new_ed = self._compute_edge_difference(v, contracted)
            if importance_queue and new_ed > importance_queue[0][0]:
                heapq.heappush(importance_queue, (new_ed, v))
                continue

            self.rank[v] = current_rank
            current_rank += 1
            contracted.add(v)

            self._add_shortcuts(v, contracted)

        self._build_query_graphs()

    def query(self, source: int, target: int) -> Tuple[float, List[int]]:
        """双向查询（只沿层级向上搜索）"""
        INF = float('inf')

        dist_forward = {v: INF for v in self.G.nodes()}
        dist_forward[source] = 0
        prev_forward = {}
        pq_forward = [(0, source)]

        dist_backward = {v: INF for v in self.G.nodes()}
        dist_backward[target] = 0
        prev_backward = {}
        pq_backward = [(0, target)]

        best_distance = INF
        meeting_node = None

        while pq_forward or pq_backward:
            if pq_forward:
                d, u = heapq.heappop(pq_forward)
                if d > dist_forward[u]:
                    continue

                if dist_backward[u] < INF:
                    total = d + dist_backward[u]
                    if total < best_distance:
                        best_distance = total
                        meeting_node = u

                for v, w in self.up_graph[u]:
                    nd = d + w
                    if nd < dist_forward[v]:
                        dist_forward[v] = nd
                        prev_forward[v] = u
                        heapq.heappush(pq_forward, (nd, v))

            if pq_backward:
                d, u = heapq.heappop(pq_backward)
                if d > dist_backward[u]:
                    continue

                if dist_forward[u] < INF:
                    total = d + dist_forward[u]
                    if total < best_distance:
                        best_distance = total
                        meeting_node = u

                for v, w in self.up_graph[u]:
                    nd = d + w
                    if nd < dist_backward[v]:
                        dist_backward[v] = nd
                        prev_backward[v] = u
                        heapq.heappush(pq_backward, (nd, v))

        # 重构路径
        if meeting_node is None:
            return INF, []

        path = []
        v = meeting_node
        while v in prev_forward:
            path.append(v)
            v = prev_forward[v]
        path.append(source)
        path.reverse()

        v = meeting_node
        while v in prev_backward:
            v = prev_backward[v]
            path.append(v)

        return best_distance, path
```

## 3. 航班调度优化

### 3.1 问题背景与动机

**场景**：

- 航空公司机队分配与航线规划
- 机组人员排班（满足法规约束）
- 航班恢复（应对延误/取消）
- 机场停机位分配

**业务价值**：

- 航空公司运营成本的10-15%可通过优化节省
- 机组成本是第二大运营支出（仅次于燃油）
- 延误成本：每分钟$100-200（窄体机）

### 3.2 算法选择原理

#### 机组排班问题（Crew Pairing + Rostering）

```
阶段1：航班配对（Pairing）
- 输入：航班时刻表
- 生成合法的工作序列（满足法规）
- 成本最小化（飞行时间、酒店、过站）
- 输出：Pairing集合

阶段2：机组分配（Rostering）
- 输入：Pairing集合 + 机组人员
- 分配Pairing给具体人员
- 考虑资质、偏好、公平性
- 输出：个人排班表

方法：列生成 + 约束规划 + 启发式
```

### 3.3 实现细节

#### 航班配对优化（列生成）

```python
class CrewPairingOptimizer:
    """
    机组配对优化：列生成方法
    主问题：选择pairings覆盖所有航班
    子问题：生成负约化成本的pairing（资源约束最短路径）
    """

    def __init__(self, flights: List[Dict], rules: Dict):
        self.flights = {f['id']: f for f in flights}
        self.rules = rules
        self.pairings = []

    def solve(self) -> Dict:
        """列生成主循环"""
        self.generate_initial_pairings()

        iteration = 0
        while True:
            iteration += 1

            # 1. 求解限制主问题（RMP）
            rmp_solution = self._solve_restricted_master_problem()
            duals = rmp_solution['duals']

            # 2. 定价子问题
            new_pairings = self._solve_pricing_problem(duals)

            if not new_pairings:
                break

            self.pairings.extend(new_pairings)

        return self._solve_integer_program()

    def _solve_pricing_problem(self, duals: Dict) -> List[Dict]:
        """
        定价子问题：资源约束最短路径
        使用标签设定算法（Label Setting）
        """
        network = self._build_connection_network()
        new_pairings = []

        for base in self.rules['bases']:
            labels = self._label_setting_algorithm(network, base, duals)

            for label in labels:
                if label['reduced_cost'] < -1e-6:
                    pairing = self._label_to_pairing(label)
                    new_pairings.append(pairing)

        return new_pairings[:10]

    def _label_setting_algorithm(self, network, source: str, duals: Dict) -> List[Dict]:
        """
        多准则最短路径的标签设定算法
        """
        from collections import deque

        labels = {node: [] for node in network.nodes()}
        initial_label = {
            'cost': 0,
            'flight_time': 0,
            'duty_time': 0,
            'flights': [],
            'reduced_cost': 0,
            'node': source
        }
        labels[source].append(initial_label)

        queue = deque([initial_label])

        while queue:
            current = queue.popleft()

            for successor in network.successors(current['node']):
                edge = network[current['node']][successor]
                flight = edge['flight']

                new_label = {
                    'node': successor,
                    'cost': current['cost'] + edge['cost'],
                    'flight_time': current['flight_time'] + flight['duration'],
                    'duty_time': current['duty_time'] + flight['duration'] + edge['sit_time'],
                    'flights': current['flights'] + [flight['id']],
                    'reduced_cost': (current['reduced_cost'] + edge['cost'] -
                                   duals.get(flight['id'], 0))
                }

                if not self._check_resource_constraints(new_label):
                    continue

                if not self._is_dominated(new_label, labels[successor]):
                    labels[successor] = [l for l in labels[successor]
                                        if not self._dominates(new_label, l)]
                    labels[successor].append(new_label)
                    queue.append(new_label)

        result = []
        for node, node_labels in labels.items():
            if node in self.rules['bases']:
                result.extend(node_labels)

        return result

    def _check_resource_constraints(self, label: Dict) -> bool:
        """检查法规约束"""
        if label['flight_time'] > self.rules['max_flight_time']:
            return False
        if label['duty_time'] > self.rules['max_duty_time']:
            return False
        return True
```
