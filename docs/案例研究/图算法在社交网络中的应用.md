#### 相似度度量对比

| 算法 | 公式 | 时间复杂度 | 适用场景 |
|-----|------|-----------|---------|
| 共同邻居 (CN) | \|N(u) ∩ N(v)\| | O(d_u + d_v) | 基础特征 |
| Jaccard系数 | CN / \|N(u) ∪ N(v)\| | O(d_u + d_v) | 度差异大时 |
| Adamic-Adar | Σ 1/log\|N(z)\| | O(CN) | 降低高度数节点影响 |

| 偏好依附 | d_u × d_v | O(1) | 快速baseline |

#### 1.3.1 基于共同邻居的快速计算

```python
import numpy as np
from collections import defaultdict
from typing import List, Set, Dict, Tuple
import heapq

class CommonNeighborsRecommender:
    """
    优化的共同邻居计算
    采用"小表驱动"策略减少计算量
    """

    def __init__(self, adj_list: Dict[int, Set[int]]):
        self.adj = adj_list
        # 预计算节点度数并排序邻居
        self.sorted_neighbors = {
            node: sorted(neighbors)
            for node, neighbors in adj_list.items()
        }
        # 按度数排序的节点列表（用于小表驱动）
        self.nodes_by_degree = sorted(
            adj_list.keys(),
            key=lambda x: len(adj_list[x])
        )

    def common_neighbors_count(self, u: int, v: int) -> int:
        """
        交集计数：遍历度数较小的邻居集合
        时间复杂度：O(min(d_u, d_v))
        """
        neighbors_u = self.adj[u]
        neighbors_v = self.adj[v]

        # 选择较小的集合进行遍历
        if len(neighbors_u) > len(neighbors_v):
            neighbors_u, neighbors_v = neighbors_v, neighbors_u

        count = 0
        for w in neighbors_u:
            if w in neighbors_v:
                count += 1
        return count

    def adamic_adar_score(self, u: int, v: int) -> float:
        """
        Adamic-Adar指数：降低高度数共同邻居的权重
        AA(u,v) = Σ_{z∈N(u)∩N(v)} 1/log|N(z)|
        """
        common = self.adj[u] & self.adj[v]
        score = 0.0
        for z in common:
            deg_z = len(self.adj[z])
            if deg_z > 1:  # 避免log(1)=0
                score += 1.0 / np.log(deg_z)
        return score

    def recommend_batch(self, user: int, top_k: int = 100) -> List[Tuple[int, float]]:
        """
        为单个用户批量生成推荐
        优化策略：
        1. 只考虑2-hop邻居（朋友的朋友）
        2. 使用min-heap维护top-k
        """
        # 已连接的邻居
        friends = self.adj[user]

        # 候选集：朋友的朋友
        candidates = defaultdict(float)
        for friend in friends:
            for fof in self.adj[friend]:
                if fof != user and fof not in friends:
                    # 累加共同邻居分数
                    candidates[fof] += 1.0 / np.log(len(self.adj[friend]) + 2)

        # 使用heapq获取top-k
        return heapq.nlargest(top_k, candidates.items(), key=lambda x: x[1])


class ApproximateCNRecommender(CommonNeighborsRecommender):
    """
    基于MinHash的近似共同邻居计算
    适用于超大规模图
    """

    def __init__(self, adj_list: Dict[int, Set[int]], num_hashes: int = 128):
        super().__init__(adj_list)
        self.num_hashes = num_hashes
        self.minhash_signatures = self._compute_minhash_signatures()

    def _compute_minhash_signatures(self) -> Dict[int, np.ndarray]:
        """
        为每个节点计算MinHash签名
        将邻居集合压缩为固定长度的签名
        """
        np.random.seed(42)
        # 生成随机哈希函数参数
        hash_params = np.random.randint(1, 100000, size=(self.num_hashes, 2))

        signatures = {}
        for node, neighbors in self.adj.items():
            sig = np.full(self.num_hashes, np.inf)
            for neighbor in neighbors:
                # 多个哈希函数
                for i, (a, b) in enumerate(hash_params):
                    hash_val = (a * neighbor + b) % 1000003
                    sig[i] = min(sig[i], hash_val)
            signatures[node] = sig
        return signatures

    def approx_jaccard(self, u: int, v: int) -> float:
        """
        估计Jaccard相似度：签名相同位置的比例
        """
        sig_u = self.minhash_signatures[u]
        sig_v = self.minhash_signatures[v]
        return np.mean(sig_u == sig_v)
```

#### 1.3.2 基于随机游走的Node2Vec

```python
import random
from gensim.models import Word2Vec

class Node2VecRecommender:
    """
    Node2Vec：通过有偏随机游走学习节点嵌入
    平衡BFS（社区结构）和DFS（角色相似）
    """

    def __init__(self, adj_list: Dict[int, List[int]],
                 dimensions: int = 128,
                 walk_length: int = 80,
                 num_walks: int = 10,
                 p: float = 1.0,  # 返回参数
                 q: float = 1.0): # 进出参数
        self.adj = adj_list
        self.dimensions = dimensions
        self.walk_length = walk_length
        self.num_walks = num_walks
        self.p = p
        self.q = q

    def node2vec_walk(self, start_node: int) -> List[int]:
        """
        生成一条有偏随机游走

        转移概率：
        π_{vx} = α_{pq}(t,x) · w_{vx}

        其中α定义如下（d_{tx}为t到x的距离）：
        - d=0（返回t）：α = 1/p
        - d=1（距离t一步）：α = 1
        - d=2（远离t）：α = 1/q
        """
        walk = [start_node]

        while len(walk) < self.walk_length:
            cur = walk[-1]
            neighbors = self.adj[cur]

            if len(neighbors) == 0:
                break

            if len(walk) == 1:
                # 第一步：均匀采样
                next_node = random.choice(neighbors)
            else:
                # 有偏采样
                prev = walk[-2]
                next_node = self._biased_sample(prev, cur, neighbors)

            walk.append(next_node)

        return walk

    def _biased_sample(self, prev: int, cur: int, neighbors: List[int]) -> int:
        """
        根据Node2Vec偏置进行采样
        """
        weights = []
        for neighbor in neighbors:
            if neighbor == prev:
                # 返回前节点：权重 = 1/p
                weight = 1.0 / self.p
            elif self._is_neighbor(prev, neighbor):
                # 距离前节点1步（BFS）：权重 = 1
                weight = 1.0
            else:
                # 远离前节点（DFS）：权重 = 1/q
                weight = 1.0 / self.q
            weights.append(weight)

        # 归一化并采样
        total = sum(weights)
        probs = [w / total for w in weights]
        return random.choices(neighbors, weights=probs, k=1)[0]

    def _is_neighbor(self, u: int, v: int) -> bool:
        return v in self.adj[u]

    def train(self):
        """
        训练Node2Vec模型
        """
        print("Generating random walks...")
        walks = []
        nodes = list(self.adj.keys())

        for _ in range(self.num_walks):
            random.shuffle(nodes)
            for node in nodes:
                walk = self.node2vec_walk(node)
                # 转换为字符串（Word2Vec要求）
                walks.append([str(n) for n in walk])

        print("Training Word2Vec...")
        self.model = Word2Vec(
            sentences=walks,
            vector_size=self.dimensions,
            window=10,
            min_count=1,
            sg=1,  # Skip-gram
            workers=8,
            epochs=5
        )

    def get_similar_users(self, user: int, top_k: int = 100) -> List[Tuple[str, float]]:
        """
        基于嵌入相似度推荐
        """
        user_str = str(user)
        if user_str not in self.model.wv:
            return []

        # 排除已连接的用户
        existing = set(str(n) for n in self.adj[user])

        similar = self.model.wv.most_similar(
            positive=[user_str],
            topn=top_k + len(existing)
        )

        # 过滤已连接
        return [(u, score) for u, score in similar if u not in existing][:top_k]
```

#### 1.3.3 深度学习推荐模型

```python
import torch
import torch.nn as nn
import torch.nn.functional as F

class GraphSAGE(nn.Module):
    """
    GraphSAGE：归纳式图神经网络
    支持 unseen 节点的表示学习
    """

    def __init__(self, input_dim: int, hidden_dim: int, output_dim: int,
                 num_layers: int = 2, aggregator: str = 'mean'):
        super().__init__()
        self.num_layers = num_layers

        # 聚合器层
        self.layers = nn.ModuleList()
        dims = [input_dim] + [hidden_dim] * (num_layers - 1) + [output_dim]

        for i in range(num_layers):
            self.layers.append(SAGELayer(dims[i], dims[i+1], aggregator))

    def forward(self, x: torch.Tensor, adj_list: List[List[int]],
                batch_nodes: List[int]) -> torch.Tensor:
        """
        前向传播

        Args:
            x: 节点特征矩阵 [N, input_dim]
            adj_list: 邻接表
            batch_nodes: 当前批次的节点
        """
        hidden = x

        for layer in self.layers:
            new_hidden = torch.zeros_like(hidden)
            for node in batch_nodes:
                neighbors = adj_list[node]
                if len(neighbors) > 0:
                    neighbor_feats = hidden[neighbors]
                    new_hidden[node] = layer(hidden[node], neighbor_feats)
                else:
                    new_hidden[node] = layer(hidden[node], hidden[node].unsqueeze(0))
            hidden = new_hidden

        return hidden


class SAGELayer(nn.Module):
    """
    GraphSAGE单层：
    h_v^k = σ(W^k · CONCAT(h_v^{k-1}, AGG({h_u^{k-1}, ∀u∈N(v)})))
    """

    def __init__(self, input_dim: int, output_dim: int, aggregator: str = 'mean'):
        super().__init__()
        self.aggregator = aggregator

        # 聚合后的维度是2*input_dim（拼接自身和邻居）
        self.weight = nn.Linear(2 * input_dim, output_dim)
        self.activation = nn.ReLU()
        self.dropout = nn.Dropout(0.5)

    def forward(self, self_feat: torch.Tensor,
                neighbor_feats: torch.Tensor) -> torch.Tensor:
        """
        Args:
            self_feat: [input_dim]
            neighbor_feats: [num_neighbors, input_dim]
        """
        # 聚合邻居特征
        if self.aggregator == 'mean':
            agg_feat = neighbor_feats.mean(dim=0)
        elif self.aggregator == 'max':
            agg_feat = neighbor_feats.max(dim=0)[0]
        elif self.aggregator == 'lstm':
            # 使用LSTM聚合邻居（需要排序）
            agg_feat = self._lstm_agg(neighbor_feats)
        else:
            agg_feat = neighbor_feats.mean(dim=0)

        # 拼接自身和邻居
        combined = torch.cat([self_feat, agg_feat], dim=0)

        # 线性变换
        output = self.weight(combined)
        output = self.activation(output)
        output = self.dropout(output)

        # L2归一化
        output = F.normalize(output, p=2, dim=0)

        return output


class FriendRecommendationModel(nn.Module):
    """
    端到端好友推荐模型
    结合图神经网络和链接预测
    """

    def __init__(self, num_nodes: int, embed_dim: int = 128):
        super().__init__()

        # 节点嵌入层（可学习）
        self.node_embeddings = nn.Embedding(num_nodes, embed_dim)

        # 图神经网络
        self.gnn = GraphSAGE(embed_dim, 256, embed_dim)

        # 链接预测MLP
        self.predictor = nn.Sequential(
            nn.Linear(2 * embed_dim, 256),
            nn.ReLU(),
            nn.Dropout(0.3),
            nn.Linear(256, 64),
            nn.ReLU(),
            nn.Linear(64, 1),
            nn.Sigmoid()
        )

    def forward(self, node_pairs: torch.Tensor) -> torch.Tensor:
        """
        预测节点对之间的连接概率

        Args:
            node_pairs: [batch_size, 2] 节点对索引
        """
        # 获取节点嵌入
        u_embed = self.node_embeddings(node_pairs[:, 0])
        v_embed = self.node_embeddings(node_pairs[:, 1])

        # 可以在这里添加GNN传播
        # node_feats = self.gnn(...)

        # 拼接并预测
        combined = torch.cat([u_embed, v_embed], dim=1)
        score = self.predictor(combined)

        return score.squeeze()

    def bpr_loss(self, pos_pairs: torch.Tensor,
                 neg_pairs: torch.Tensor) -> torch.Tensor:
        """
        贝叶斯个性化排序损失
        """
        pos_scores = self.forward(pos_pairs)
        neg_scores = self.forward(neg_pairs)

        # BPR损失：-log(σ(pos_score - neg_score))
        loss = -torch.log(torch.sigmoid(pos_scores - neg_scores) + 1e-10)
        return loss.mean()
```

```python
class ScalableRecommender:
    """
    工业级推荐系统优化策略
    """

    def __init__(self):
        # 多级缓存
        self.hot_cache = {}      # 热点用户推荐结果
        self.embed_cache = {}    # 节点嵌入缓存

    def recommend(self, user_id: int) -> List[int]:
        """
        分层推荐策略：
        1. L1缓存：热点用户直接返回预计算结果
        2. L2缓存：嵌入向量缓存加速相似度计算
        3. 实时计算：冷用户或缓存未命中
        """
        # L1: 热点缓存
        if user_id in self.hot_cache:
            return self.hot_cache[user_id]

        # 候选生成（多路召回）
        candidates = []

        # 路1：图算法召回
        graph_candidates = self.graph_recall(user_id, top_k=500)
        candidates.extend(graph_candidates)

        # 路2：嵌入相似度召回
        embed_candidates = self.embedding_recall(user_id, top_k=500)
        candidates.extend(embed_candidates)

        # 路3：基于内容的召回
        content_candidates = self.content_recall(user_id, top_k=200)
        candidates.extend(content_candidates)

        # 去重
        candidates = list(dict.fromkeys(candidates))

        # 精排（轻量级模型）
        ranked = self.light_ranker(user_id, candidates, top_k=100)

        # 多样性重排
        final = self.diversify(ranked, top_k=20)

        return final

    def graph_recall(self, user_id: int, top_k: int) -> List[int]:
        """
        基于图算法的快速召回
        使用预计算的相似度矩阵
        """
        # 从预计算的Top-K相似度表查询
        return self.precomputed_similarity.get(user_id, [])[:top_k]
```

#### 2.3.1 Louvain算法实现

```python
import networkx as nx
from collections import defaultdict
import numpy as np

class LouvainCommunityDetection:
    """
    Louvain算法：模块化度优化的贪心算法

    模块化度（Modularity）：
    Q = 1/(2m) * Σ_{ij} [A_{ij} - (k_i k_j)/(2m)] δ(c_i, c_j)

    其中：
    - m: 边的总数
    - A: 邻接矩阵
    - k_i: 节点i的度数
    - c_i: 节点i的社区
    - δ: 指示函数
    """

    def __init__(self, resolution: float = 1.0):
        self.resolution = resolution  # 分辨率参数，控制社区粒度

    def fit(self, G: nx.Graph) -> Dict[int, int]:
        """
        主算法流程
        两阶段迭代：
        1. 局部移动：将节点移动到模块化度增益最大的社区
        2. 社区聚合：将每个社区收缩为超节点，构建新图
        """
        # 初始化：每个节点一个社区
        n = G.number_of_nodes()
        node_to_community = {node: i for i, node in enumerate(G.nodes())}

        # 预计算
        self.m = G.number_of_edges()
        self.k = dict(G.degree())
        self.total_degree = 2 * self.m

        # 社区内部数据结构
        self.community_degrees = defaultdict(float)  # Σ_{i∈C} k_i
        self.community_internal = defaultdict(float)  # 社区内部边权重

        for node in G.nodes():
            comm = node_to_community[node]
            self.community_degrees[comm] += self.k[node]

        for u, v, w in G.edges(data='weight', default=1):
            if node_to_community[u] == node_to_community[v]:
                self.community_internal[node_to_community[u]] += 2 * w

        improved = True
        iteration = 0

        while improved:
            improved = False
            iteration += 1
            print(f"Iteration {iteration}")

            # Phase 1: 局部移动
            nodes = list(G.nodes())
            np.random.shuffle(nodes)  # 随机顺序

            for node in nodes:
                best_community = node_to_community[node]
                best_gain = 0.0

                # 计算移动到各邻居社区的增益
                neighbor_communities = defaultdict(float)
                for neighbor in G.neighbors(node):
                    comm = node_to_community[neighbor]
                    weight = G[node][neighbor].get('weight', 1)
                    neighbor_communities[comm] += weight

                # 当前社区的度数贡献
                node_comm = node_to_community[node]
                k_i_in = neighbor_communities.get(node_comm, 0)

                for comm, k_i_to_comm in neighbor_communities.items():
                    if comm == node_comm:
                        continue

                    # 模块化度增益公式
                    # ΔQ = k_{i,in} / m - resolution * k_i * Σ_{tot} / (2m²)
                    gain = (k_i_to_comm - k_i_in) / self.m - \
                           self.resolution * self.k[node] * \
                           (self.community_degrees[comm] -
                            self.community_degrees[node_comm]) / \
                           (self.total_degree ** 2)

                    if gain > best_gain:
                        best_gain = gain
                        best_community = comm

                # 移动节点
                if best_community != node_comm:
                    self._move_node(node, node_comm, best_community,
                                   neighbor_communities, G)
                    node_to_community[node] = best_community
                    improved = True

            # Phase 2: 社区聚合
            if improved:
                G, node_to_community = self._aggregate_communities(
                    G, node_to_community
                )

        return node_to_community

    def _move_node(self, node: int, old_comm: int, new_comm: int,
                   neighbor_communities: Dict[int, float], G: nx.Graph):
        """更新社区内部数据结构"""
        # 更新度数
        self.community_degrees[old_comm] -= self.k[node]
        self.community_degrees[new_comm] += self.k[node]

        # 更新内部边
        for neighbor in G.neighbors(node):
            weight = G[node][neighbor].get('weight', 1)
            if node_to_community[neighbor] == old_comm:
                self.community_internal[old_comm] -= 2 * weight
            if node_to_community[neighbor] == new_comm:
                self.community_internal[new_comm] += 2 * weight

    def _aggregate_communities(self, G: nx.Graph,
                               node_to_comm: Dict[int, int]) -> Tuple[nx.Graph, Dict]:
        """
        将社区聚合为超节点
        """
        # 构建社区到节点的映射
        comm_to_nodes = defaultdict(list)
        for node, comm in node_to_comm.items():
            comm_to_nodes[comm].append(node)

        # 新图
        new_G = nx.Graph()
        new_node_map = {}

        for new_id, (old_comm, nodes) in enumerate(comm_to_nodes.items()):
            new_node_map[old_comm] = new_id
            new_G.add_node(new_id, nodes=nodes)

        # 构建超边
        for u, v, w in G.edges(data='weight', default=1):
            comm_u = node_to_comm[u]
            comm_v = node_to_comm[v]
            if comm_u != comm_v:
                new_u = new_node_map[comm_u]
                new_v = new_node_map[comm_v]
                if new_G.has_edge(new_u, new_v):
                    new_G[new_u][new_v]['weight'] += w
                else:
                    new_G.add_edge(new_u, new_v, weight=w)

        # 新的节点映射
        new_mapping = {}
        for node, old_comm in node_to_comm.items():
            new_mapping[node] = new_node_map[old_comm]

        return new_G, new_mapping
```

```python
class LabelPropagation:
    """
    标签传播算法：近线性时间的社区发现

    核心思想：
    - 每个节点选择邻居中最多的标签
    - 异步更新打破对称性
    - 适用于大规模图
    """

    def fit(self, G: nx.Graph, max_iter: int = 100) -> Dict[int, int]:
        """
        标签传播算法实现
        """
        # 初始化：每个节点唯一标签
        labels = {node: i for i, node in enumerate(G.nodes())}

        for iteration in range(max_iter):
            changed = 0
            nodes = list(G.nodes())
            np.random.shuffle(nodes)  # 异步更新

            for node in nodes:
                if G.degree(node) == 0:
                    continue

                # 统计邻居标签
                label_counts = defaultdict(int)
                for neighbor in G.neighbors(node):
                    label_counts[labels[neighbor]] += \
                        G[node][neighbor].get('weight', 1)

                # 选择权重最大的标签（随机打破平局）
                max_count = max(label_counts.values())
                best_labels = [l for l, c in label_counts.items()
                              if c == max_count]
                new_label = np.random.choice(best_labels)

                if labels[node] != new_label:
                    labels[node] = new_label
                    changed += 1

            print(f"Iteration {iteration}: {changed} nodes changed")

            if changed == 0:
                break

        return labels


class CPMCommunityDetection:
    """
    Clique Percolation Method（CPM）：
    基于k-团的社区发现，支持重叠社区
    """

    def __init__(self, k: int = 3):
        self.k = k  # 团的大小

    def find_k_cliques(self, G: nx.Graph) -> List[Set[int]]:
        """找出图中所有k-团"""
        from itertools import combinations

        cliques = []
        for clique in nx.find_cliques(G):
            if len(clique) >= self.k:
                # 生成所有k大小的子团
                for k_subset in combinations(clique, self.k):
                    cliques.append(set(k_subset))

        return cliques

    def fit(self, G: nx.Graph) -> List[Set[int]]:
        """
        CPM算法：
        1. 找出所有k-团
        2. 构建团图（共享k-1个节点的团相连）
        3. 团图的连通分量即为社区
        """
        k_cliques = self.find_k_cliques(G)

        if not k_cliques:
            return []

        # 构建团图
        n = len(k_cliques)
        clique_graph = nx.Graph()
        clique_graph.add_nodes_from(range(n))

        for i in range(n):
            for j in range(i + 1, n):
                # 如果两个k-团共享k-1个节点
                shared = len(k_cliques[i] & k_cliques[j])
                if shared >= self.k - 1:
                    clique_graph.add_edge(i, j)

        # 团图的连通分量对应社区
        communities = []
        for component in nx.connected_components(clique_graph):
            community = set()
            for clique_idx in component:
                community.update(k_cliques[clique_idx])
            communities.append(community)

        return communities
```

```python
class METISPartition:
    """
    METIS：多级图划分算法
    适用于将图划分为平衡的部分
    """

    def __init__(self, num_partitions: int):
        self.k = num_partitions

    def partition(self, G: nx.Graph) -> Dict[int, int]:
        """
        METIS三阶段算法：
        1. 粗化：逐步合并节点，减少图规模
        2. 初始划分：在粗图上进行初始划分
        3. 细化：逐步投影回原始图，优化划分
        """
        # 使用networkx-metis或自定义实现
        try:
            import metis
            # 转换为METIS格式
            adjacency = {}
            for node in G.nodes():
                adjacency[node] = list(G.neighbors(node))

            cuts, labels = metis.part_graph(adjacency, self.k)
            return {node: label for node, label in zip(G.nodes(), labels)}

        except ImportError:
            # 简化实现：谱聚类
            return self._spectral_partition(G)

    def _spectral_partition(self, G: nx.Graph) -> Dict[int, int]:
        """
        谱聚类作为METIS的替代
        """
        import numpy as np
        from sklearn.cluster import KMeans

        # 计算归一化拉普拉斯矩阵
        L = nx.normalized_laplacian_matrix(G)

        # 计算前k个特征向量
        eigenvalues, eigenvectors = np.linalg.eigh(L.toarray())

        # 使用前k个特征向量进行k-means聚类
        X = eigenvectors[:, 1:self.k+1]  # 跳过第一个特征向量（常数）
        kmeans = KMeans(n_clusters=self.k, random_state=42)
        labels = kmeans.fit_predict(X)

        return {node: label for node, label in zip(G.nodes(), labels)}
```

## 3. 影响力最大化

### 3.1 问题背景与动机

**场景**：

- 病毒营销：选择K个种子用户最大化广告传播
- 舆情监控：识别关键意见领袖
- 谣言控制：阻断信息传播的关键节点
- 产品推广：最优的种子用户选择

**问题定义（Kempe et al., 2003）**：
给定图G=(V,E)和传播模型，找到大小为k的节点集S，使得影响力传播σ(S)最大。

**挑战**：

- NP-hard问题
- 影响力传播具有随机性
- 图规模巨大
- 需要实时响应

### 3.2 算法选择原理

#### 传播模型

**1. 独立级联模型（IC）**：

- 节点u激活后，有概率p(u,v)激活邻居v
- 每个节点只有一次激活机会
- 适用于一次性决策场景

**2. 线性阈值模型（LT）**：

- 节点v有阈值θ_v，当邻居影响力之和≥θ_v时被激活
- 适用于累积影响场景

#### 算法对比

| 算法 | 近似比 | 时间复杂度 | 适用场景 |
|-----|--------|-----------|---------|
| 贪心算法 | 1-1/e | O(knRm) | 小规模，高精度 |
| CELF | 1-1/e | O(knR) 均摊 | 中等规模 |
| CELF++ | 1-1/e | 更快 | 中等规模 |
| TIM/TIM+ | 1-1/e-ε | O((k+l)(n+m)log n/ε²) | 大规模 |
| IMM | 1-1/e-ε | O((k+l)(n+m)log n/ε²) | 超大规模 |
| 度中心性 | 无保证 | O(m) | 超快速baseline |
| PageRank | 无保证 | O(m) | 综合影响力 |

```python
import numpy as np
from collections import deque
import random
from typing import Set, List, Dict

class InfluenceMaximization:
    """
    影响力最大化算法实现
    """

    def __init__(self, G, propagation_model='IC', p=0.1):
        """
        Args:
            G: NetworkX图
            propagation_model: 'IC' 或 'LT'
            p: 传播概率（IC模型）
        """
        self.G = G
        self.model = propagation_model
        self.p = p

    def independent_cascade(self, seeds: Set[int],
                           mc_simulations: int = 1000) -> float:
        """
        独立级联模型的蒙特卡洛模拟

        Returns:
            期望影响力传播范围
        """
        total_spread = 0

        for _ in range(mc_simulations):
            activated = set(seeds)
            newly_activated = set(seeds)

            while newly_activated:
                next_activated = set()
                for node in newly_activated:
                    for neighbor in self.G.neighbors(node):
                        if neighbor not in activated:
                            # 以概率p激活
                            if random.random() < self.p:
                                next_activated.add(neighbor)

                activated.update(next_activated)
                newly_activated = next_activated

            total_spread += len(activated)

        return total_spread / mc_simulations

    def greedy_selection(self, k: int, mc_simulations: int = 1000) -> List[int]:
        """
        贪心算法：
        每次选择边际增益最大的节点

        性质：单调性 + 子模性 → 1-1/e近似保证
        """
        seeds = []

        for i in range(k):
            best_node = None
            best_marginal_gain = 0

            for node in self.G.nodes():
                if node in seeds:
                    continue

                # 计算边际增益
                marginal_gain = (
                    self.independent_cascade(set(seeds + [node]), mc_simulations) -
                    self.independent_cascade(set(seeds), mc_simulations)
                )

                if marginal_gain > best_marginal_gain:
                    best_marginal_gain = marginal_gain
                    best_node = node

            if best_node is not None:
                seeds.append(best_node)
                print(f"Selected {i+1}/{k}: node {best_node}, "
                      f"marginal gain: {best_marginal_gain:.2f}")

        return seeds


class CELF:
    """
    Cost-Effective Lazy Forward（CELF）优化

    利用子模性：
    S ⊆ T 时，f(u|S) ≥ f(u|T)
    边际增益递减，可用堆优化避免重复计算
    """

    def __init__(self, G, p=0.1):
        self.G = G
        self.p = p
        self.mc = 1000

    def select(self, k: int) -> List[int]:
        """
        CELF算法实现
        """
        import heapq

        # 初始化：计算所有节点的边际增益（相对于空集）
        heap = []
        for node in self.G.nodes():
            gain = self._marginal_gain(node, set())
            # 最大堆使用负数
            heapq.heappush(heap, (-gain, node, 0))  # (负增益, 节点, 上次更新轮次)

        seeds = []
        round_num = 0

        while len(seeds) < k and heap:
            round_num += 1
            neg_gain, node, last_updated = heapq.heappop(heap)

            if last_updated == len(seeds):
                # 此轮已更新过，直接选中
                seeds.append(node)
                print(f"CELF selected {len(seeds)}/{k}: node {node}")
            else:
                # 需要重新计算边际增益
                marginal_gain = self._marginal_gain(node, set(seeds))
                heapq.heappush(heap, (-marginal_gain, node, len(seeds)))

        return seeds

    def _marginal_gain(self, node: int, seeds: Set[int]) -> float:
        """计算边际增益"""
        if not seeds:
            return self._influence_spread({node})

        spread_with = self._influence_spread(seeds | {node})
        spread_without = self._influence_spread(seeds)
        return spread_with - spread_without

    def _influence_spread(self, seeds: Set[int]) -> float:
        """蒙特卡洛估计影响力"""
        total = 0
        for _ in range(self.mc):
            activated = self._simulate_ic(seeds)
            total += len(activated)
        return total / self.mc

    def _simulate_ic(self, seeds: Set[int]) -> Set[int]:
        """单次IC模拟"""
        activated = set(seeds)
        newly_activated = list(seeds)

        while newly_activated:
            node = newly_activated.pop()
            for neighbor in self.G.neighbors(node):
                if neighbor not in activated and random.random() < self.p:
                    activated.add(neighbor)
                    newly_activated.append(neighbor)

        return activated
```

```python
class RISInfluenceMaximization:
    """
    Reverse Influence Sampling (RIS)

    TIM/TIM+ 和 IMM 算法的核心思想：
    - 生成随机反向可达集（RR set）
    - 影响力最大化 ≈ 最大覆盖问题
    - 提供 (1-1/e-ε) 近似保证

    RR Set生成：从随机节点v出发，沿反向边按概率采样
    """

    def __init__(self, G, p=0.1):
        self.G = G
        self.p = p

    def generate_rr_set(self) -> Set[int]:
        """
        生成随机反向可达集

        从随机节点v开始，沿入边反向传播
        一个节点u在RR集中，当且仅当u可到达v
        """
        # 随机选择目标节点
        v = random.choice(list(self.G.nodes()))

        rr_set = {v}
        queue = deque([v])

        while queue:
            node = queue.popleft()

            # 遍历入边邻居（反向图）
            for neighbor in self.G.predecessors(node) if self.G.is_directed() \
                          else self.G.neighbors(node):
                if neighbor not in rr_set:
                    # 以概率p加入（反向传播概率）
                    if random.random() < self.p:
                        rr_set.add(neighbor)
                        queue.append(neighbor)

        return rr_set

    def tim_plus(self, k: int, epsilon: float = 0.1) -> List[int]:
        """
        TIM+算法实现

        两阶段：
        1. 估计所需RR集数量
        2. 最大覆盖贪心选择
        """
        n = self.G.number_of_nodes()

        # Phase 1: 估计最优影响力
        lambda_ = (8 + 2 * epsilon) * n * \
                  (np.log(n) + np.log(2) + np.log(n)) / (epsilon ** 2)

        # 生成RR集
        print("Generating RR sets...")
        rr_sets = []
        for _ in range(int(lambda_)):
            rr_sets.append(self.generate_rr_set())

        # Phase 2: 最大覆盖
        print("Running maximum coverage...")
        seeds = self._max_coverage(rr_sets, k)

        return seeds

    def _max_coverage(self, rr_sets: List[Set[int]], k: int) -> List[int]:
        """
        贪心最大覆盖
        选择能覆盖最多RR集的k个节点
        """
        # 建立节点到RR集的映射
        node_to_rrs = defaultdict(set)
        for i, rr_set in enumerate(rr_sets):
            for node in rr_set:
                node_to_rrs[node].add(i)

        seeds = []
        covered = set()

        for _ in range(k):
            best_node = None
            best_coverage = 0

            for node, rr_indices in node_to_rrs.items():
                if node in seeds:
                    continue

                # 计算新覆盖的RR集
                new_coverage = len(rr_indices - covered)
                if new_coverage > best_coverage:
                    best_coverage = new_coverage
                    best_node = node

            if best_node is not None:
                seeds.append(best_node)
                covered.update(node_to_rrs[best_node])

        return seeds

    def imm(self, k: int, epsilon: float = 0.1) -> List[int]:

        IMM（Influence Maximization via Martingales）

        相比TIM+进一步优化了RR集数量的估计

        n = self.G.number_of_nodes()

            for _ in range(int(theta) - len(rr_sets)):
                rr_sets.append(self.generate_rr_set())
```

```python
class EmbeddingBasedIM:
    """
    基于图嵌入的影响力最大化
    牺牲一定精度换取速度
    """

    def __init__(self, G):
        self.G = G
        self.embeddings = None

    def train_embeddings(self, dimensions=128):
        """使用Node2Vec训练嵌入"""
        from node2vec import Node2Vec

        node2vec = Node2Vec(self.G, dimensions=dimensions,
                           walk_length=30, num_walks=200)
        model = node2vec.fit(window=10, min_count=1)

        self.embeddings = {
            node: model.wv[str(node)]
            for node in self.G.nodes()
        }

    def select_by_embedding(self, k: int) -> List[int]:
        """
        基于嵌入的选择策略：
        1. 选择嵌入空间中分散的节点（最大边际相关性）
        2. 同时考虑节点度中心性
        """
        if self.embeddings is None:
            self.train_embeddings()

        seeds = []
        candidates = set(self.G.nodes())

        for _ in range(k):
            best_score = -float('inf')
            best_node = None

            for node in candidates:
                if node in seeds:
                    continue

                # 节点影响力分数
                influence_score = self.G.degree(node)

                # 多样性分数（与已选节点的嵌入距离）
                diversity_score = 0
                if seeds:
                    min_dist = min(
                        np.linalg.norm(self.embeddings[node] - self.embeddings[s])
                        for s in seeds
                    )
                    diversity_score = min_dist
                else:
                    diversity_score = 1.0

                # 综合分数
                score = 0.7 * influence_score + 0.3 * diversity_score

                if score > best_score:
                    best_score = score
                    best_node = node

            seeds.append(best_node)

        return seeds
```
