## 概述

零知识证明（Zero-Knowledge Proof, ZKP）允许证明者向验证者证明某个陈述为真，而无需透露任何额外信息。
这一密码学原语正在革命性地改变区块链、身份认证和云计算领域。本案例研究深入分析三个核心应用场景。

---

## 1. zk-SNARKs在区块链中的应用

### 1.1 问题背景与动机

**场景**：

- Zcash：保护交易隐私的加密货币
- zk-Rollups：以太坊Layer 2扩容方案
- 匿名投票和治理系统

**业务价值**：

- 交易隐私：隐藏发送方、接收方、金额
- 链上压缩：将数千笔交易压缩为单个证明
- 合规性：在保护隐私的同时满足监管要求

**挑战**：

- 证明生成计算成本高
- 可信设置的安全性
- 量子计算威胁
- 用户体验（证明生成时间）

#### 1.3.1 Groth16实现

```python
import numpy as np
from typing import Tuple, List
from dataclasses import dataclass
import hashlib

@dataclass
class G1Point:
    """椭圆曲线G1群上的点（简化表示）"""
    x: int
    y: int

    def add(self, other: 'G1Point') -> 'G1Point':
        # 椭圆曲线加法（简化）
        return G1Point(self.x + other.x, self.y + other.y)

    def scalar_mul(self, scalar: int) -> 'G1Point':
        # 标量乘法（简化）
        return G1Point(self.x * scalar, self.y * scalar)

@dataclass
class G2Point:
    """椭圆曲线G2群上的点"""
    x: Tuple[int, int]
    y: Tuple[int, int]

@dataclass
class Groth16Proof:
    """Groth16证明结构"""
    A: G1Point
    B: G2Point
    C: G1Point

class Groth16:
    """
    Groth16 zk-SNARK实现

    基于二次算术程序（QAP）：
    - 将计算转换为多项式等式
    - 使用配对友好椭圆曲线（BN254/BLS12-381）

    证明大小：3个椭圆曲线点（~200字节）
    验证：3次配对运算

    论文：On the Size of Pairing-based Non-interactive Arguments
    """

    def __init__(self, curve='BN254'):
        self.curve = curve
        # 在实际实现中，这里初始化椭圆曲线参数

    def compile_circuit(self, circuit_fn):
        """
        将计算电路编译为R1CS，再转换为QAP

        步骤：
        1. 电路 → R1CS（秩一约束系统）
        2. R1CS → QAP（二次算术程序）
        3. 可信设置生成proving key和verification key
        """
        # 1. 构建约束系统
        constraints = self._build_constraints(circuit_fn)

        # 2. 转换为R1CS
        r1cs = self._to_r1cs(constraints)

        # 3. R1CS到QAP（拉格朗日插值）
        qap = self._r1cs_to_qap(r1cs)

        return qap

    def _r1cs_to_qap(self, r1cs):
        """
        将R1CS转换为QAP

        R1CS: A·w ⊙ B·w = C·w
        QAP: A(x)·B(x) = C(x) 在特定点上成立
        """
        n = len(r1cs.constraints)

        # 选择roots of unity作为评估点
        roots = self._roots_of_unity(n)

        # 对每个列进行拉格朗日插值
        A_polys = self._interpolate_columns(r1cs.A, roots)
        B_polys = self._interpolate_columns(r1cs.B, roots)
        C_polys = self._interpolate_columns(r1cs.C, roots)

        return {
            'A': A_polys,
            'B': B_polys,
            'C': C_polys,
            'Z': self._vanishing_polynomial(roots)  # 零多项式
        }

    def trusted_setup(self, qap) -> Tuple[dict, dict]:
        """
        可信设置（Trusted Setup）

        生成：
        - proving key (pk)：用于生成证明
        - verification key (vk)：用于验证证明

        安全性依赖于"toxic waste"（随机数τ, α, β）的安全销毁
        """
        # 随机数（toxic waste）
        tau = self._random_scalar()    # 用于多项式求值
        alpha = self._random_scalar()  # 用于A的一致性
        beta = self._random_scalar()   # 用于B的一致性
        gamma = self._random_scalar()  # 用于公开输入
        delta = self._random_scalar()  # 用于私有输入

        m = len(qap['A'])  # 变量数
        n = len(qap['A'][0])  # 约束数

        # 生成proving key元素
        pk = {
            'alpha_g1': self.g1.scalar_mul(alpha),
            'beta_g1': self.g1.scalar_mul(beta),
            'beta_g2': self.g2.scalar_mul(beta),
            'delta_g1': self.g1.scalar_mul(delta),
            'delta_g2': self.g2.scalar_mul(delta),
            'A_query': [self.g1.scalar_mul(qap['A'][i].eval(tau))
                       for i in range(m)],
            'B_query': [self.g2.scalar_mul(qap['B'][i].eval(tau))
                       for i in range(m)],
            'C_query': [self.g1.scalar_mul(qap['C'][i].eval(tau))
                       for i in range(m)],
            'H_query': [self.g1.scalar_mul(tau ** i)
                       for i in range(n)],
        }

        # 生成verification key
        vk = {
            'alpha_g1': pk['alpha_g1'],
            'beta_g2': pk['beta_g2'],
            'gamma_g2': self.g2.scalar_mul(gamma),
            'delta_g2': pk['delta_g2'],
            'IC': [self.g1.scalar_mul((beta * qap['A'][i].eval(tau) +
                                      alpha * qap['B'][i].eval(tau) +
                                      qap['C'][i].eval(tau)) / gamma)
                   for i in range(self.num_public_inputs)]
        }

        # 安全删除toxic waste!
        del tau, alpha, beta, gamma, delta

        return pk, vk

    def prove(self, pk: dict, witness: List[int]) -> Groth16Proof:
        """
        生成证明

        证明者知道witness使得电路约束满足
        """
        # 公开输入和私有输入
        public_inputs = witness[:self.num_public_inputs]
        private_inputs = witness[self.num_public_inputs:]

        # 计算多项式A(x), B(x), C(x)在τ处的值
        A_eval = sum(pk['A_query'][i].scalar_mul(witness[i])
                    for i in range(len(witness)))
        B_eval = sum(pk['B_query'][i].scalar_mul(witness[i])
                    for i in range(len(witness)))
        C_eval = sum(pk['C_query'][i].scalar_mul(witness[i])
                    for i in range(len(witness)))

        # 计算H(x) = (A(x)B(x) - C(x)) / Z(x)
        # 在QAP中，A·B - C = H·Z
        H = self._compute_h_polynomial(witness)

        # 随机化证明（零知识）
        r = self._random_scalar()
        s = self._random_scalar()

        # 构造证明
        A = A_eval.add(pk['alpha_g1']).add(pk['delta_g1'].scalar_mul(r))
        B = B_eval.add(pk['beta_g2']).add(pk['delta_g2'].scalar_mul(s))
        C = C_eval.add(pk['delta_g1'].scalar_mul(r * s)).add(
            A_eval.scalar_mul(s)).add(B_eval.scalar_mul(r))

        return Groth16Proof(A, B, C)

    def verify(self, vk: dict, public_inputs: List[int],
               proof: Groth16Proof) -> bool:
        """
        验证证明

        验证方程：
        e(A, B) = e(alpha_g1, beta_g2) · e(IC · public_inputs, gamma_g2) · e(C, delta_g2)

        其中e是配对运算
        """
        # 计算输入承诺
        acc = vk['IC'][0]  # 常数项
        for i, inp in enumerate(public_inputs):
            acc = acc.add(vk['IC'][i + 1].scalar_mul(inp))

        # 配对检查
        # 左边：e(proof.A, proof.B)
        left = self.pairing(proof.A, proof.B)

        # 右边乘积
        right1 = self.pairing(vk['alpha_g1'], vk['beta_g2'])
        right2 = self.pairing(acc, vk['gamma_g2'])
        right3 = self.pairing(proof.C, vk['delta_g2'])

        right = right1.multiply(right2).multiply(right3)

        return left == right


class ZcashTransaction:
    """
    Zcash风格的隐私交易

    使用zk-SNARK证明：
    1. 输入utxo存在且未花费
    2. 发送者有花费权限（知道私钥）
    3. 输入输出金额平衡
    4. 输出承诺正确
    """

    def __init__(self, groth16: Groth16):
        self.groth16 = groth16
        # 加载Zcash电路的proving key和verification key

    def create_shielded_transaction(self, inputs, outputs):
        """
        创建隐私交易

        公开信息：nullifiers, commitments, 交易签名
        私有信息：金额, 地址, 随机数
        """
        # 构建witness
        witness = self._build_witness(inputs, outputs)

        # 生成zk-SNARK证明
        proof = self.groth16.prove(self.proving_key, witness)

        # 构造交易
        tx = {
            'nullifiers': [inp['nullifier'] for inp in inputs],
            'commitments': [out['commitment'] for out in outputs],
            'proof': proof,
            'public_value': sum(inp['value'] for inp in inputs) - \
                          sum(out['value'] for out in outputs)
        }

        return tx
```

#### 1.3.2 PLONK实现

```python
class PLONK:
    """
    PLONK (Permutations over Lagrange-bases for Oecumenical Noninteractive
           arguments of Knowledge)

    优势：
    1. 通用可信设置（SRS与电路无关）
    2. 更灵活的约束（自定义门）
    3. 更好的递归证明支持

    论文：PLONK: Permutations over Lagrange-bases for Oecumenical
          Noninteractive arguments of Knowledge
    """

    def __init__(self, max_constraints=2**20):
        self.max_constraints = max_constraints
        # 初始化通用参考字符串
        self.srs = self._universal_setup(max_constraints)

    def _universal_setup(self, max_degree):
        """
        通用可信设置

        生成结构化参考字符串（SRS）：
        {[g^τ^i]_1, [g^τ^i]_2} for i=0..max_degree

        这个SRS可用于任何不超过max_degree个约束的电路
        """
        tau = self._random_scalar()  # toxic waste

        srs_g1 = [self.g1.scalar_mul(tau ** i) for i in range(max_degree)]
        srs_g2 = [self.g2.scalar_mul(tau ** i) for i in range(max_degree)]

        del tau  # 安全删除

        return {'g1': srs_g1, 'g2': srs_g2}

    def compile_circuit(self, circuit):
        """
        编译电路为PLONK格式

        PLONK使用自定义门和置换论证
        """
        # 提取门约束
        gates = circuit.get_gates()

        # 构建置换（wire mapping）
        permutations = self._build_permutations(gates)

        # 选择多项式
        q_L, q_R, q_O, q_M, q_C = self._selector_polynomials(gates)

        return {
            'gates': gates,
            'permutations': permutations,
            'selectors': {
                'q_L': q_L, 'q_R': q_R, 'q_O': q_O,
                'q_M': q_M, 'q_C': q_C
            }
        }

    def prove(self, circuit, witness):
        """
        PLONK证明生成

        核心协议：
        1. 提交witness多项式（a, b, c）
        2. 置换论证（copy constraints）
        3. 商多项式（证明约束满足）
        """
        n = len(circuit['gates'])

        # 1. witness多项式
        a_poly = self._interpolate_witness(witness['a'])
        b_poly = self._interpolate_witness(witness['b'])
        c_poly = self._interpolate_witness(witness['c'])

        # 2. 承诺witness多项式
        a_comm = self.commit(a_poly)
        b_comm = self.commit(b_poly)
        c_comm = self.commit(c_poly)

        # 3. 置换论证（beta, gamma挑战）
        beta = self._challenge([a_comm, b_comm, c_comm])
        gamma = self._challenge([a_comm, b_comm, c_comm, beta])

        z_poly = self._build_permutation_poly(
            circuit['permutations'], witness, beta, gamma
        )
        z_comm = self.commit(z_poly)

        # 4. 商多项式（alpha挑战）
        alpha = self._challenge([z_comm])

        # 构建约束多项式
        t_poly = self._compute_quotient_poly(
            circuit, a_poly, b_poly, c_poly, z_poly, alpha, beta, gamma
        )

        # 5. 分裂和承诺商多项式
        t_comms = self._split_and_commit(t_poly)

        # 6. 评估论证（zeta挑战）
        zeta = self._challenge(t_comms)

        evaluations = {
            'a': a_poly.eval(zeta),
            'b': b_poly.eval(zeta),
            'c': c_poly.eval(zeta),
            'z': z_poly.eval(zeta),
            'z_omega': z_poly.eval(zeta * self.omega)
        }

        # 7. 线性化多项式
        r_poly = self._linearization_poly(
            circuit, evaluations, alpha, beta, gamma
        )

        # 8. 开启证明
        w_comm = self._compute_opening_proof(zeta)
        w_omega_comm = self._compute_opening_proof(zeta * self.omega)

        return {
            'commitments': [a_comm, b_comm, c_comm, z_comm] + t_comms,
            'evaluations': evaluations,
            'openings': [w_comm, w_omega_comm]
        }
```

### 1.5 经验教训

1. **证明生成是瓶颈**：需要GPU加速或证明外包
2. **可信设置是安全隐患**：优先选择通用或透明设置方案
3. **递归证明可实现压缩**：一个证明验证多个证明
4. **EVM兼容性有代价**：zkEVM效率低于专用电路

#### 2.3.1 BBS+签名方案

```python
class BBSPlusSignature:
    """
    BBS+签名方案：支持多消息的签名和零知识证明

    特性：
    1. 对多个消息签名
    2. 选择性披露（只揭示部分消息）
    3. 零知识证明拥有签名

    论文：Short Group Signatures
    """

    def __init__(self, num_messages):
        self.num_messages = num_messages
        # 生成参数
        self.g1 = self._generate_g1()
        self.g2 = self._generate_g2()
        self.h_points = [self._generate_g1() for _ in range(num_messages)]

    def keygen(self):
        """生成密钥对"""
        sk = self._random_scalar()  # 私钥
        pk = self.g2.scalar_mul(sk)  # 公钥
        return sk, pk

    def sign(self, sk, messages):
        """
        对多个消息签名

        签名: σ = (A, e, s)
        其中 A = (g1 · h0^s · ∏ hi^mi)^(1/(e+sk))
        """
        e = self._random_scalar()
        s = self._random_scalar()

        # 计算B = g1 · h0^s · ∏ hi^mi
        B = self.g1.copy()
        B = B.add(self.h_points[0].scalar_mul(s))

        for i, msg in enumerate(messages):
            B = B.add(self.h_points[i + 1].scalar_mul(msg))

        # A = B^(1/(e + sk))
        exp = pow(e + sk, -1, self.order)
        A = B.scalar_mul(exp)

        return (A, e, s)

    def prove_possession(self, signature, messages, disclosed_indices):
        """
        零知识证明拥有签名，并选择性披露部分消息

        披露：disclosed_indices指定的消息
        隐藏：其他消息和签名随机数
        """
        A, e, s = signature

        # 随机盲化
        r1 = self._random_scalar()
        r2 = self._random_scalar()

        # A' = A^r1
        A_prime = A.scalar_mul(r1)

        # A_bar = A'^(-e) · B^r1
        # (用于验证签名结构)

        # 计算承诺
        # 隐藏的消息乘以随机数
        hidden_commitment = self.g1.scalar_mul(r2)
        for i, msg in enumerate(messages):
            if i not in disclosed_indices:
                hidden_commitment = hidden_commitment.add(
                    self.h_points[i + 1].scalar_mul(msg * r1)
                )

        # Fiat-Shamir挑战
        challenge = self._hash_points([A_prime, hidden_commitment])

        # 生成响应（ Schnorr协议 ）
        responses = {}
        for i, msg in enumerate(messages):

                responses[i] = msg * r1 + challenge * msg  # 简化的响应

        return {
            'A_prime': A_prime,
            'hidden_commitment': hidden_commitment,
            'disclosed_messages': {
                i: messages[i] for i in disclosed_indices

            'challenge': challenge
```

```python
class VerifiableML:

        - 矩阵乘法：累加乘法门
        - ReLU：比较门（需分解）
        - 归一化：除法门

        compiler = CircuitCompiler()

        for layer in model.layers:
            if isinstance(layer, torch.nn.Linear):
                compiler.add_dense_layer(layer.weight, layer.bias)
            elif isinstance(layer, torch.nn.ReLU):
                compiler.add_relu_layer()
            # ...

        return compiler.compile()

    def prove_inference(self, input_data):
        """
        生成推理证明
        """
        # 运行推理获取输出
        output = self.model(input_data)

        # 构建witness
        witness = self._build_witness(input_data, output)

        # 生成zk-SNARK证明

class VerifiableDatabase:
    """
    可验证数据库查询

    证明：查询结果正确且完整
    """

    def __init__(self):
        self.commitment = None  # 数据库承诺

    def commit_database(self, data):
        """
        使用Merkle树承诺数据库
        """
        from merkletools import MerkleTools

        mt = MerkleTools()
        for record in data:
            mt.add_leaf(self._hash_record(record), True)

        mt.make_tree()
        self.commitment = mt.get_merkle_root()
        self.merkle_tree = mt

        return self.commitment

    def prove_query(self, query, result):
        """
        生成查询结果证明

        证明：
        1. 结果中的每条记录都在数据库中（Merkle proof）
        2. 没有遗漏符合条件的记录（范围证明）
        """
        proofs = []

        for record in result:
            # 获取Merkle路径
            index = self._find_record_index(record)
            proof = self.merkle_tree.get_proof(index)
            proofs.append({
                'record': record,
                'merkle_proof': proof
            })

        # 完整性证明（范围证明）
        # 证明数据库中没有其他记录满足查询条件
        completeness_proof = self._prove_completeness(query, result)

        return {
            'result': result,
            'membership_proofs': proofs,
            'completeness_proof': completeness_proof,
            'database_commitment': self.commitment
        }

    def verify_query(self, query, proof_response):
        """
        验证查询结果
        """
        # 验证每条记录的成员资格
        for proof in proof_response['membership_proofs']:
            if not self._verify_merkle_proof(
                proof['record'],
                proof['merkle_proof'],
                proof_response['database_commitment']
            ):
                return False

        # 验证完整性
        if not self._verify_completeness(
            query,
            proof_response['result'],
            proof_response['completeness_proof']
        ):
            return False

        return True
```
