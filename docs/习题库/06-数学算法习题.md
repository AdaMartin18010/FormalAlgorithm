#### 题目 6.1.1：欧几里得算法（GCD/LCM）
**难度**：⭐  
**标签**：GCD、数论基础

**题目描述**：  
求两个数的最大公约数和最小公倍数。

**参考答案概要**：  
```cpp
// 辗转相除法
int gcd(int a, int b) { return b ? gcd(b, a % b) : a; }
int lcm(int a, int b) { return a / gcd(a, b) * b; }

// 扩展欧几里得：求 ax + by = gcd(a,b) 的解
int exgcd(int a, int b, int &x, int &y) {
    if(!b) { x = 1; y = 0; return a; }
    int d = exgcd(b, a % b, y, x);
    y -= a / b * x;
    return d;
}
```

#### 题目 6.1.3：埃氏筛/欧拉筛
**难度**：⭐  
**标签**：素数筛

**题目描述**：  
求 n 以内的所有素数。

**参考答案概要**：  
```cpp
// 欧拉筛 O(n)
vector<int> sieve(int n) {
    vector<bool> isPrime(n+1, true);
    vector<int> primes;
    isPrime[0] = isPrime[1] = false;
    for(int i = 2; i <= n; i++) {
        if(isPrime[i]) primes.push_back(i);
        for(int p : primes) {
            if(1LL * i * p > n) break;
            isPrime[i * p] = false;
            if(i % p == 0) break;
        }
    }
    return primes;
}
```

#### 题目 6.1.4：快速幂
**难度**：⭐  
**标签**：快速幂、模运算

**题目描述**：  
求 a^b mod p。

**参考答案概要**：  
```cpp
long long qpow(long long a, long long b, long long p) {
    long long res = 1;
    a %= p;
    while(b) {
        if(b & 1) res = res * a % p;
        a = a * a % p;
        b >>= 1;
    }
    return res;
}
```

```cpp
vector<pair<int, int>> factorize(int n) {
    vector<pair<int, int>> factors;
    for(int i = 2; i * i <= n; i++) {
        if(n % i == 0) {
            int cnt = 0;
            while(n % i == 0) {
                n /= i;
                cnt++;
            }
            factors.push_back({i, cnt});
        }
    }
    if(n > 1) factors.push_back({n, 1});
    return factors;
}
```

#### 题目 6.1.9：中国剩余定理（CRT）
**难度**：⭐⭐⭐  
**标签**：CRT、同余方程组

**题目描述**：  
求解同余方程组 x ≡ a_i (mod m_i)，其中 m_i 两两互质。

**参考答案概要**：  
```cpp
// 两两合并
typedef long long ll;
ll crt(vector<ll> &a, vector<ll> &m) {
    ll M = 1, x = 0;
    for(ll mi : m) M *= mi;
    for(int i = 0; i < m.size(); i++) {
        ll Mi = M / m[i];
        ll ti = inv(Mi % m[i], m[i]);  // 逆元
        x = (x + a[i] * Mi % M * ti % M) % M;
    }
    return (x % M + M) % M;
}
```

**标签**：素性测试、随机算法

**题目描述**：

**参考答案概要**：  
```cpp
bool millerRabin(long long n) {
    if(n < 3 || n % 2 == 0) return n == 2;
    long long d = n - 1, s = 0;
    while(d % 2 == 0) { d /= 2; s++; }
    // 测试基数
    for(long long a : {2, 325, 9375, 28178, 450775, 9780504, 1795265022}) {
        if(a % n == 0) continue;
        long long x = qpow(a, d, n);
        if(x == 1 || x == n-1) continue;
        bool composite = true;
        for(int r = 1; r < s; r++) {
            x = (__int128)x * x % n;
            if(x == n-1) { composite = false; break; }
        }
        if(composite) return false;
    }
    return true;
}
```

**参考答案概要**：  
```cpp
struct LinearBasis {
    long long p[64];  // p[i] 表示最高位为 i 的基
    LinearBasis() { memset(p, 0, sizeof(p)); }
    
    void insert(long long x) {
        for(int i = 63; i >= 0; i--) {
            if(!(x >> i & 1)) continue;
            if(!p[i]) { p[i] = x; return; }
            x ^= p[i];
        }
    }
    
    long long queryMax() {
        long long res = 0;
        for(int i = 63; i >= 0; i--)
            if((res ^ p[i]) > res) res ^= p[i];
        return res;
    }
};
```

#### 题目 6.2.1：排列组合基础
**难度**：⭐  
**标签**：排列、组合

**题目描述**：  
计算排列数 P(n,m) 和组合数 C(n,m)。

**参考答案概要**：  
```cpp
// 预处理阶乘和逆阶乘
const int MOD = 1e9 + 7;
const int MAXN = 1e6 + 5;
long long fac[MAXN], invfac[MAXN];

long long qpow(long long a, long long b) {
    long long res = 1;
    while(b) {
        if(b & 1) res = res * a % MOD;
        a = a * a % MOD;
        b >>= 1;
    }
    return res;
}

void init() {
    fac[0] = 1;
    for(int i = 1; i < MAXN; i++) fac[i] = fac[i-1] * i % MOD;
    invfac[MAXN-1] = qpow(fac[MAXN-1], MOD-2);
    for(int i = MAXN-2; i >= 0; i--) 
        invfac[i] = invfac[i+1] * (i+1) % MOD;
}

long long C(int n, int m) {
    if(m < 0 || m > n) return 0;
    return fac[n] * invfac[m] % MOD * invfac[n-m] % MOD;
}
```

**难度**：⭐  
**标签**：矩阵、基础

**题目描述**：  
实现矩阵乘法。

**参考答案概要**：  
```cpp
using Matrix = vector<vector<long long>>;

Matrix mul(Matrix &A, Matrix &B, long long MOD) {
    int n = A.size(), m = B[0].size(), p = B.size();
    Matrix C(n, vector<long long>(m, 0));
    for(int i = 0; i < n; i++)
        for(int k = 0; k < p; k++)
            for(int j = 0; j < m; j++)
                C[i][j] = (C[i][j] + A[i][k] * B[k][j]) % MOD;
    return C;
}
```

```cpp
Matrix power(Matrix A, long long n, long long MOD) {
    int sz = A.size();
    Matrix res(sz, vector<long long>(sz, 0));
    for(int i = 0; i < sz; i++) res[i][i] = 1;
    while(n) {
        if(n & 1) res = mul(res, A, MOD);
        A = mul(A, A, MOD);

    }
    return res;
}
```

```cpp
long long qpow(long long a, long long b, long long p) {
    long long res = 1;
    a %= p;
    while(b) {
        if(b & 1) res = res * a % p;
        a = a * a % p;
        b >>= 1;
    }
    return res;
}
```

### 扩展欧几里得
```cpp
int exgcd(int a, int b, int &x, int &y) {
    if(b == 0) { x = 1; y = 0; return a; }
    int d = exgcd(b, a % b, y, x);
    y -= a / b * x;
    return d;
}

// 求逆元
int inv(int a, int p) {
    int x, y;
    int d = exgcd(a, p, x, y);
    if(d != 1) return -1;  // 逆元不存在
    return (x % p + p) % p;
}
```

```cpp
void sieve(int n) {
    isPrime.resize(n+1, true);
    isPrime[0] = isPrime[1] = false;
    for(int i = 2; i <= n; i++) {
        if(isPrime[i]) primes.push_back(i);
        for(int p : primes) {
            if(1LL * i * p > n) break;
            isPrime[i * p] = false;

        }
    }
}
```

### 矩阵结构体
```cpp
struct Matrix {
    int n, m;
    vector<vector<long long>> a;
    Matrix(int n=0, int m=0): n(n), m(m), a(n, vector<long long>(m)) {}
    
    Matrix operator*(const Matrix &o) const {
        Matrix res(n, o.m);
        for(int i = 0; i < n; i++)
            for(int k = 0; k < m; k++)
                for(int j = 0; j < o.m; j++)
                    res.a[i][j] = (res.a[i][j] + a[i][k] * o.a[k][j]) % MOD;
        return res;
    }
    
    Matrix power(long long k) {
        Matrix res(n, n), x = *this;
        for(int i = 0; i < n; i++) res.a[i][i] = 1;
        while(k) {
            if(k & 1) res = res * x;
            x = x * x;
            k >>= 1;
        }
        return res;
    }
};
```

```cpp
// 莫比乌斯函数
vector<int> mu;
void mobius(int n) {
    mu.resize(n+1);
    vector<int> primes;
    vector<bool> isComp(n+1);
    mu[1] = 1;
    for(int i = 2; i <= n; i++) {
        if(!isComp[i]) { primes.push_back(i); mu[i] = -1; }
        for(int p : primes) {
            if(1LL * i * p > n) break;
            isComp[i * p] = true;
            if(i % p == 0) { mu[i * p] = 0; break; }
            mu[i * p] = -mu[i];
        }
    }
}

// 欧拉函数
vector<int> phi;
void eulerTotient(int n) {
    phi.resize(n+1);
    iota(phi.begin(), phi.end(), 0);
    for(int i = 2; i <= n; i++)
        if(phi[i] == i)
            for(int j = i; j <= n; j += i)
                phi[j] -= phi[j] / i;
}
```
