#### 题目 5.1.1：KMP模板题
**难度**：⭐  
**标签**：KMP、字符串匹配

**题目描述**：  
给定文本串 S 和模式串 P，找出 P 在 S 中所有出现的位置。

**约束**：  
|S| ≤ 10⁶, |P| ≤ 10⁵

**参考答案概要**：  
```cpp
// 计算next数组（前缀函数）
vector<int> buildNext(string &p) {
    int m = p.size();
    vector<int> nxt(m, 0);
    for(int i = 1, j = 0; i < m; i++) {
        while(j > 0 && p[i] != p[j]) j = nxt[j-1];
        if(p[i] == p[j]) j++;
        nxt[i] = j;
    }
    return nxt;
}

// KMP匹配
void kmp(string &s, string &p) {
    vector<int> nxt = buildNext(p);
    int n = s.size(), m = p.size();
    for(int i = 0, j = 0; i < n; i++) {
        while(j > 0 && s[i] != p[j]) j = nxt[j-1];
        if(s[i] == p[j]) j++;
        if(j == m) {
            cout << i - m + 1 << endl;  // 匹配位置
            j = nxt[j-1];
        }
    }
}
```

```cpp
vector<int> zFunction(string &s) {
    int n = s.size();
    vector<int> z(n);
    for(int i = 1, l = 0, r = 0; i < n; i++) {
        if(i <= r) z[i] = min(r - i + 1, z[i - l]);
        while(i + z[i] < n && s[z[i]] == s[i + z[i]]) z[i]++;
        if(i + z[i] - 1 > r) l = i, r = i + z[i] - 1;
    }
    return z;
}
```

```cpp
vector<int> manacher(string &s) {
    // 插入分隔符处理奇偶
    string t = "#";
    for(char c : s) t += c, t += "#";
    int n = t.size();
    vector<int> d(n);
    for(int i = 0, l = 0, r = -1; i < n; i++) {
        int k = (i > r) ? 1 : min(d[l + r - i], r - i + 1);
        while(i - k >= 0 && i + k < n && t[i-k] == t[i+k]) k++;
        d[i] = k--;
        if(i + k > r) l = i - k, r = i + k;
    }
    return d;
}
```

#### 题目 5.2.1：后缀数组模板
**难度**：⭐⭐  
**标签**：后缀数组、SA

**题目描述**：  
构建字符串的后缀数组和LCP数组。

**约束**：  
n ≤ 10⁶

**参考答案概要**：  
```cpp
// 倍增法构建后缀数组
void buildSA(string &s) {
    int n = s.size();
    vector<int> sa(n), rk(n), tmp(n);
    for(int i = 0; i < n; i++) sa[i] = i, rk[i] = s[i];
    for(int k = 1; k < n; k <<= 1) {
        auto cmp = [&](int i, int j) {
            if(rk[i] != rk[j]) return rk[i] < rk[j];
            int ri = i + k < n ? rk[i+k] : -1;
            int rj = j + k < n ? rk[j+k] : -1;
            return ri < rj;
        };
        sort(sa.begin(), sa.end(), cmp);
        tmp[sa[0]] = 0;
        for(int i = 1; i < n; i++)
            tmp[sa[i]] = tmp[sa[i-1]] + (cmp(sa[i-1], sa[i]) ? 1 : 0);
        rk = tmp;
    }
}

// 计算LCP数组（Height数组）
void buildLCP(string &s, vector<int> &sa) {
    int n = s.size();
    vector<int> rk(n), lcp(n-1);
    for(int i = 0; i < n; i++) rk[sa[i]] = i;
    for(int i = 0, h = 0; i < n; i++) {
        if(rk[i] < n-1) {
            int j = sa[rk[i]+1];
            while(i+h < n && j+h < n && s[i+h] == s[j+h]) h++;
            lcp[rk[i]] = h;
            if(h > 0) h--;
        }
    }
}
```

```cpp
// 用ST表预处理LCP数组
void buildST(vector<int> &lcp) {
    int n = lcp.size();
    int LOG = 20;
    vector<vector<int>> st(LOG, vector<int>(n));
    st[0] = lcp;
    for(int j = 1; j < LOG; j++)
        for(int i = 0; i + (1<<j) <= n; i++)
            st[j][i] = min(st[j-1][i], st[j-1][i+(1<<(j-1))]);
}

// 查询 suffix[i] 和 suffix[j] 的LCP
int getLCP(int i, int j, vector<int> &rk) {
    if(i == j) return n - i;
    int ri = rk[i], rj = rk[j];
    if(ri > rj) swap(ri, rj);
    return rmq(ri, rj-1);  // 查询lcp[ri..rj-1]的最小值
}
```

**难度**：⭐  
**标签**：Trie、字典树

**题目描述**：  
实现Trie，支持插入字符串和查询字符串是否存在。

**参考答案概要**：  
```cpp
struct TrieNode {
    TrieNode* child[26];
    bool isEnd;
    TrieNode() {
        memset(child, 0, sizeof(child));
        isEnd = false;
    }
};

class Trie {
    TrieNode* root;
public:
    Trie() { root = new TrieNode(); }
    
    void insert(string &word) {
        TrieNode* cur = root;
        for(char c : word) {
            int idx = c - 'a';
            if(!cur->child[idx]) cur->child[idx] = new TrieNode();
            cur = cur->child[idx];
        }
        cur->isEnd = true;
    }
    
    bool search(string &word) {
        TrieNode* cur = root;
        for(char c : word) {
            int idx = c - 'a';
            if(!cur->child[idx]) return false;
            cur = cur->child[idx];
        }
        return cur->isEnd;
    }
};
```

```cpp
// 可持久化Trie查询区间最大异或
int query(int left, int right, int x) {
    TrieNode *L = root[left-1], *R = root[right];
    int res = 0;
    for(int i = 30; i >= 0; i--) {
        int b = (x >> i) & 1;
        // 优先走与b不同的分支
        if(R->child[b^1] && R->child[b^1]->cnt > L->child[b^1]->cnt) {

            L = L->child[b^1]; R = R->child[b^1];
        } else {
            L = L->child[b]; R = R->child[b];
        }
    }
    return res;
}
```

### 入门题（4题）

#### 题目 5.4.1：AC自动机模板
**难度**：⭐⭐  
**标签**：AC自动机、多模式匹配

**题目描述**：  
给定多个模式串和一个文本串，统计每个模式串在文本串中的出现次数。

**参考答案概要**：  
```cpp
struct ACNode {
    int next[26], fail;
    int cnt;  // 以该节点结尾的模式串数量
    ACNode() {
        memset(next, 0, sizeof(next));
        fail = 0; cnt = 0;
    }
};

struct ACAutomaton {
    vector<ACNode> nodes;
    ACAutomaton() { nodes.emplace_back(); }
    
    void insert(string &s) {
        int p = 0;
        for(char c : s) {
            int idx = c - 'a';
            if(!nodes[p].next[idx]) {
                nodes[p].next[idx] = nodes.size();
                nodes.emplace_back();
            }
            p = nodes[p].next[idx];
        }
        nodes[p].cnt++;
    }
    
    void build() {
        queue<int> q;
        for(int i = 0; i < 26; i++)
            if(nodes[0].next[i]) q.push(nodes[0].next[i]);
        while(!q.empty()) {
            int u = q.front(); q.pop();
            for(int i = 0; i < 26; i++) {
                if(nodes[u].next[i]) {
                    nodes[nodes[u].next[i]].fail = 
                        nodes[nodes[u].fail].next[i];
                    q.push(nodes[u].next[i]);
                } else {
                    nodes[u].next[i] = nodes[nodes[u].fail].next[i];
                }
            }
        }
    }
    
    int query(string &t) {
        int p = 0, res = 0;
        for(char c : t) {
            p = nodes[p].next[c-'a'];
            for(int j = p; j && nodes[j].cnt != -1; j = nodes[j].fail) {
                res += nodes[j].cnt;
                nodes[j].cnt = -1;  // 标记已访问
            }
        }
        return res;
    }
};
```

### KMP完整模板
```cpp
class KMP {
    string pattern;
    vector<int> nxt;
public:
    KMP(string &p) : pattern(p) {
        int m = p.size();
        nxt.resize(m);
        for(int i = 1, j = 0; i < m; i++) {
            while(j > 0 && p[i] != p[j]) j = nxt[j-1];
            if(p[i] == p[j]) j++;
            nxt[i] = j;
        }
    }
    
    vector<int> search(string &text) {
        vector<int> res;
        int n = text.size(), m = pattern.size();
        for(int i = 0, j = 0; i < n; i++) {
            while(j > 0 && text[i] != pattern[j]) j = nxt[j-1];
            if(text[i] == pattern[j]) j++;
            if(j == m) {
                res.push_back(i - m + 1);
                j = nxt[j-1];
            }
        }
        return res;
    }
};
```

### 后缀数组（SA-IS简化版）
```cpp
// 倍增法 SA，复杂度 O(n log n)
vector<int> buildSA(string &s) {
    int n = s.size();
    vector<int> sa(n), rk(n), tmp(n);
    iota(sa.begin(), sa.end(), 0);
    for(int i = 0; i < n; i++) rk[i] = s[i];
    
    for(int k = 1; ; k <<= 1) {
        auto cmp = [&](int i, int j) {
            if(rk[i] != rk[j]) return rk[i] < rk[j];
            int ri = i + k < n ? rk[i+k] : -1;
            int rj = j + k < n ? rk[j+k] : -1;
            return ri < rj;
        };
        sort(sa.begin(), sa.end(), cmp);
        tmp[sa[0]] = 0;
        for(int i = 1; i < n; i++)
            tmp[sa[i]] = tmp[sa[i-1]] + (cmp(sa[i-1], sa[i]) ? 1 : 0);
        rk.swap(tmp);
        if(rk[sa[n-1]] == n-1) break;
    }
    return sa;
}
```

### 后缀自动机（SAM）
```cpp
struct SAM {
    struct State {
        int len, link;
        int next[26];
        State() { len = 0; link = -1; memset(next, -1, sizeof(next)); }
    };
    vector<State> st;
    int last;
    SAM(int maxlen) { st.reserve(2*maxlen); st.push_back(State()); last = 0; }
    
    void extend(char c) {
        int cur = st.size();
        st.push_back(State());
        st[cur].len = st[last].len + 1;
        int p = last;
        while(p != -1 && st[p].next[c-'a'] == -1) {
            st[p].next[c-'a'] = cur;
            p = st[p].link;
        }
        if(p == -1) {
            st[cur].link = 0;
        } else {
            int q = st[p].next[c-'a'];
            if(st[p].len + 1 == st[q].len) {
                st[cur].link = q;
            } else {
                int clone = st.size();
                st.push_back(st[q]);
                st[clone].len = st[p].len + 1;
                while(p != -1 && st[p].next[c-'a'] == q) {
                    st[p].next[c-'a'] = clone;
                    p = st[p].link;
                }
                st[q].link = st[cur].link = clone;
            }
        }
        last = cur;
    }
};
```

### 字符串哈希
```cpp
struct StringHash {
    using ull = unsigned long long;
    static const ull BASE = 13131;
    vector<ull> h, p;
    StringHash(string &s) {
        int n = s.size();
        h.resize(n+1);
        p.resize(n+1);
        p[0] = 1;
        for(int i = 0; i < n; i++) {
            h[i+1] = h[i] * BASE + s[i];
            p[i+1] = p[i] * BASE;
        }
    }
    ull get(int l, int r) { // [l, r)
        return h[r] - h[l] * p[r-l];
    }
};
```
