## 📊 图的基本表示

### 邻接矩阵 vs 邻接表

| 特性 | 邻接矩阵 | 邻接表 |
|-----|---------|--------|
| 空间 | O(V²) | O(V+E) |
| 加边 | O(1) | O(1) |
| 删边 | O(1) | O(E) |
| 查边 | O(1) | O(V) |
| 遍历邻居 | O(V) | O(deg(v)) |
| 适用 | 稠密图 | 稀疏图 |

```cpp
// 邻接矩阵
vector<vector<int>> adjMatrix(n, vector<int>(n, INF));
adjMatrix[u][v] = w;

// 邻接表（带权）
vector<vector<pair<int, int>>> adj(n);
adj[u].push_back({v, w});

// 邻接表（无权）
vector<vector<int>> adj(n);
adj[u].push_back(v);
```

## 🔍 图遍历

### BFS（广度优先搜索）

```cpp
vector<int> bfs(int start, vector<vector<int>>& adj) {
    int n = adj.size();
    vector<bool> visited(n, false);
    vector<int> dist(n, -1);  // 距离
    queue<int> q;

    visited[start] = true;
    dist[start] = 0;
    q.push(start);

    while (!q.empty()) {
        int u = q.front(); q.pop();

        for (int v : adj[u]) {
            if (!visited[v]) {
                visited[v] = true;
                dist[v] = dist[u] + 1;
                q.push(v);
            }
        }
    }
    return dist;
}

// BFS找最短路径（无权图）
vector<int> shortestPath(int start, int end, vector<vector<int>>& adj) {
    int n = adj.size();
    vector<int> parent(n, -1);
    queue<int> q;
    q.push(start);
    parent[start] = start;

    while (!q.empty()) {
        int u = q.front(); q.pop();
        if (u == end) break;

        for (int v : adj[u]) {
            if (parent[v] == -1) {
                parent[v] = u;
                q.push(v);
            }
        }
    }

    // 重建路径
    vector<int> path;
    if (parent[end] == -1) return path;
    for (int v = end; v != start; v = parent[v])
        path.push_back(v);
    path.push_back(start);
    reverse(path.begin(), path.end());
    return path;
}
```

### DFS（深度优先搜索）

```cpp
vector<bool> visited;

void dfs(int u, vector<vector<int>>& adj) {
    visited[u] = true;
    // 处理节点u
    for (int v : adj[u]) {
        if (!visited[v]) {
            dfs(v, adj);
        }
    }
}

// 非递归DFS
void dfsIterative(int start, vector<vector<int>>& adj) {
    stack<int> st;
    st.push(start);

    while (!st.empty()) {
        int u = st.top(); st.pop();
        if (visited[u]) continue;
        visited[u] = true;

        for (int v : adj[u]) {
            if (!visited[v]) {
                st.push(v);
            }
        }
    }
}
```

## 🛤️ 最短路径

### Dijkstra 算法

```cpp
// 优先队列实现 O(E log V)
vector<int> dijkstra(int start, vector<vector<pair<int,int>>>& adj) {
    int n = adj.size();
    vector<int> dist(n, INF);
    dist[start] = 0;

    priority_queue<pair<int,int>, vector<pair<int,int>>, greater<>> pq;
    pq.push({0, start});

    while (!pq.empty()) {
        auto [d, u] = pq.top(); pq.pop();
        if (d > dist[u]) continue;  // 过期节点

        for (auto [v, w] : adj[u]) {
            if (dist[u] + w < dist[v]) {
                dist[v] = dist[u] + w;
                pq.push({dist[v], v});
            }
        }
    }
    return dist;
}
```

**适用条件**：非负权边

### Bellman-Ford 算法

```cpp
// O(VE)，可检测负环
vector<int> bellmanFord(int start, int n, vector<tuple<int,int,int>>& edges) {
    vector<int> dist(n, INF);
    dist[start] = 0;

    // 松弛 n-1 次
    for (int i = 0; i < n - 1; i++) {
        for (auto [u, v, w] : edges) {
            if (dist[u] != INF && dist[u] + w < dist[v]) {
                dist[v] = dist[u] + w;
            }
        }
    }

    // 检测负环
    for (auto [u, v, w] : edges) {
        if (dist[u] != INF && dist[u] + w < dist[v]) {
            // 存在负环
            return {};
        }
    }

    return dist;
}
```

**适用条件**：含负权边，可检测负环

### SPFA 算法

```cpp
// 队列优化的 Bellman-Ford
vector<int> spfa(int start, vector<vector<pair<int,int>>>& adj) {
    int n = adj.size();
    vector<int> dist(n, INF);
    vector<bool> inQueue(n, false);
    vector<int> count(n, 0);  // 入队次数
    queue<int> q;

    dist[start] = 0;
    q.push(start);
    inQueue[start] = true;

    while (!q.empty()) {
        int u = q.front(); q.pop();
        inQueue[u] = false;

        for (auto [v, w] : adj[u]) {
            if (dist[u] + w < dist[v]) {
                dist[v] = dist[u] + w;
                if (!inQueue[v]) {
                    q.push(v);
                    inQueue[v] = true;
                    if (++count[v] > n) {
                        // 存在负环
                        return {};
                    }
                }
            }
        }
    }
    return dist;
}
```

### Floyd-Warshall 算法

```cpp
// O(V³)，全源最短路
void floydWarshall(vector<vector<int>>& dist) {
    int n = dist.size();

    for (int k = 0; k < n; k++)
        for (int i = 0; i < n; i++)
            for (int j = 0; j < n; j++)
                if (dist[i][k] != INF && dist[k][j] != INF)
                    dist[i][j] = min(dist[i][j], dist[i][k] + dist[k][j]);
}
```

## 🌲 最小生成树

### Kruskal 算法

```cpp
struct Edge {
    int u, v, w;
    bool operator<(const Edge& other) const {
        return w < other.w;
    }
};

// 并查集
class UnionFind {
    vector<int> parent, rank_;
public:
    UnionFind(int n) {
        parent.resize(n);
        rank_.resize(n);
        for (int i = 0; i < n; i++) parent[i] = i;
    }
    int find(int x) {
        return parent[x] == x ? x : parent[x] = find(parent[x]);
    }
    bool unite(int x, int y) {
        x = find(x), y = find(y);
        if (x == y) return false;
        if (rank_[x] < rank_[y]) swap(x, y);
        parent[y] = x;
        if (rank_[x] == rank_[y]) rank_[x]++;
        return true;
    }
};

int kruskal(int n, vector<Edge>& edges) {
    sort(edges.begin(), edges.end());
    UnionFind uf(n);
    int mstWeight = 0, edgesUsed = 0;

    for (auto& e : edges) {
        if (uf.unite(e.u, e.v)) {
            mstWeight += e.w;
            if (++edgesUsed == n - 1) break;
        }
    }
    return edgesUsed == n - 1 ? mstWeight : -1;
}
```

### Prim 算法

```cpp
int prim(int n, vector<vector<pair<int,int>>>& adj) {
    vector<bool> visited(n, false);
    vector<int> minEdge(n, INF);
    minEdge[0] = 0;

    priority_queue<pair<int,int>, vector<pair<int,int>>, greater<>> pq;
    pq.push({0, 0});

    int mstWeight = 0;

    while (!pq.empty()) {
        auto [w, u] = pq.top(); pq.pop();
        if (visited[u]) continue;

        visited[u] = true;
        mstWeight += w;

        for (auto [v, weight] : adj[u]) {
            if (!visited[v] && weight < minEdge[v]) {
                minEdge[v] = weight;
                pq.push({weight, v});
            }
        }
    }

    return mstWeight;
}
```

## 🔄 拓扑排序

```cpp
// Kahn算法
vector<int> topologicalSort(int n, vector<vector<int>>& adj) {
    vector<int> indegree(n, 0);
    for (int u = 0; u < n; u++)
        for (int v : adj[u])
            indegree[v]++;

    queue<int> q;
    for (int i = 0; i < n; i++)
        if (indegree[i] == 0) q.push(i);

    vector<int> result;
    while (!q.empty()) {
        int u = q.front(); q.pop();
        result.push_back(u);

        for (int v : adj[u]) {
            if (--indegree[v] == 0) q.push(v);
        }
    }

    // 有环则结果不完整
    return result.size() == n ? result : vector<int>{};
}

// DFS版本
vector<int> topoOrder;
bool dfsTopo(int u, vector<vector<int>>& adj, vector<int>& state) {
    state[u] = 1;  // 访问中
    for (int v : adj[u]) {
        if (state[v] == 1) return false;  // 有环
        if (state[v] == 0 && !dfsTopo(v, adj, state)) return false;
    }
    state[u] = 2;  // 已访问
    topoOrder.push_back(u);
    return true;
}
```

## 🔗 连通分量

### 强连通分量（Tarjan）

```cpp
vector<vector<int>> adj;
vector<int> dfn, low, stk;
vector<bool> inStk;
int timestamp, sccCount;
vector<int> sccId;

void tarjan(int u) {
    dfn[u] = low[u] = ++timestamp;
    stk.push_back(u);
    inStk[u] = true;

    for (int v : adj[u]) {
        if (!dfn[v]) {
            tarjan(v);
            low[u] = min(low[u], low[v]);
        } else if (inStk[v]) {
            low[u] = min(low[u], dfn[v]);
        }
    }

    if (dfn[u] == low[u]) {
        sccCount++;
        int v;
        do {
            v = stk.back(); stk.pop_back();
            inStk[v] = false;
            sccId[v] = sccCount;
        } while (v != u);
    }
}

// 使用
void findSCC(int n) {
    dfn.resize(n);
    low.resize(n);
    inStk.resize(n);
    sccId.resize(n);
    timestamp = sccCount = 0;

    for (int i = 0; i < n; i++)
        if (!dfn[i]) tarjan(i);
}
```

### 割点与桥

```cpp
vector<int> dfn, low;
int timestamp;
vector<bool> isCut;
vector<pair<int,int>> bridges;

void findCutAndBridge(int u, int parent, vector<vector<int>>& adj) {
    dfn[u] = low[u] = ++timestamp;
    int children = 0;

    for (int v : adj[u]) {
        if (!dfn[v]) {
            children++;
            findCutAndBridge(v, u, adj);
            low[u] = min(low[u], low[v]);

            // 割点
            if (parent != -1 && low[v] >= dfn[u])
                isCut[u] = true;

            // 桥
            if (low[v] > dfn[u])
                bridges.push_back({u, v});
        } else if (v != parent) {
            low[u] = min(low[u], dfn[v]);
        }
    }

    // 根节点是割点的条件
    if (parent == -1 && children > 1)
        isCut[u] = true;
}
```

## 🌊 网络流

### Dinic 算法

```cpp
struct Edge {
    int to, rev, cap;
};

class Dinic {
    vector<vector<Edge>> graph;
    vector<int> level, iter;

public:
    Dinic(int n) : graph(n), level(n), iter(n) {}

    void addEdge(int from, int to, int cap) {
        graph[from].push_back({to, (int)graph[to].size(), cap});
        graph[to].push_back({from, (int)graph[from].size()-1, 0});
    }

    bool bfs(int s, int t) {
        fill(level.begin(), level.end(), -1);
        queue<int> q;
        level[s] = 0;
        q.push(s);
        while (!q.empty()) {
            int v = q.front(); q.pop();
            for (auto& e : graph[v]) {
                if (e.cap > 0 && level[e.to] < 0) {
                    level[e.to] = level[v] + 1;
                    q.push(e.to);
                }
            }
        }
        return level[t] >= 0;
    }

    int dfs(int v, int t, int f) {
        if (v == t) return f;
        for (int& i = iter[v]; i < graph[v].size(); i++) {
            Edge& e = graph[v][i];
            if (e.cap > 0 && level[v] < level[e.to]) {
                int d = dfs(e.to, t, min(f, e.cap));
                if (d > 0) {
                    e.cap -= d;
                    graph[e.to][e.rev].cap += d;
                    return d;
                }
            }
        }
        return 0;
    }

    int maxFlow(int s, int t) {
        int flow = 0;
        while (bfs(s, t)) {
            fill(iter.begin(), iter.end(), 0);
            int f;
            while ((f = dfs(s, t, INF)) > 0) flow += f;
        }
        return flow;
    }
};
```

## 📌 算法选择指南

| 问题 | 推荐算法 | 复杂度 | 注意事项 |
|-----|---------|--------|---------|
| 无权图最短路 | BFS | O(V+E) | 适用于无权图 |
| 非负权最短路 | Dijkstra | O(E log V) | 不能有负权 |
| 含负权最短路 | Bellman-Ford/SPFA | O(VE) | 可检测负环 |
| 全源最短路 | Floyd | O(V³) | V较小时使用 |
| 稠密图MST | Prim | O(V²) | 邻接矩阵 |
| 稀疏图MST | Kruskal | O(E log E) | 边排序 |
| 拓扑排序 | Kahn/DFS | O(V+E) | 检测DAG |
| 强连通分量 | Tarjan/Kosaraju | O(V+E) | SCC分解 |
| 最大流 | Dinic | O(V²E) | 单位网络更快 |

### 判环

```cpp
// 无向图判环
bool hasCycleUndirected(int n, vector<vector<int>>& adj) {
    vector<bool> visited(n, false);
    function<bool(int, int)> dfs = [&](int u, int parent) {
        visited[u] = true;
        for (int v : adj[u]) {
            if (!visited[v]) {
                if (dfs(v, u)) return true;
            } else if (v != parent) {
                return true;
            }
        }
        return false;
    };

    for (int i = 0; i < n; i++)
        if (!visited[i] && dfs(i, -1)) return true;
    return false;
}

// 有向图判环（拓扑排序）
bool hasCycleDirected(int n, vector<vector<int>>& adj) {
    return topologicalSort(n, adj).empty();
}
```
