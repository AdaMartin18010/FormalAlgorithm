## 🎯 场景选择决策树

```
数据规模?
├── 小规模 (n < 50)
│   └── 插入排序 ✅ 常数小，实现简单
│
├── 中等规模
│   └── 数据基本有序?
│       ├── 是 → 插入排序 ✅

│
└── 大规模
    ├── 内存受限?
    │   ├── 是 → 堆排序 ✅
    │   └── 否 → 快速排序 ✅
    │
    ├── 需要稳定性?
    │   ├── 是 → 归并排序 ✅
    │   └── 否 → 快速排序 ✅
    │
    └── 数据类型?
        ├── 整数且范围小 → 计数/基数排序 ✅
        ├── 数据分布均匀 → 桶排序 ✅
        └── 链表数据 → 归并排序 ✅
```

## 📝 核心实现代码

### 1. 快速排序（Quick Sort）

```cpp
// 标准快速排序
void quickSort(vector<int>& arr, int left, int right) {
    if (left >= right) return;

    // 三数取中优化
    int mid = left + (right - left) / 2;
    if (arr[mid] < arr[left]) swap(arr[left], arr[mid]);
    if (arr[right] < arr[left]) swap(arr[left], arr[right]);
    if (arr[mid] < arr[right]) swap(arr[mid], arr[right]);

    int pivot = arr[right];
    int i = left - 1;

    for (int j = left; j < right; j++) {
        if (arr[j] <= pivot) {
            swap(arr[++i], arr[j]);
        }
    }
    swap(arr[i + 1], arr[right]);

    quickSort(arr, left, i);
    quickSort(arr, i + 2, right);
}

// 三路快排（处理大量重复元素）
void quickSort3Way(vector<int>& arr, int left, int right) {
    if (left >= right) return;

    int pivot = arr[left];
    int lt = left, gt = right;
    int i = left + 1;

    while (i <= gt) {
        if (arr[i] < pivot) swap(arr[lt++], arr[i++]);
        else if (arr[i] > pivot) swap(arr[i], arr[gt--]);
        else i++;
    }

    quickSort3Way(arr, left, lt - 1);
    quickSort3Way(arr, gt + 1, right);
}
```

### 2. 归并排序（Merge Sort）

```cpp
void merge(vector<int>& arr, int left, int mid, int right) {
    vector<int> temp(right - left + 1);
    int i = left, j = mid + 1, k = 0;

    while (i <= mid && j <= right) {
        if (arr[i] <= arr[j]) temp[k++] = arr[i++];
        else temp[k++] = arr[j++];
    }

    while (i <= mid) temp[k++] = arr[i++];
    while (j <= right) temp[k++] = arr[j++];

    for (i = left, k = 0; i <= right; i++, k++) {
        arr[i] = temp[k];
    }
}

void mergeSort(vector<int>& arr, int left, int right) {
    if (left >= right) return;
    int mid = left + (right - left) / 2;
    mergeSort(arr, left, mid);
    mergeSort(arr, mid + 1, right);
    merge(arr, left, mid, right);
}
```

### 3. 堆排序（Heap Sort）

```cpp
void heapify(vector<int>& arr, int n, int i) {
    int largest = i;
    int left = 2 * i + 1;
    int right = 2 * i + 2;

    if (left < n && arr[left] > arr[largest])
        largest = left;
    if (right < n && arr[right] > arr[largest])
        largest = right;

    if (largest != i) {
        swap(arr[i], arr[largest]);
        heapify(arr, n, largest);
    }
}

void heapSort(vector<int>& arr) {
    int n = arr.size();

    // 建堆
    for (int i = n / 2 - 1; i >= 0; i--)
        heapify(arr, n, i);

    // 排序
    for (int i = n - 1; i > 0; i--) {
        swap(arr[0], arr[i]);
        heapify(arr, i, 0);
    }
}
```

### 4. 计数排序（Counting Sort）

```cpp
void countingSort(vector<int>& arr) {
    if (arr.empty()) return;

    int maxVal = *max_element(arr.begin(), arr.end());
    int minVal = *min_element(arr.begin(), arr.end());
    int range = maxVal - minVal + 1;

    vector<int> count(range), output(arr.size());

    // 统计频率
    for (int num : arr) count[num - minVal]++;

    // 计算位置
    for (int i = 1; i < range; i++) count[i] += count[i-1];

    // 放置元素（从后往前保证稳定性）
    for (int i = arr.size() - 1; i >= 0; i--) {
        output[count[arr[i] - minVal] - 1] = arr[i];
        count[arr[i] - minVal]--;
    }

    arr = output;
}
```

### 5. 基数排序（Radix Sort）

```cpp
void countingSortForRadix(vector<int>& arr, int exp) {
    vector<int> output(arr.size());
    vector<int> count(10, 0);

    for (int num : arr) count[(num / exp) % 10]++;
    for (int i = 1; i < 10; i++) count[i] += count[i-1];

    for (int i = arr.size() - 1; i >= 0; i--) {
        output[count[(arr[i] / exp) % 10] - 1] = arr[i];
        count[(arr[i] / exp) % 10]--;
    }

    arr = output;
}

void radixSort(vector<int>& arr) {
    int maxVal = *max_element(arr.begin(), arr.end());

    for (int exp = 1; maxVal / exp > 0; exp *= 10)
        countingSortForRadix(arr, exp);
}
```

### 6. 插入排序（Insertion Sort）

```cpp
void insertionSort(vector<int>& arr) {
    for (int i = 1; i < arr.size(); i++) {
        int key = arr[i];
        int j = i - 1;
        while (j >= 0 && arr[j] > key) {
            arr[j + 1] = arr[j];
            j--;
        }
        arr[j + 1] = key;
    }
}

// 二分插入排序
void binaryInsertionSort(vector<int>& arr) {
    for (int i = 1; i < arr.size(); i++) {
        int key = arr[i];
        int left = 0, right = i;
        while (left < right) {
            int mid = left + (right - left) / 2;
            if (arr[mid] <= key) left = mid + 1;
            else right = mid;
        }
        for (int j = i - 1; j >= left; j--)
            arr[j + 1] = arr[j];
        arr[left] = key;
    }
}
```

```cpp
#include <algorithm>
#include <vector>

vector<int> arr = {3, 1, 4, 1, 5, 9};

// 基础排序
sort(arr.begin(), arr.end());                    // 升序
sort(arr.begin(), arr.end(), greater<int>());    // 降序

// 自定义比较
sort(arr.begin(), arr.end(), [](int a, int b) {
    return a > b;  // 降序
});

// 稳定排序
stable_sort(arr.begin(), arr.end());

// 部分排序（前k小）
partial_sort(arr.begin(), arr.begin() + 3, arr.end());
```

```python
arr = [3, 1, 4, 1, 5, 9]
```

## 🎓 面试常考问题

### Q1: 为什么快排比堆排快？

**A:** 虽然两者时间复杂度都是 O(n log n)，但：

- 快排有更好的缓存局部性
- 快排的比较次数更少
- 堆排的 swap 操作更多

### Q2: 什么时候选择归并排序？

**A:**

- 需要稳定排序
- 链表排序（无需额外空间）
- 外部排序（大数据无法放入内存）

### Q3: 如何优化快排？

**A:**

- 三数取中法选 pivot
- 随机化 pivot
- 三路划分处理重复元素
- 小数组切换插入排序

### Q4: 计数排序 vs 基数排序？

**A:**

- 计数排序：适合范围小的整数
- 基数排序：适合范围大的整数，按位处理
- 两者都是线性时间，但需要额外空间
