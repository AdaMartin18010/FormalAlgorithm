```lean
-- 依赖类型示例：长度索引向量
inductive Vec (α : Type u) : Nat → Type u
  | nil : Vec α 0
  | cons : α → Vec α n → Vec α (n+1)

-- 安全取元素（类型保证索引在范围内）
def get {α : Type u} {n : Nat} (v : Vec α n) (i : Fin n) : α :=
  match v, i with
  | .cons x _, ⟨0, _⟩ => x
  | .cons _ xs, ⟨i+1, h⟩ => get xs ⟨i, by simp at h; omega⟩

-- 类型即命题，程序即证明
theorem vec_length_cons {α : Type u} {n : Nat} (x : α) (xs : Vec α n) :
    (Vec.cons x xs).length = n + 1 := by
  simp [Vec.length]
```

```coq
(* 多态类型定义 *)
Inductive list (A : Type) : Type :=
  | nil : list A
  | cons : A -> list A -> list A.

(* 依赖类型：长度保持的映射 *)
Fixpoint map {A B : Type} (f : A -> B) (l : list A) : list B :=
  match l with
  | nil => nil B
  | cons _ x xs => cons B (f x) (map f xs)
  end.

(* 类型安全性定理 *)
Theorem map_preserves_length : forall A B (f : A -> B) (l : list A),
    length (map f l) = length l.
Proof.
  intros A B f l. induction l as [| x xs IH].
  - reflexivity.
  - simpl. rewrite IH. reflexivity.
Qed.
```

```agda
-- 相等类型（恒等类型）
data _≡_ {A : Set} (x : A) : A → Set where
  refl : x ≡ x

-- 替换原理（J规则）
J : {A : Set} {x : A} (P : (y : A) → x ≡ y → Set)
    → P x refl → {y : A} (p : x ≡ y) → P y p
J P pr refl = pr

-- 同伦类型论中的路径归纳
trans : {A : Set} {x y z : A} → x ≡ y → y ≡ z → x ≡ z
trans refl q = q
```
